61  public class AvroMeasurementStoreTest {
62  
63    private static final Logger LOG = LoggerFactory.getLogger(AvroMeasurementStoreTest.class);
64  
65    @Test
66    @SuppressFBWarnings({ "PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS", "AFBR_ABNORMAL_FINALLY_BLOCK_RETURN" })
67    public void testStore() throws IOException {
68      AvroMeasurementStore store = new AvroMeasurementStore(org.spf4j.base.Runtime.TMP_FOLDER_PATH,
69              "testMetrics", Compressor.ZSTANDARD);
70      try {
71        long mid = store.alocateMeasurements(new MeasurementsInfoImpl("test", "test", new String[]{"v1", "v2"},
72                new String[]{"t1", "t2"},
73                new Aggregation[]{Aggregation.SUM, Aggregation.LAST},
74                MeasurementType.GAUGE), 1000);
75        store.saveMeasurements(mid, 0L, 1, 2);
76  
77        store.saveMeasurements(mid, 1000L, 3, 4);
78  
79        store.saveMeasurements(mid, 2000L, 5, 6);
80  
81        store.saveMeasurements(mid, 2000L, 7, 8);
82  
83        store.saveMeasurements(mid, 3000L, 9, 10);
84  
85        store.saveMeasurements(mid, 4000L, 11, 12);
86  
87        store.flush();
88  
89        MeasurementStoreQuery query = store.query();
90        Collection<Schema> measurements = query.getMeasurements((x) -> true);
91        Schema metric = measurements.iterator().next();
92        Assert.assertEquals("test", metric.getName());
93        List<TimeSeriesRecord> results = getMetrics(query, metric, Instant.EPOCH, Instant.now());
94        Assert.assertEquals(6, results.size());
95        TimeSeriesRecord rec = results.get(0);
96        Assert.assertEquals(Instant.ofEpochMilli(0L), rec.getTimeStamp());
97        Assert.assertEquals(1L, rec.getLongValue("v1"));
98        Assert.assertEquals(2L, rec.getLongValue("v2"));
99        rec = results.get(5);
100       Assert.assertEquals(Instant.ofEpochMilli(4000L), rec.getTimeStamp());
101       Assert.assertEquals(11L, rec.getLongValue("v1"));
102       Assert.assertEquals(12L, rec.getLongValue("v2"));
103       results = getMetrics(query, metric, Instant.ofEpochMilli(2000L), Instant.ofEpochMilli(2000L));
104       Assert.assertEquals(2, results.size());
105       rec = results.get(0);
106       Assert.assertEquals(Instant.ofEpochMilli(2000L), rec.getTimeStamp());
107       Assert.assertEquals(5L, rec.getLongValue("v1"));
108       Assert.assertEquals(6L, rec.getLongValue("v2"));
109       rec = results.get(1);
110       Assert.assertEquals(Instant.ofEpochMilli(2000L), rec.getTimeStamp());
111       Assert.assertEquals(7L, rec.getLongValue("v1"));
112       Assert.assertEquals(8L, rec.getLongValue("v2"));
113 
114       List<TimeSeriesRecord> aggr = getMetrics(query, metric, Instant.EPOCH, Instant.now(), 1000);
115       Assert.assertEquals(5, aggr.size());
116       rec = aggr.get(0);
117       Assert.assertEquals(Instant.ofEpochMilli(0L), rec.getTimeStamp());
118       Assert.assertEquals(1L, rec.getLongValue("v1"));
119       Assert.assertEquals(2L, rec.getLongValue("v2"));
120       rec = aggr.get(2);
121       Assert.assertEquals(Instant.ofEpochMilli(2000L), rec.getTimeStamp());
122       Assert.assertEquals(12L, rec.getLongValue("v1"));
123       Assert.assertEquals(8L, rec.getLongValue("v2"));
124 
125       aggr = getMetrics(query, metric, Instant.EPOCH, Instant.now(), 1500);
126       Assert.assertEquals(5, aggr.size());
127 
128       aggr = getMetrics(query, metric, Instant.EPOCH, Instant.now(), 0);
129       Assert.assertEquals(6, aggr.size());
130 
131       try (AvroCloseableIterable<TimeSeriesRecord> it = query.getAggregatedMeasurementData(metric,
132               Instant.EPOCH, Instant.now(),
133               Duration.between(Instant.EPOCH, Instant.now()).getSeconds(), TimeUnit.SECONDS)) {
134         Iterator<TimeSeriesRecord> iterator = it.iterator();
135         TimeSeriesRecord next = iterator.next();
136         LOG.debug("The aggregate", next);
137         Assert.assertFalse(iterator.hasNext());
138         Assert.assertEquals(36L, next.getLongValue("v1"));
139         Assert.assertEquals(12L, next.getLongValue("v2"));
140       }
141     } finally {
142       store.close();
143       Files.delete(store.getInfoFile());
144       Files.delete(store.getDataFile());
145     }
146 
147   }

149   public static List<TimeSeriesRecord> getMetrics(final MeasurementStoreQuery query,
150           final Schema metric, final Instant from, final Instant to) throws IOException {
151     List<TimeSeriesRecord> results = new ArrayList<>();
152     try (AvroCloseableIterable<TimeSeriesRecord> data = query.getMeasurementData(metric, from, to)) {
153       for (TimeSeriesRecord rec : data) {
154         LOG.debug("data", rec);
155         results.add(rec);
156       }
157     }
158     return results;
159   }
160 
161   public static List<TimeSeriesRecord> getMetrics(final MeasurementStoreQuery query,
162           final Schema metric, final Instant from, final Instant to, final int aggMillis) throws IOException {
163     List<TimeSeriesRecord> results = new ArrayList<>();
164     try (AvroCloseableIterable<TimeSeriesRecord> data = query.getAggregatedMeasurementData(metric, from, to,
165             aggMillis, TimeUnit.MILLISECONDS)) {
166       for (TimeSeriesRecord rec : data) {
167         LOG.debug("agg", rec);
168         results.add(rec);
169       }
170     }
171     return results;
172   }
