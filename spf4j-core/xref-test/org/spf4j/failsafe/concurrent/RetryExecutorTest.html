32  package org.spf4j.failsafe.concurrent;

34  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
35  import java.util.concurrent.ExecutionException;
36  import java.util.concurrent.Future;
37  import java.util.concurrent.TimeUnit;
38  import java.util.concurrent.atomic.AtomicInteger;
39  import org.junit.Assert;
40  import org.junit.Test;
41  import org.slf4j.Logger;
42  import org.slf4j.LoggerFactory;
43  import org.spf4j.base.TimeSource;
44  import org.spf4j.concurrent.DefaultExecutor;
45  import org.spf4j.failsafe.RetryPolicy;
46  import org.spf4j.failsafe.RetryPredicate;
47  import org.spf4j.log.Level;
48  import org.spf4j.test.log.annotations.ExpectLog;
49  
50  /**
51   * @author Zoltan Farkas
52   */
53  @SuppressFBWarnings("MDM_THREAD_YIELD")
54  public class RetryExecutorTest {
55    private static final Logger LOG = LoggerFactory.getLogger(RetryExecutorTest.class);
56  
57    @Test
58    @ExpectLog(category = "org.spf4j.failsafe.concurrent.RetryExecutorTest", level = Level.DEBUG,
59            nrTimes = 3, messageRegexp = "Executing .*")
60    public void testHedgedretryExecution() throws InterruptedException, ExecutionException {
61      try (FailSafeExecutorImpl exec = new FailSafeExecutorImpl(DefaultExecutor.INSTANCE)) {
62        Future fut = exec.submit(() -> {
63          LOG.debug("Executing {}", this);
64          return Thread.currentThread().getName();
65        }, RetryPredicate.NORETRY, 2, 0, TimeUnit.MILLISECONDS);
66        Assert.assertNotNull(fut.get());
67      }
68    }
69  
70    @Test
71    @ExpectLog(category = "org.spf4j.failsafe.concurrent.RetryExecutorTest", level = Level.DEBUG,
72            nrTimes = 1, messageRegexp = "Executing .*")
73    public void testHedgedretryExecution2() throws InterruptedException, ExecutionException {
74      try (FailSafeExecutorImpl exec = new FailSafeExecutorImpl(DefaultExecutor.INSTANCE)) {
75        Future fut = exec.submit(() -> {
76          LOG.debug("Executing {}", this);
77          return Thread.currentThread().getName();
78        }, RetryPredicate.NORETRY, 2, 1000, TimeUnit.MILLISECONDS);
79        Assert.assertNotNull(fut.get());
80      }
81    }
82  
83    @Test(expected = ExecutionException.class)
84    public void testHedgedretryExecution3() throws InterruptedException, ExecutionException {
85      try (FailSafeExecutorImpl exec = new FailSafeExecutorImpl(DefaultExecutor.INSTANCE)) {
86        Future fut = exec.submit(() -> {
87          throw new RuntimeException();
88        }, RetryPredicate.NORETRY, 2, 10, TimeUnit.MILLISECONDS);
89        Assert.assertNotNull(fut.get());
90      }
91    }
92  
93    @Test(timeout = 1000)
94    public void testHedgedretryExecution4() throws InterruptedException, ExecutionException {
95      AtomicInteger ai = new AtomicInteger();
96      try (FailSafeExecutorImpl exec = new FailSafeExecutorImpl(DefaultExecutor.INSTANCE)) {
97        long nanoTime = TimeSource.nanoTime();
98        Future fut = exec.submit(() -> {
99          if (ai.getAndIncrement() == 0) {
100           Thread.sleep(1000000);
101           return "B";
102         }
103         LOG.debug("Executing {}", this);
104         return "A";
105       }, RetryPolicy.defaultPolicy().getRetryPredicate(nanoTime, nanoTime + 1000000000), 2, 10, TimeUnit.MILLISECONDS);
106       Assert.assertEquals("A", fut.get());
107     }
108   }
