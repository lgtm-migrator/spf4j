60    @Test
61    public void testStreamPiping() throws IOException {
62      test("This is a super cool, mega dupper test string for testing piping..........E", 8, false);
63      final IntMath.XorShift32 random = new IntMath.XorShift32();
64      for (int i = 0; i < 100; i++) {
65        int nrChars = Math.abs(random.nextInt() % 100000);
66        StringBuilder sb = generateTestStr(nrChars);
67        test(sb.toString(), Math.abs(random.nextInt() % 10000) + 2, false);
68      }
69      test(generateTestStr(133).toString(), 2, false);
70    }
71  
72    public static StringBuilder generateTestStr(final int nrChars) {
73      final IntMath.XorShift32 random = new IntMath.XorShift32();
74      StringBuilder sb = new StringBuilder(nrChars);
75      for (int i = 0; i < nrChars; i++) {
76        sb.append((char) Math.abs(random.nextInt() % 100) + 20);
77      }
78      return sb;
79    }
80  
81    public static void test(final String testStr, final int buffSize, final boolean buffered) throws IOException {
82      final PipedOutputStream pos = new PipedOutputStream(buffSize);
83      final InputStream pis;
84      if (buffered) {
85        pis = new MemorizingBufferedInputStream(pos.getInputStream());
86      } else {
87        pis = pos.getInputStream();
88      }
89  
90      DefaultExecutor.INSTANCE.execute(new AbstractRunnable() {
91  
92        @Override
93        public void doRun() throws Exception {
94          try (OutputStream os = pos) {
95            final byte[] utf8 = Strings.toUtf8(testStr);
96            os.write(utf8[0]);
97            os.write(utf8, 1,  utf8.length - 1);
98          }
99        }
100     });
101     StringBuilder sb = new StringBuilder();
102     try (InputStream is = pis) {
103       byte[] buffer = new byte[1024];
104       int read;
105       while ((read = is.read(buffer)) > 0) {
106         sb.append(Strings.fromUtf8(buffer, 0, read));
107       }
108     }
109     Assert.assertEquals("buffSize = " + buffSize + ", buffered = " + buffered, testStr, sb.toString());
110   }

112   @Test
113   public void testNoReaderBehaviour() throws IOException, InterruptedException,
114           ExecutionException, TimeoutException {
115     PipedOutputStream os = new PipedOutputStream(1024);
116     Future<Integer> nr;
117     int j = 0;
118     try (PipedOutputStream pos = os) {
119       pos.write(123);
120       LOG.debug("write {}", j);
121       j++;
122       nr = DefaultExecutor.instance().submit(() -> {
123         try (InputStream is = pos.getInputStream()) {
124           int i = 0;
125           int val;
126           while ((val = is.read()) >= 0) {
127             Assert.assertEquals(123, val);
128             LOG.debug("read {}", i);
129             i++;
130           }
131           return i;
132         }
133       });
134       for (; j < 2000; j++) {
135         pos.write(123);
136         LOG.debug("write {}", j);
137       }
138     }
139     LOG.debug("os = {}", os);
140     Assert.assertEquals(j, nr.get(3, TimeUnit.SECONDS).intValue());
141   }

170   @Test
171   public void testCloseWithReason() throws IOException {
172     try (PipedOutputStream pos = new PipedOutputStream(10)) {
173       pos.write(123);
174       IOException ex = new IOException();
175       pos.close(ex);
176       try {
177         pos.write(123);
178         Assert.fail();
179       } catch (IOException ex2) {
180         Assert.assertEquals(ex, ex2.getCause());
181       }
182       try (InputStream is = pos.getInputStream()) {
183         Assert.assertEquals(123, is.read());
184       }
185     }
186   }
