67  /**
68   * Second generation Time-Series database format. The linked list structure from first generation is dropped to reduce
69   * write overhead.
70   *
71   *
72   * @author zoly
73   */
74  public final class TSDBWriter implements Closeable, Flushable {
75  
76    public static final Schema FILE_RECORD_SCHEMA
77            = Schema.createUnion(Arrays.asList(TableDef.SCHEMA$, DataBlock.SCHEMA$));
78  
79    static final byte[] MAGIC = Strings.toUtf8("TSDB2");
80  
81    private final File file;
82    private final FileChannel channel;
83    private final BinaryEncoder encoder;
84    private final Header header;
85    private final SpecificDatumWriter<Object> recordWriter = new SpecificDatumWriter<>(FILE_RECORD_SCHEMA);
86    private final DataBlock writeBlock;
87    private final int maxRowsPerBlock;
88    private final RandomAccessFile raf;
89  
90    private final ByteArrayBuilder bab;
91  
92    @CreatesObligation
93    public TSDBWriter(final File file, final int maxRowsPerBlock,
94            final String description, final boolean append) throws IOException {
95      this.file = file;
96      this.maxRowsPerBlock = maxRowsPerBlock;
97      this.writeBlock = new DataBlock(System.currentTimeMillis(), new ArrayList<DataRow>(maxRowsPerBlock));
98      raf = new RandomAccessFile(file, "rw");
99      bab = new ByteArrayBuilder(32768, ArraySuppliers.Bytes.JAVA_NEW);
100     encoder = EncoderFactory.get().directBinaryEncoder(bab, null);
101     channel = raf.getChannel();
102     channel.lock();
103     if (!append) {
104       raf.setLength(0);
105       channel.force(true);
106     }
107     if (raf.length() <= 0) {
108       // new file or overwite, will write header;
109       bab.write(MAGIC);
110       toOutputStream(0, bab);
111       header = Header.newBuilder()
112               .setContentSchema(FILE_RECORD_SCHEMA.toString())
113               .setDescription(description)
114               .build();
115       SpecificDatumWriter<Header> headerWriter = new SpecificDatumWriter<>(Header.SCHEMA$);
116       headerWriter.write(header, encoder);
117       encoder.flush();
118       byte[] buffer = bab.getBuffer();
119       final int size = bab.size();
120       toByteArray(size, buffer, MAGIC.length);
121       raf.write(buffer, 0, size);
122       channel.force(true);
123     } else {
124       if (description != null) {
125         throw new IllegalArgumentException("Providing description when appending is not allowed for " + file);
126       }
127 
128       try (BufferedInputStream bis = new BufferedInputStream(Files.newInputStream(file.toPath()));
129               DataInputStream dis = new DataInputStream(bis)) {
130         validateType(dis);
131         long size = dis.readLong();
132         SpecificDatumReader<Header> reader = new SpecificDatumReader<>(Header.getClassSchema());
133         BinaryDecoder directBinaryDecoder = DecoderFactory.get().directBinaryDecoder(dis, null);
134         header = reader.read(null, directBinaryDecoder);
135         raf.seek(size);
136       }
137     }
138   }
139 
140   static void validateType(final InputStream dis) throws IOException {
141     byte[] readMagic = new byte[MAGIC.length];
142     ByteStreams.readFully(dis, readMagic);
143     if (!Arrays.equals(MAGIC, readMagic)) {
144       throw new IOException("wrong file type, magic is " + Arrays.toString(readMagic));
145     }
146   }
147 
148   public synchronized long writeTableDef(final TableDef tableDef) throws IOException {
149     final long position = raf.getFilePointer();
150     bab.reset();
151     tableDef.setId(position);
152     recordWriter.write(tableDef, encoder);
153     encoder.flush();
154     raf.write(bab.getBuffer(), 0, bab.size());
155     return position;
156   }
157 
158   public synchronized void writeDataRow(final long tableId, final long timestamp, final long... data)
159           throws IOException {
160     List<DataRow> blockValues = this.writeBlock.getValues();
161     if (blockValues.size() >= this.maxRowsPerBlock) {
162       flush();
163     }
164     long baseTs = writeBlock.getBaseTimestamp();
165     DataRow row = new DataRow((int) (timestamp - baseTs), tableId, Longs.asList(data));
166     blockValues.add(row);
167   }
168 
169   @Override
170   public synchronized void close() throws IOException {
171     try (RandomAccessFile f = raf) {
172       flush();
173     }
174   }
175 
176   public File getFile() {
177     return file;
178   }
179 
180   public static void toByteArray(final long pvalue, final byte[] bytes, final int idx) {
181     long value = pvalue;
182     for (int i = idx + 7; i >= idx; i--) {
183       bytes[i] = (byte) (value & 0xffL);
184       value >>= 8;
185     }
186   }
187 
188   public static void toOutputStream(final long pvalue, final OutputStream os) throws IOException {
189     long value = pvalue;
190     for (int i = 7; i >= 0; i--) {
191       os.write((byte) (value & 0xffL));
192       value >>= 8;
193     }
194   }
195 
196   /**
197    * Commits the data to disk.
198    *
199    * @throws IOException
200    */
201   @Override
202   public synchronized void flush() throws IOException {
203     List<DataRow> blockValues = writeBlock.getValues();
204     if (!blockValues.isEmpty()) {
205       bab.reset();
206       this.recordWriter.write(writeBlock, this.encoder);
207       encoder.flush();
208       raf.write(bab.getBuffer(), 0, bab.size());
209       channel.force(true);
210       updateEOFPtrPointer();
211       blockValues.clear();
212     }
213     channel.force(true);
214   }
215 
216   private void updateEOFPtrPointer() throws IOException {
217     long filePointer = raf.getFilePointer();
218     raf.seek(MAGIC.length);
219     raf.writeLong(filePointer);
220     raf.seek(filePointer);
221   }
222 
223   @SuppressFBWarnings("EI_EXPOSE_REP")
224   public Header getHeader() {
225     return header;
226   }
227 
228   @Override
229   public String toString() {
230     return "TSDBWriter{" + "file=" + file + ", raf=" + raf + '}';
231   }
