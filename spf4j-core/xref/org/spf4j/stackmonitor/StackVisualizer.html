49  @SuppressFBWarnings({"CBX_CUSTOM_BUILT_XML", "PREDICTABLE_RANDOM"})

52    private static final String[] COLORS = {"#CCE01B",
53      "#DDE01B", "#EEE01B", "#FFE01B", "#FFD01B",
54      "#FFC01B", "#FFA01B", "#FF901B", "#FF801B",
55      "#FF701B", "#FF601B", "#FF501B", "#FF401B"};

57    private StackVisualizer() {

60    public static void generateHtmlTable(final Writer writer, final Method m,
61            final StackSamples node, final int tableWidth, final int maxDepth) throws IOException {
62      TMap<Method, ? extends StackSamples> subNodes = node.getSubNodes();
63      writer.append("<table border=\"0\" cellpadding=\"0\" cellspacing=\"0\" "
64              + "style=\"overflow:hidden;table-layout:fixed;width:").
65              append(Integer.toString(tableWidth)).append("px\">\n");
66      int totalSamples = node.getSampleCount();
67  
68      if (maxDepth > 0 && !subNodes.isEmpty()) {
69        writer.append("<tr style=\"height:1em\">");
70        for (Map.Entry<Method, ? extends StackSamples> entry : subNodes.entrySet()) {
71          int width = entry.getValue().getSampleCount() * tableWidth / totalSamples;
72          writer.append("<td style=\"vertical-align:bottom; width:").
73                  append(Integer.toString(width)).append("px\">");
74          generateHtmlTable(writer, entry.getKey(), entry.getValue(), width, maxDepth - 1);
75          writer.append("</td>");
76        }
77        writer.append("<td></td>");
78        writer.append("</tr>\n");
79      }
80      writer.append("<tr style=\"height:1em\" ><td ");
81      if (!subNodes.isEmpty()) {
82        writer.append("colspan=\"").append(Integer.toString(subNodes.size() + 1)).append("\" ");
83      }
84      writer.append(" title=\"");
85      Methods.writeHtml(m, writer);
86      writer.append(":");
87      writer.append(Integer.toString(node.getSampleCount()));
88      writer.append("\" style=\"overflow:hidden;width:100%;vertical-align:bottom ;background:").
89              append(COLORS[(int) (Math.random() * COLORS.length)]).append("\">");
90      Methods.writeHtml(m, writer);
91      writer.append(":");
92      writer.append(Integer.toString(totalSamples));
93      writer.append("</td></tr>\n");
94  
95      writer.append("</table>\n");
96    }

98    public static void generateSvg(final Writer writer, final Method m,
99            final SampleNode node, final int x, final int y,
100           final int width, final int maxDepth, final String idPfx) throws IOException {
101     if (width < 200) {
102       throw new IllegalArgumentException("width must be greater than 200, it is " + width);
103     }
104     writer.append("<svg width=\"" + width + "\" height= \"" + (15 * node.height() + 15) + "\" onload=\""
105             + idPfx + "init(evt)\" >\n");
106     writer.append("<script type=\"text/ecmascript\">\n"
107             + "<![CDATA[\n"
108             + "var " + idPfx + "tooltip;\n"
109             + "var " + idPfx + "tooltip_bg;\n"
110             + "function " + idPfx + "init(evt)\n"
111             + "  {\n"
112             + "    if ( window.svgDocument == null )\n"
113             + "    {\n"
114             + "      svgDocument = evt.target.ownerDocument;\n"
115             + "    }\n"
116             + "    " + idPfx + "tooltip = svgDocument.getElementById('" + idPfx + "tooltip'); "
117             + idPfx + "tooltip_bg = svgDocument.getElementById('" + idPfx + "tooltip_bg');"
118             + "  }\n"
119             + "function " + idPfx + "ss(evt, mouseovertext, xx, yy)\n"
120             + "{\n"
121             + "  " + idPfx + "tooltip.setAttributeNS(null,\"x\",xx );\n"
122             + "  " + idPfx + "tooltip.setAttributeNS(null,\"y\",yy+13 );\n"
123             + "  " + idPfx + "tooltip.firstChild.data = mouseovertext;\n"
124             + "  " + idPfx + "tooltip.setAttributeNS(null,\"visibility\",\"visible\");\n"
125             + "length = " + idPfx + "tooltip.getComputedTextLength();\n"
126             + idPfx + "tooltip_bg.setAttributeNS(null,\"width\",length+8);\n"
127             + idPfx + "tooltip_bg.setAttributeNS(null,\"x\",xx);\n"
128             + idPfx + "tooltip_bg.setAttributeNS(null,\"y\",yy);\n"
129             + idPfx + "tooltip_bg.setAttributeNS(null,\"visibility\",\"visibile\");"
130             + "}\n"
131             + "function " + idPfx + "hh()\n"
132             + "{\n"
133             + "  " + idPfx + "tooltip.setAttributeNS(null,\"visibility\",\"hidden\");\n"
134             + idPfx + "tooltip_bg.setAttributeNS(null,\"visibility\",\"hidden\");\n"
135             + "}]]>"
136             + "</script>");
137 
138     generateSubSvg(writer, m, node, x, y + 15, width - 100, maxDepth, idPfx);
139 
140     writer.append("<rect fill=\"rgb(255,255,255)\" id=\"" + idPfx + "tooltip_bg\"\n"
141             + "      x=\"0\" y=\"0\" rx=\"4\" ry=\"4\"\n"
142             + "      width=\"55\" height=\"17\" visibility=\"hidden\"/>");
143     writer.append("<text font-size=\"12\" font-family=\"Verdana\" fill=\"rgb(0,0,0)\"  id=\""
144             + idPfx + "tooltip\" x=\"0\" y=\"0\" visibility=\"hidden\">Tooltip</text>");
145     writer.append("</svg>\n");
146   }

148   @SuppressFBWarnings("ISB_TOSTRING_APPENDING")
149   public static void generateSubSvg(final Writer writer, final Method m, final StackSamples node,
150           final int x, final int y, final int width, final int maxDepth, final String idPfx) throws IOException {
151 
152     Map<Method, ? extends StackSamples> subNodes = node.getSubNodes();
153 
154     int totalSamples = node.getSampleCount();
155     String id = idPfx + "ix" + x + 'y' + y;
156     String content = HtmlEscapers.htmlEscaper().escape(m.toString() + ':' + totalSamples);
157     writer.append("<g onmouseover=\"" + idPfx + "ss(evt,'" + content + "'," + x + ", "
158             + y + " )\" onmouseout=\"" + idPfx + "hh()\">");
159     writer.append("<rect id=\"" + id + "\" x=\"" + x + "\" y=\"" + y + "\" width=\"" + width
160             + "\" height=\"15\" fill=\""
161             + COLORS[(int) (Math.random() * COLORS.length)] + "\"  />");
162 
163     writer.append("<text x=\"" + x + "\" y=\"" + (y + 13)
164             + "\" font-size=\"12\" font-family=\"Verdana\" fill=\"rgb(0,0,0)\" "
165             + " >");
166     writer.append(content.substring(0, Math.min(width / 9, content.length())));
167     writer.append("</text>\n");
168     writer.append("</g>");
169 
170     if (maxDepth > 0 && !subNodes.isEmpty()) {
171       int rx = 0;
172       for (Map.Entry<Method, ? extends StackSamples> entry : subNodes.entrySet()) {
173         int cwidth = (int) (((long) entry.getValue().getSampleCount()) * width / totalSamples);
174         generateSubSvg(writer, entry.getKey(), entry.getValue(), rx + x, y + 15, cwidth, maxDepth - 1, idPfx);
175         rx += cwidth;
176       }
177 
178     }
179   }
