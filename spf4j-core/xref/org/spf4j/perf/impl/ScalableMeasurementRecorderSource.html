74  // a recorder instance is tipically alive for the entire life of the process

77          MeasurementRecorderSource, MeasurementsSource, CloseableMeasurementRecorderSource, JmxSupport {

79    private final Map<Thread, Map<Object, MeasurementAccumulator>> measurementProcessorMap;

81    private final ThreadLocal<Map<Object, MeasurementAccumulator>> threadLocalMeasurementProcessorMap;
82  
83    private final ScheduledFuture<?> samplingFuture;
84    private final MeasurementAccumulator processorTemplate;
85  
86    private final TObjectLongMap<MeasurementsInfo> tableIds;

88    private final Persister persister;
89    private final Runnable shutdownHook;

91    ScalableMeasurementRecorderSource(final MeasurementAccumulator processor,
92            final int sampleTimeMillis, final MeasurementStore database, final boolean closeOnShutdown) {
93      if (sampleTimeMillis < 1000) {
94        throw new IllegalArgumentException("sample time needs to be at least 1000 and not " + sampleTimeMillis);
95      }
96      this.processorTemplate = processor;
97      measurementProcessorMap = new HashMap<>();
98      threadLocalMeasurementProcessorMap = new ThreadLocal<Map<Object, MeasurementAccumulator>>() {
99  
100       @Override
101       protected Map<Object, MeasurementAccumulator> initialValue() {
102         Map<Object, MeasurementAccumulator> result = new HashMap<>();
103         synchronized (measurementProcessorMap) {
104           measurementProcessorMap.put(Thread.currentThread(), result);
105         }
106         return result;
107       }
108 
109     };
110     tableIds = new TObjectLongHashMap<>();
111     persister = new Persister(database, sampleTimeMillis, processor);
112     samplingFuture = DefaultScheduler.scheduleAllignedAtFixedRateMillis(persister, sampleTimeMillis);
113     if (closeOnShutdown) {
114       shutdownHook = closeOnShutdown();
115     } else {
116       shutdownHook = null;
117     }
118   }
119 
120   private Runnable closeOnShutdown() {
121     final AbstractRunnable runnable = new AbstractRunnable(true) {
122 
123       @Override
124       public void doRun() {
125         close();
126       }
127     };
128     org.spf4j.base.Runtime.queueHook(0, runnable);
129     return runnable;
130   }

132   @Override
133   public MeasurementRecorder getRecorder(final Object forWhat) {
134     Map<Object, MeasurementAccumulator> recorders = threadLocalMeasurementProcessorMap.get();
135     synchronized (recorders) {
136       MeasurementAccumulator result = recorders.get(forWhat);
137       if (result == null) {
138         result = (MeasurementAccumulator) processorTemplate.createLike(
139                 Pair.of(processorTemplate.getInfo().getMeasuredEntity(), forWhat));
140         recorders.put(forWhat, result);
141       }
142       return result;
143     }
144   }

146   @Override
147   public Map<Object, MeasurementAccumulator> getEntitiesMeasurements() {
148     Map<Object, MeasurementAccumulator> result = new HashMap<>();
149 
150     synchronized (measurementProcessorMap) {
151       Iterator<Map.Entry<Thread, Map<Object, MeasurementAccumulator>>> iterator
152               = measurementProcessorMap.entrySet().iterator();
153       while (iterator.hasNext()) {
154         Map.Entry<Thread, Map<Object, MeasurementAccumulator>> entry = iterator.next();
155         Map<Object, MeasurementAccumulator> measurements = entry.getValue();
156         synchronized (measurements) {
157           for (Map.Entry<Object, MeasurementAccumulator> lentry : measurements.entrySet()) {
158 
159             Object what = lentry.getKey();
160             MeasurementAccumulator existingMeasurement = result.get(what);
161             if (existingMeasurement == null) {
162               existingMeasurement = lentry.getValue().createClone();
163             } else {
164               existingMeasurement = existingMeasurement.aggregate(lentry.getValue().createClone());
165             }
166             result.put(what, existingMeasurement);
167           }
168         }
169       }
170     }
171     return result;
172   }
173 
174   @Override
175   @Nonnull
176   public Map<Object, MeasurementAccumulator> getEntitiesMeasurementsAndReset() {
177     Map<Object, MeasurementAccumulator> result = new HashMap<>();
178 
179     synchronized (measurementProcessorMap) {
180       Iterator<Map.Entry<Thread, Map<Object, MeasurementAccumulator>>> iterator
181               = measurementProcessorMap.entrySet().iterator();
182       while (iterator.hasNext()) {
183         Map.Entry<Thread, Map<Object, MeasurementAccumulator>> entry = iterator.next();
184         Thread thread = entry.getKey();
185         if (!thread.isAlive()) {
186           iterator.remove();
187         }
188         Map<Object, MeasurementAccumulator> measurements = entry.getValue();
189         synchronized (measurements) {
190           Iterator<Map.Entry<Object, MeasurementAccumulator>> iterator1 = measurements.entrySet().iterator();
191           while (iterator1.hasNext()) {
192             Map.Entry<Object, MeasurementAccumulator> lentry = iterator1.next();
193             Object what = lentry.getKey();
194             MeasurementAccumulator existingMeasurement = result.get(what);
195             if (existingMeasurement == null) {
196               existingMeasurement = lentry.getValue().reset();
197               if (existingMeasurement == null) {
198                 iterator1.remove();
199               } else {
200                 result.put(what, existingMeasurement);
201               }
202             } else {
203               final MeasurementAccumulator vals = lentry.getValue().reset();
204               if (vals != null) {
205                 existingMeasurement = existingMeasurement.aggregate(vals);
206                 result.put(what, existingMeasurement);
207               } else {
208                 iterator1.remove();
209               }
210             }
211           }
212         }
213       }
214     }
215     return result;
216   }

218   @SuppressWarnings("unchecked")
219   @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CHECKED")
220   public void registerJmx() {
221     MeasurementsInfo info = this.processorTemplate.getInfo();
222      CompositeType targetType = addNameDescription(info.toCompositeType());
223     try {
224       String description = info.getDescription();
225       if (description.isEmpty()) {
226         description = "Dynamic measurements";
227       }
228       new DynamicMBeanBuilder().withJmxExportObject(this)
229               .withAttribute(new GenericExportedValue<>("measurements", description,
230                       this::getMeasurements, null, new TabularType(info.getMeasuredEntity().toString(),
231                               description, targetType, new String[]{"name"})))
232               .register("org.spf4j.perf.recorders", info.getMeasuredEntity().toString());
233     } catch (OpenDataException ex) {
234       throw new RuntimeException("Cannot create tabular type for " +  targetType, ex);
235     }
236   }
237 
238   @Override
239   @SuppressFBWarnings("EXS_EXCEPTION_SOFTENING_NO_CHECKED")
240   public void close() {
241     synchronized (persister) {
242       if (!samplingFuture.isCancelled()) {
243         if (shutdownHook != null) {
244           org.spf4j.base.Runtime.removeQueuedShutdownHook(shutdownHook);
245         }
246         samplingFuture.cancel(false);
247         try {
248           persister.persist(false);
249         } catch (IOException ex) {
250           throw new UncheckedIOException(ex);
251         }
252         Registry.unregister("org.spf4j.perf.recorders",
253                 this.processorTemplate.getInfo().getMeasuredEntity().toString());
254       }
255     }
256   }

258   @JmxExport(description = "measurements as csv")
259   public String getMeasurementsAsString() {
260     StringWriter sw = new StringWriter(128);
261     Map<Object, MeasurementAccumulator> entitiesMeasurements = getEntitiesMeasurements();
262     MeasurementsInfo info = this.processorTemplate.getInfo();
263     try {
264       Csv.writeCsvRow2(sw, "Measured", (Object[]) info.getMeasurementNames());
265       Csv.writeCsvRow2(sw, "string", (Object[]) info.getMeasurementUnits());
266       for (Map.Entry<Object, MeasurementAccumulator> entry : entitiesMeasurements.entrySet()) {
267         Csv.writeCsvElement(entry.getKey().toString(), sw);
268         sw.write(',');
269         final long[] measurements = entry.getValue().get();
270         if (measurements != null) {
271           Csv.writeCsvRow(sw, measurements);
272         }
273       }
274 
275     } catch (IOException ex) {
276       throw new UncheckedIOException(ex);
277     }
278     return sw.toString();
279   }

281   static CompositeType addNameDescription(final CompositeType initType) {
282     Set<String> keys = initType.keySet();
283     int colNr = keys.size() + 2;
284     String[] names = new String[colNr];
285     String[] descrs = new String[colNr];
286     OpenType[] types = new OpenType[colNr];
287     names[0] = "name";
288     names[1] = "description";
289     descrs[0] = "metric name";
290     descrs[1] = "metric description";
291     types[0] = javax.management.openmbean.SimpleType.STRING;
292     types[1] = javax.management.openmbean.SimpleType.STRING;
293     int i = 2;
294     for (String key : keys) {
295       names[i] = key;
296       descrs[i] = initType.getDescription(key);
297       types[i++] = initType.getType(key);
298     }
299     try {
300       return new CompositeType(initType.getTypeName(), initType.getDescription(),
301               names, descrs, types);
302     } catch (OpenDataException ex) {
303       throw new IllegalArgumentException("Invalid type contructed from " + initType, ex);
304     }
305   }

307   static CompositeData addNameDescription(final CompositeType targetType, final CompositeData data,
308           final String name, final String description) {
309     Set<String> keySet = targetType.keySet();
310     int size = keySet.size();
311     Map<String, Object> vals = com.google.common.collect.Maps.newLinkedHashMapWithExpectedSize(size);
312     vals.put("name", name);
313     vals.put("description", description);
314     for (String key : data.getCompositeType().keySet()) {
315       vals.put(key, data.get(key));
316     }
317     try {
318       return new CompositeDataSupport(targetType, vals);
319     } catch (OpenDataException ex) {
320       throw new IllegalArgumentException("Invalid open data contructed from " + data, ex);
321     }
322   }

324   public TabularDataSupport getMeasurements() {
325     Map<Object, MeasurementAccumulator> entitiesMeasurements = getEntitiesMeasurements();
326     MeasurementsInfo info = this.processorTemplate.getInfo();
327     CompositeType targetType = addNameDescription(info.toCompositeType());
328     TabularDataSupport result;
329     try {
330       String name = info.getMeasuredEntity().toString();
331       String description = info.getDescription();
332       if (description.isEmpty()) {
333         description = name;
334       }
335       result = new TabularDataSupport(new TabularType(name, description, targetType, new String[]{"name"}));
336     } catch (OpenDataException ex) {
337       throw new RuntimeException("Enable to contruct tabular data " + entitiesMeasurements, ex);
338     }
339     for (Map.Entry<Object, MeasurementAccumulator> entry : entitiesMeasurements.entrySet()) {
340       MeasurementAccumulator acc = entry.getValue();
341       MeasurementsInfo eInfo = acc.getInfo();
342       String cattrName = eInfo.getMeasuredEntity().toString();
343       String cattrDesc = eInfo.getDescription();
344       if (cattrDesc.isEmpty()) {
345         cattrDesc = cattrName;
346       }
347       result.put(addNameDescription(targetType, acc.getCompositeData(), cattrName, cattrDesc));
348     }
349     return result;
350   }

357   private class Persister extends AbstractRunnable {
358 
359     private final MeasurementStore database;
360     private final int sampleTimeMillis;
361     private final MeasurementAccumulator processor;
362     private volatile long lastRun = 0;
363 
364     Persister(final MeasurementStore database, final int sampleTimeMillis,
365             final MeasurementAccumulator processor) {
366       super(true);
367       this.database = database;
368       this.sampleTimeMillis = sampleTimeMillis;
369       this.processor = processor;
370     }
371 
372     @Override
373     public void doRun() throws IOException {
374       persist(true);
375     }
376 
377     public void persist(final boolean warn) throws IOException {
378       long currentTime = System.currentTimeMillis();
379       if (currentTime > lastRun) {
380         lastRun = currentTime;
381         for (MeasurementAccumulator m
382                 : ScalableMeasurementRecorderSource.this.getEntitiesMeasurementsAndReset().values()) {
383           final MeasurementsInfo info = m.getInfo();
384           long tableId;
385           synchronized (tableIds) {
386             tableId = tableIds.get(info);
387             if (tableId == 0) {
388               tableId = database.alocateMeasurements(info, sampleTimeMillis);
389               tableIds.put(info, tableId);
390             }
391           }
392           final long[] data = m.getThenReset();
393           if (data != null) {
394             database.saveMeasurements(tableId, currentTime, data);
395           }
396         }
397       } else if (warn) {
398         Logger.getLogger(ScalableMeasurementRecorderSource.class.getName())
399                 .log(Level.WARNING,
400                         "Last measurement recording for {0} was at {1} current run is {2}, something is wrong",
401                         new Object[] {processor.getInfo(), lastRun, currentTime});
402       }
403     }
404   }

408     return "ScalableMeasurementRecorderSource{" + "measurementProcessorMap=" + measurementProcessorMap

410             + ", samplingFuture=" + samplingFuture + ", processorTemplate=" + processorTemplate
411             + ", tableIds=" + tableIds + ", persister=" + persister + ", shutdownHook=" + shutdownHook + '}';
