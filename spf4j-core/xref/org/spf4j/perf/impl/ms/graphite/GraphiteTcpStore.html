68  public final class GraphiteTcpStore implements MeasurementStore {
69  
70    private final RecyclingSupplier<Writer> socketWriterSupplier;
71  
72    private final InetSocketAddress address;
73  
74    @Override
75    @Nullable
76    public MeasurementStoreQuery query() {
77      return null;
78    }
79  
80  
81  
82    private static class WriterSupplierFactory implements RecyclingSupplier.Factory<Writer> {
83  
84      private final String hostName;
85      private final int port;
86      private final SocketFactory socketFactory;
87  
88      WriterSupplierFactory(final SocketFactory socketFactory, final String hostName, final int port) {
89        this.hostName = hostName;
90        this.port = port;
91        this.socketFactory = socketFactory;
92      }
93  
94      @Override
95      public Writer create() throws ObjectCreationException {
96        Socket socket;
97        try {
98          socket = socketFactory.createSocket(hostName, port);
99        } catch (IOException ex) {
100         throw new ObjectCreationException(ex);
101       }
102       try {
103         return new BufferedWriter(new OutputStreamWriter(socket.getOutputStream(), StandardCharsets.UTF_8));
104       } catch (IOException ex) {
105         try {
106           socket.close();
107         } catch (IOException ex1) {
108           ex1.addSuppressed(ex);
109           throw new ObjectCreationException(ex1);
110         }
111         throw new ObjectCreationException(ex);
112       }
113     }
114 
115     @Override
116     public void dispose(final Writer object) throws ObjectDisposeException {
117       try {
118         object.close();
119       } catch (IOException ex) {
120         throw new ObjectDisposeException(ex);
121       }
122     }
123 
124     @Override
125     public boolean validate(final Writer object, final Exception e) {
126       return e == null || !(Throwables.getRootCause(e) instanceof IOException);
127     }
128 
129   }
130 
131   public GraphiteTcpStore(final String hostPort) throws ObjectCreationException, URISyntaxException {
132     this(new URI("graphiteTcp://" + hostPort));
133   }
134 
135   public GraphiteTcpStore(final URI uri) throws ObjectCreationException {
136     this(uri.getHost(), uri.getPort());
137   }
138 
139   public GraphiteTcpStore(final String hostName, final int port) throws ObjectCreationException {
140     this(hostName, port, SocketFactory.getDefault());
141   }
142 
143   public GraphiteTcpStore(final String hostName, final int port, final SocketFactory socketFactory)
144           throws ObjectCreationException {
145     address = new InetSocketAddress(hostName, port);
146     socketWriterSupplier = new RecyclingSupplierBuilder<>(1,
147             new WriterSupplierFactory(socketFactory, hostName, port)).build();
148   }
149 
150   @Override
151   public long alocateMeasurements(final MeasurementsInfo measurement, final int sampleTimeMillis) {
152     return Id2Info.getId(measurement);
153   }
154 
155   @Override
156   @SuppressFBWarnings("BED_BOGUS_EXCEPTION_DECLARATION") // fb nonsense
157   public void saveMeasurements(final long tableId,
158           final long timeStampMillis, final long... measurements) throws IOException {
159     try {
160       Template.doOnSupplied(new HandlerImpl(measurements, Id2Info.getInfo(tableId), timeStampMillis),
161               1, TimeUnit.MINUTES,
162               socketWriterSupplier, RetryPolicy.defaultPolicy(), IOException.class);
163     } catch (InterruptedException ex) {
164       Thread.currentThread().interrupt();
165       throw new IOException("Interupted while saving measurements to " + tableId, ex);
166     } catch (TimeoutException ex) {
167       throw new UncheckedTimeoutException(ex);
168     }
169   }
170 
171   @Override
172   public String toString() {
173     return "GraphiteTcpStore{address=" + address + '}';
174   }
175 
176   @Override
177   public void close() {
178     try {
179       socketWriterSupplier.dispose();
180     }  catch (InterruptedException ex) {
181       Thread.currentThread().interrupt();
182       throw new UncheckedExecutionException(ex);
183     } catch (ObjectDisposeException ex) {
184       throw new UncheckedExecutionException(ex);
185     }
186   }
187 
188   private static class HandlerImpl implements HandlerNano<Writer, Void, IOException> {
189 
190     private final long[] measurements;
191     private final MeasurementsInfo measurementInfo;
192     private final long timeStampMillis;
193 
194     HandlerImpl(final long[] measurements, final MeasurementsInfo measurementInfo,
195             final long timeStampMillis) {
196       this.measurements = measurements;
197       this.measurementInfo = measurementInfo;
198       this.timeStampMillis = timeStampMillis;
199     }
200 
201     @Override
202     @Nullable
203     public Void handle(final Writer socketWriter, final long deadline) throws IOException {
204       for (int i = 0; i < measurements.length; i++) {
205         writeMetric(measurementInfo, measurementInfo.getMeasurementName(i),
206                 measurements[i], timeStampMillis, socketWriter);
207       }
208       socketWriter.flush();
209       return null;
210     }
211   }
212 
213   @Override
214   public void flush() {
215     // No buffering yet
216   }
