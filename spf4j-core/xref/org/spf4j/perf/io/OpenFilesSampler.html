53  /**
54   * This class allows you to poll and recordAt to a file the heap commitxted and heap used for your java process. start
55   * data recording by calling the startMemoryUsageSampling method, stop the data recording by calling the method:
56   * startMemoryUsageSampling.
57   *
58   * @author zoly
59   */
60  public final class OpenFilesSampler {
61  
62  
63    private static ScheduledFuture<?> samplingFuture;
64    private static AccumulatorRunnable accumulator;
65  
66    private static volatile CharSequence lastWarnLsof = "";
67  
68    private static volatile Instant lastWarnLsofTime;
69  
70    static {
71      Runtime.queueHook(2, new AbstractRunnable(true) {
72        @Override
73        public void doRun() {
74          stop();
75        }
76      });
77      Registry.export(OpenFilesSampler.class);
78    }
79  
80    private OpenFilesSampler() {
81    }
82  
83    public static void start(final long sampleTimeMillis) {
84      long maxFileDescriptorCount = OperatingSystem.getMaxFileDescriptorCount();
85      start(sampleTimeMillis, maxFileDescriptorCount - maxFileDescriptorCount / 10,
86              maxFileDescriptorCount, true);
87    }
88  
89    @JmxExport
90    public static void start(@JmxExport("sampleTimeMillis") final long sampleTimeMillis,
91            @JmxExport("shutdownOnError") final boolean shutdownOnError) {
92      long maxFileDescriptorCount = OperatingSystem.getMaxFileDescriptorCount();
93      start(sampleTimeMillis, maxFileDescriptorCount - maxFileDescriptorCount / 10,
94              maxFileDescriptorCount, shutdownOnError);
95    }
96  
97    @JmxExport
98    public static String getWarnLsofDetail() {
99      return lastWarnLsof.toString();
100   }
101 
102   @JmxExport
103   @Nullable
104   public static synchronized String getWarnLsofTime() {
105     Instant lwt = lastWarnLsofTime;
106     if (lwt == null) {
107       return null;
108     } else {
109       return lwt.toString();
110     }
111   }
112 
113 
114   public static synchronized void start(final long sampleTimeMillis,
115           final long warnThreshold, final long errorThreshold, final boolean shutdownOnError) {
116     if (samplingFuture == null) {
117       accumulator = new AccumulatorRunnable(errorThreshold, shutdownOnError,
118               warnThreshold, (int) sampleTimeMillis);
119       samplingFuture = DefaultScheduler.INSTANCE.scheduleWithFixedDelay(accumulator,
120               sampleTimeMillis, sampleTimeMillis, TimeUnit.MILLISECONDS);
121     } else {
122       throw new IllegalStateException("Open file usage sampling already started " + samplingFuture);
123     }
124   }
125 
126   @JmxExport
127   public static synchronized void stop()  {
128     if (samplingFuture != null) {
129       samplingFuture.cancel(false);
130       samplingFuture = null;
131       accumulator = null;
132     }
133   }
134 
135   @JmxExport
136   public static synchronized boolean isStarted() {
137     return samplingFuture != null;
138   }
139 
140   @JmxExport
141   public static String getLsof() {
142     CharSequence lsofOutput = Lsof.getLsofOutput();
143     return lsofOutput == null ? "unable to obtain lsof" : lsofOutput.toString();
144   }

146   @Deprecated
147   public static long getMaxNrOpenFiles() {
148     return OperatingSystem.getMaxFileDescriptorCount();
149   }
150 
151   @Deprecated
152   public static long getNrOpenFiles() {
153     return OperatingSystem.getOpenFileDescriptorCount();
154   }

156   @JmxExport
157   public static long getWarnThreshold() {
158     if (accumulator == null) {
159       return -1;
160     }
161     return accumulator.getWarnThreshold();
162   }
163 
164   @JmxExport
165   public static long getErrorThreshold() {
166     if (accumulator == null) {
167       return -1;
168     }
169     return accumulator.getErrorThreshold();
170   }

172 
173   private static class AccumulatorRunnable extends AbstractRunnable {
174 
175     private final long errorThreshold;
176     private final boolean shutdownOnError;
177     private final long warnThreshold;
178     private final MeasurementRecorder nrOpenFiles;
179 
180     AccumulatorRunnable(final long errorThreshold, final boolean shutdownOnError,
181             final long warnThreshold, final int sampleMillis) {
182       this.errorThreshold = errorThreshold;
183       this.shutdownOnError = shutdownOnError;
184       this.warnThreshold = warnThreshold;
185       this.nrOpenFiles = RecorderFactory.createDirectRecorder("process.nr_open_files", "count",
186               sampleMillis, MeasurementType.GAUGE);
187     }
188 
189     @Override
190     @SuppressFBWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
191     public void doRun() {
192       long time = System.currentTimeMillis();
193       long nrOf = OperatingSystem.getOpenFileDescriptorCount();
194       if (nrOf > errorThreshold) {
195         lastWarnLsof = Lsof.getLsofOutput();
196         lastWarnLsofTime = Instant.now();
197         Logger log = Logger.getLogger(OpenFilesSampler.class.getName());
198         log.log(Level.SEVERE, "Nr open files is {0} and exceeds error threshold {1}, detail:\n{2}",
199                 new Object[] {nrOf, errorThreshold, lastWarnLsof});
200         if (shutdownOnError) {
201           Runtime.goDownWithError(null, SysExits.EX_IOERR);
202         }
203       } else if (nrOf > warnThreshold) {
204         lastWarnLsof = Lsof.getLsofOutput();
205         lastWarnLsofTime = Instant.now();
206         Logger log = Logger.getLogger(OpenFilesSampler.class.getName());
207         log.log(Level.WARNING, "Nr open files is {0} and exceeds warn threshold {1}, detail:\n{2} ",
208                 new Object[] {nrOf, warnThreshold, lastWarnLsof});
209         if (!Runtime.gc(60000)) {
210           log.warning("Unable to trigger GC although running low on file resources");
211         } else {
212           log.log(Level.WARNING, "gc executed nr open files reduced by {0} files",
213                   nrOf - OperatingSystem.getOpenFileDescriptorCount());
214         }
215       }
216       this.nrOpenFiles.recordAt(time, nrOf);
217     }
218 
219     public long getErrorThreshold() {
220       return errorThreshold;
221     }
222 
223     public boolean isShutdownOnError() {
224       return shutdownOnError;
225     }
226 
227     public long getWarnThreshold() {
228       return warnThreshold;
229     }
