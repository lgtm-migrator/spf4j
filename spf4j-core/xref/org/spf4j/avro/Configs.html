70    @Nullable
71    private static MediaType getConfigMediaType(final CharSequence seq) {
72      if (!CharSequences.startsWith(seq, "Content-Type:", 0)) {
73        return null;
74      }
75      int length = seq.length();
76      int indexOf = CharSequences.indexOf(seq, 0, length, ':');
77      int next = indexOf + 1;
78      int endMt = CharSequences.indexOf(seq, next, length, '\n');
79      if (endMt < 0) {
80        endMt = length;
81      }
82      return MediaType.parse(seq.subSequence(next, endMt).toString());
83    }

85    @Nullable
86    public static <T> T read(final Class<T> type, final SchemaResolver schemaResolver, final Reader... reader)
87            throws IOException {
88      if (reader.length == 0) {
89        return null;
90      }
91      SpecificData sd = SpecificData.get();
92      Schema rSchema = sd.getSchema(type);
93  
94      int l = reader.length - 1;
95      ConfigHeader header = parseHeader(reader[l], rSchema, schemaResolver);
96      JsonNode node = readTree(header.getReader(), header.getMediaType());
97      Schema wSchema = header.getwSchema();
98      for (int i = l - 1; i >= 0; i--) {
99        header = parseHeader(reader[i], rSchema, schemaResolver);
100       JsonNode newNode = readTree(header.getReader(), header.getMediaType());
101       wSchema = header.getwSchema();
102       node = override(newNode, node, wSchema);
103     }
104     DatumReader<T> dr = new SpecificDatumReader<>(wSchema, rSchema);
105     Adapter adapter = AvroCompatUtils.getAdapter();
106     Decoder decoder = adapter.getJsonDecoder(wSchema, node.traverse());
107     return dr.read(null, decoder);
108 
109   }

111   private static JsonNode readTree(final Reader reader, final MediaType mt) throws IOException {
112     if ("application".equals(mt.type()) && "json".equals(mt.subtype())) {
113       return Json.MAPPER.readTree(reader);
114     } else if ("text".equals(mt.type()) && "yaml".equals(mt.subtype())) {
115       return Yaml.MAPPER.readTree(reader);
116     } else {
117       throw new IllegalArgumentException("Unsupported media type " + mt);
118     }
119   }

121   @Nullable
122   public static JsonNode override(@Nullable final JsonNode target,
123           @Nullable final JsonNode defaults,
124           final Schema targetSchema) {
125     if (target == null) {
126       return defaults;
127     }
128     Schema.Type type = targetSchema.getType();
129     switch (type) {
130       case RECORD:
131         ObjectNode newNode = JsonNodeFactory.instance.objectNode();
132         if (!(target instanceof ObjectNode)) {
133           throw new IllegalArgumentException("Target must be of type Object/record, not: " + target);
134         }
135         newNode.setAll((ObjectNode) target);
136         for (Field field : targetSchema.getFields()) {
137            String fieldName  = field.name();
138            JsonNode d = defaults.get(fieldName);
139            if (d == null) {
140              for (String alias : field.aliases()) {
141                d = defaults.get(alias);
142                if (d != null) {
143                  break;
144                }
145              }
146            }
147            if (d == null) {
148              continue;
149            }
150            JsonNode targetField = target.get(fieldName);
151            if (targetField == null) {
152              newNode.set(fieldName, d);
153            } else {
154              newNode.set(fieldName, override(targetField, d,
155                      targetSchema.getField(fieldName).schema()));
156            }
157         }
158         return newNode;
159       case MAP:
160         newNode = JsonNodeFactory.instance.objectNode();
161         newNode.setAll((ObjectNode) target);
162         for (Entry<String, JsonNode> field : (Iterable<Entry<String, JsonNode>>) defaults::fields) {
163            String fieldName  = field.getKey();
164            JsonNode targetField = target.get(fieldName);
165            if (targetField == null) {
166              newNode.set(fieldName, field.getValue());
167            } else {
168              newNode.set(fieldName, override(targetField, field.getValue(),
169                      targetSchema.getValueType()));
170            }
171         }
172         return newNode;
173       case UNION:
174 
175       case ARRAY:
176       case BOOLEAN:
177       case BYTES:
178       case DOUBLE:
179       case ENUM:
180       case FIXED:
181       case FLOAT:
182       case INT:
183       case LONG:
184       case NULL:
185       case STRING:
186         return target;
187       default:
188         throw new UnsupportedOperationException("Unsupported type: " + type);
189     }
190   }

192   public static Pair<Schema, JsonNode> getMatchingSchema(final JsonNode node, final List<Schema> schemas) {
193     if (node.isObject()) {
194        Iterator<Entry<String, JsonNode>> it = node.fields();
195        if (it.hasNext()) {
196          Entry<String, JsonNode> entry = it.next();
197          if (it.hasNext()) { // must be a record schema.
198            Schema theSchema = getOneSchema(schemas, Schema.Type.RECORD, Schema.Type.MAP);
199            return Pair.of(theSchema, node);
200          } else {
201            Schema s = getOneSchema(schemas, entry.getKey());
202            if (s == null) {
203               Schema theSchema = getOneSchema(schemas, Schema.Type.RECORD, Schema.Type.MAP);
204               return Pair.of(theSchema, node);
205            } else {
206               return Pair.of(s, entry.getValue());
207            }
208          }
209        } else {
210            Schema theSchema = getOneSchema(schemas, Schema.Type.RECORD, Schema.Type.MAP);
211            return Pair.of(theSchema, node);
212        }
213     } else if (node.isNull()) {
214       return Pair.of(getOneSchema(schemas, Schema.Type.NULL), node);
215     } else if (node.isArray()) {
216       return Pair.of(getOneSchema(schemas, Schema.Type.ARRAY), node);
217     } else if (node.isTextual()) {
218       return Pair.of(getOneSchema(schemas, Schema.Type.STRING,  Schema.Type.BYTES, Schema.Type.FIXED), node);
219     } else if (node.isBoolean()) {
220       return Pair.of(getOneSchema(schemas, Schema.Type.BOOLEAN), node);
221     } else if (node.isNumber()) {
222       // direct encoded types not really handled here
223       return Pair.of(getOneSchema(schemas,
224               Schema.Type.DOUBLE,  Schema.Type.FLOAT, Schema.Type.INT, Schema.Type.LONG), node);
225     } else {
226       throw new IllegalStateException("IUnsupported node type " + node);
227     }
228   }

230   @Nullable
231   private static Schema getOneSchema(final List<Schema> schemas, final String name) {
232     for (Schema s : schemas) {
233       if (s.getFullName().equals(name)) {
234         return s;
235       }
236     }
237    return null;
238   }

240   private static Schema getOneSchema(final List<Schema> schemas, final Schema.Type type) {
241     Schema result = null;
242     for (Schema s : schemas) {
243       if (s.getType() == type) {
244         if (result == null) {
245           result = s;
246         } else {
247           throw new IllegalArgumentException("Must have only one " + type + " schema in " + schemas);
248         }
249       }
250     }
251     if (result == null) {
252       throw new IllegalArgumentException("No record " + type + "  in " + schemas);
253     } else {
254       return result;
255     }
256   }
257 
258   private static Schema getOneSchema(final List<Schema> schemas, final Schema.Type... types) {
259     Schema result = null;
260     for (Schema s : schemas) {
261       for (Schema.Type t : types) {
262         if (s.getType() == t) {
263           if (result == null) {
264             result = s;
265           } else {
266             throw new IllegalArgumentException("Must have only one schema of types: "
267                     + Arrays.toString(types) + " in " + schemas);
268           }
269         }
270       }
271     }
272     if (result == null) {
273       throw new IllegalArgumentException("No schema of type: " + Arrays.toString(types) + " in " + schemas);
274     } else {
275       return result;
276     }
277   }

286   private static final class ConfigHeader {
287 
288     private final Schema wSchema;
289 
290     private final Reader reader;
291 
292     private final MediaType mediaType;
293 
294     ConfigHeader(final Schema wSchema, final Reader reader, final MediaType mediaType) {
295       this.wSchema = wSchema;
296       this.reader = reader;
297       this.mediaType = mediaType;
298     }
299 
300     @SuppressFBWarnings("EI_EXPOSE_REP")
301     public Schema getwSchema() {
302       return wSchema;
303     }
304 
305     /**
306      * The reader with the content of the config. (header component was consumed)
307      * @return
308      */
309     @CreatesObligation
310     public Reader getReader() {
311       return reader;
312     }
313 
314     public MediaType getMediaType() {
315       return mediaType;
316     }
317 
318     @Override
319     public String toString() {
320       return "ConfigHeader{" + "wSchema=" + wSchema + ", reader=" + reader + ", mediaType=" + mediaType + '}';
321     }

323   }

325   private static ConfigHeader parseHeader(@WillNotClose final Reader reader, final Schema rSchema,
326           final SchemaResolver schemaResolver) throws IOException {
327     Schema wSchema;
328     MediaType mt;
329     PushbackReader pbr = new PushbackReader(reader);
330     int firstChar = pbr.read();
331     if (firstChar < 0) {
332       mt = MediaType.JSON_UTF_8;
333       Reader r = pbr;
334       wSchema = rSchema;
335       return new ConfigHeader(wSchema, r, mt);
336     }
337     BufferedReader content;
338     Adapter adapter = AvroCompatUtils.getAdapter();
339     if (firstChar == '#') {
340       content = new BufferedReader(pbr);
341         String header = content.readLine();
342         if (header == null) {
343             mt = MediaType.JSON_UTF_8;
344             pbr.unread(firstChar);
345             Reader r = pbr;
346             wSchema = rSchema;
347             return new ConfigHeader(wSchema, r, mt);
348         }
349         mt = getConfigMediaType(header);
350         if (mt != null) {
351           List<String> schemaStrList = mt.parameters().get("avsc");
352           if (schemaStrList.isEmpty()) {
353             wSchema = rSchema;
354           } else {
355             wSchema = adapter.parseSchema(new StringReader(schemaStrList.get(0)), true, schemaResolver);
356           }
357         } else {
358            mt = MediaType.create(header, header);
359            wSchema = rSchema;
360         }
361     } else {
362       mt = MediaType.JSON_UTF_8;
363       pbr.unread(firstChar);
364       content = new BufferedReader(pbr);
365       wSchema = rSchema;
366     }
367     return new ConfigHeader(wSchema, content, mt);
368   }

371   @Nullable
372   public static <T> T read(final Class<T> type, final SchemaResolver schemaResolver, final Reader reader)
373           throws IOException {
374     SpecificData sd = SpecificData.get();
375     Schema rSchema = sd.getSchema(type);
376     ConfigHeader header = parseHeader(reader, rSchema, schemaResolver);
377     Schema wSchema = header.getwSchema();
378     Reader content = header.getReader();
379     MediaType mt = header.getMediaType();
380     DatumReader<T> dr = new SpecificDatumReader<>(wSchema, rSchema);
381     Decoder decoder;
382     Adapter adapter = AvroCompatUtils.getAdapter();
383     if ("application".equals(mt.type()) && "json".equals(mt.subtype())) {
384       decoder = adapter.getJsonDecoder(wSchema, content);
385     } else if ("text".equals(mt.type()) && "yaml".equals(mt.subtype())) {
386       decoder = adapter.getYamlDecoder(wSchema, content);
387     } else {
388       throw new IllegalArgumentException("Unsupported media type " + mt);
389     }
390     return dr.read(null, decoder);
391   }
