58    @Test
59    public void testGetFormatParameterNumber() {
60      Assert.assertEquals(0, Slf4jMessageFormatter.getFormatParameterNumber("bla bla"));
61      Assert.assertEquals(1, Slf4jMessageFormatter.getFormatParameterNumber("bla bla {}"));
62      Assert.assertEquals(2, Slf4jMessageFormatter.getFormatParameterNumber("{} bla bla {}"));
63      Assert.assertEquals(1, Slf4jMessageFormatter.getFormatParameterNumber("\\{} bla bla {}"));
64    }

67    @Test
68    public void testFormatter() throws IOException {
69      StringBuilder sb = new StringBuilder();
70      Slf4jMessageFormatter.format(sb, "bla bla");
71      Assert.assertEquals("bla bla", sb.toString());
72      sb.setLength(0);
73      Slf4jMessageFormatter.format(sb, "bla bla {}", "coco");
74      Assert.assertEquals("bla bla coco", sb.toString());
75      sb.setLength(0);
76      Slf4jMessageFormatter.format(sb, "\\{}bla bla {}", "coco");
77      Assert.assertEquals("{}bla bla coco", sb.toString());
78      sb.setLength(0);
79      int processed = Slf4jMessageFormatter.format(sb, "Some Message", "coco");
80      Assert.assertEquals("Some Message", sb.toString());
81      Assert.assertEquals(0, processed);
82      sb.setLength(0);
83      processed = Slf4jMessageFormatter.format(sb, "Some Message {}", new ConfigurableAppenderSupplier(),
84              Pair.of("a", "b"));
85      Assert.assertEquals("Some Message a,b", sb.toString());
86      Assert.assertEquals(1, processed);
87    }

89    @Test
90    public void testFormatter2() throws IOException {
91      ConfigurableAppenderSupplier appSupp = new ConfigurableAppenderSupplier();
92      LOG.debug("ConfAppenderSupp = {}", appSupp);
93      StringBuilder sb = new StringBuilder();
94      final long currentTimeMillis = System.currentTimeMillis();
95      Slf4jMessageFormatter.format(sb, "bla bla {}", appSupp, new java.sql.Date(currentTimeMillis));
96      Assert.assertEquals("bla bla " + DateTimeFormats.DT_FORMAT.format(Instant.now()), sb.toString());
97      sb.setLength(0);
98      Method method = new Method("c1", "m1");
99      int written = Slf4jMessageFormatter.format(sb, "bla bla {}", appSupp, method);
100     Assert.assertEquals("bla bla {\"declaringClass\":\"c1\",\"name\":\"m1\"}", sb.toString());
101     Assert.assertEquals(1, written);
102     sb.setLength(0);
103     written = Slf4jMessageFormatter.format(1, sb, "bla bla {}", appSupp, "ifff", method);
104     Assert.assertEquals("bla bla {\"declaringClass\":\"c1\",\"name\":\"m1\"}", sb.toString());
105     Assert.assertEquals(2, written);
106     sb.setLength(0);
107     written = Slf4jMessageFormatter.format(sb, "bla bla {}", appSupp, method, "yohooo");
108     LOG.debug("formatted message: {}", sb);
109     Assert.assertEquals(1, written);
110     sb.setLength(0);
111     written = Slf4jMessageFormatter.format(sb, "bla bla {}", appSupp);
112     LOG.debug("formatted message: {}", sb);
113     Assert.assertEquals(0, written);
114     sb.setLength(0);
115     EscapeJsonStringAppendableWrapper escaper = new EscapeJsonStringAppendableWrapper(sb);
116     Slf4jMessageFormatter.format(escaper, "bla bla {} {}", appSupp, "\n\u2013\u0010",
117             new int[]{1, 2, 3});
118     LOG.debug("formatted message: {}", sb);
119     Assert.assertEquals("bla bla \\n–\\u0010 [1, 2, 3]", sb.toString());
120     appSupp.replace(CoreTextMediaType.TEXT_PLAIN,
121             String.class, (final ObjectAppender<? super String> input) -> new ObjectAppender<String>() {
122       @Override
123       public void append(final String object, final Appendable appendTo) throws IOException {
124         try (AppendableLimiterWithOverflow limiter
125                 = new AppendableLimiterWithOverflow(90, File.createTempFile("string", ".overflow"),
126                         "...@", StandardCharsets.UTF_8, appendTo)) {
127           limiter.append(object);
128         }
129       }
130     });
131     sb.setLength(0);
132     Slf4jMessageFormatter.format(sb, "bla bla {}", appSupp,
133             "012345678901234567890123456789012345678901234567"
134                     + "89012345678901234567890123456789012345678901234567890123456789");
135     LOG.debug("formatted message: {}", sb);
136     Assert.assertThat(sb.toString(), Matchers.containsString("...@"));
137   }

139   @Test
140   @SuppressFBWarnings({ "LO_INCORRECT_NUMBER_OF_ANCHOR_PARAMETERS", "UCC_UNRELATED_COLLECTION_CONTENTS" })
141   public void testFormatterRecursion() throws IOException {
142     StringBuilder builder = new StringBuilder();
143     Object[] arr = new Object[4];
144     arr[0] = "a";
145     arr[1] = arr;
146     arr[2] = "b";
147     arr[3] = arr;
148     Slf4jMessageFormatter.format(builder, "{} {}", (Object) arr, arr);
149     LOG.debug("", builder);
150     Assert.assertEquals("[a, [...], b, [...]] [a, [...], b, [...]]", builder.toString());
151   }
