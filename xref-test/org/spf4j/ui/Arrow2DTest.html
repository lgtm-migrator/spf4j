16  package org.spf4j.ui;

18  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
19  import java.awt.BorderLayout;
20  import java.awt.Graphics;
21  import java.awt.Graphics2D;
22  import java.util.concurrent.CountDownLatch;
23  import java.util.concurrent.TimeUnit;
24  import javax.swing.JFrame;
25  import javax.swing.JPanel;
26  import javax.swing.SwingUtilities;
27  import org.junit.Assert;
28  import org.junit.Test;
29  import org.spf4j.test.log.TestUtils;
30  
31  /**
32   * @author Zoltan Farkas
33   */
34  public class Arrow2DTest {
35  
36    @Test
37    public void testArrowDraw() throws InterruptedException {
38      CountDownLatch latch = new CountDownLatch(1);
39      CountDownLatch closeLatch = new CountDownLatch(1);
40      SwingUtilities.invokeLater(() -> {
41        JFrame frame = new JFrame("Arrow");
42  
43        frame.add(new JPanel() {
44          @SuppressFBWarnings("BC_UNCONFIRMED_CAST")
45          public void paintComponent(final Graphics g) {
46            new Arrow2D(0, 0, 100, 100).draw((Graphics2D) g);
47          }
48        }, BorderLayout.CENTER);
49        frame.setSize(800, 400);
50        frame.addWindowListener(new LatchWindowCloseListener(closeLatch));
51        frame.setDefaultCloseOperation(JFrame.DISPOSE_ON_CLOSE);
52        frame.setVisible(true);
53        latch.countDown();
54      });
55      Assert.assertTrue(latch.await(10, TimeUnit.SECONDS));
56      if (TestUtils.isExecutedFromIDEInvididually()) {
57        Assert.assertTrue(closeLatch.await(1, TimeUnit.HOURS));
58      }
59    }
60  
61  }
