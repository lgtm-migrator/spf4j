74    private static String semddl;

76    static {
77      try {
78        hbddl = Resources.toString(Resources.getResource("heartBeats.sql"), StandardCharsets.US_ASCII);
79        semddl = Resources.toString(Resources.getResource("semaphoreTable.sql"), StandardCharsets.US_ASCII);
80      } catch (IOException ex) {
81        throw new ExceptionInInitializerError(ex);
82      }
83    }

85    static void createSchemaObjects(final DataSource ds) throws SQLException {
86      try (Connection conn = ds.getConnection()) {
87        try (Statement stmt = conn.createStatement()) {
88          stmt.execute(hbddl);
89        }
90  
91        try (Statement stmt = conn.createStatement()) {
92          stmt.execute(semddl);
93        }
94      }
95    }
96  
97    static PooledDataSource createPooledDS(final JdbcDataSource ds) throws ObjectCreationException {
98  
99      return new PooledDataSource(0, 4, new RecyclingSupplier.Factory<Connection>() {
100       @Override
101       public Connection create() throws ObjectCreationException {
102         try {
103           return ds.getConnection();
104         } catch (SQLException ex) {
105           throw new ObjectCreationException(ex);
106         }
107       }
108 
109       @Override
110       public void dispose(final Connection object) throws ObjectDisposeException {
111         try {
112           object.close();
113         } catch (SQLException ex) {
114           throw new ObjectDisposeException(ex);
115         }
116       }
117 
118       @Override
119       public boolean validate(final Connection object, final Exception e) throws SQLException {
120         return object.isValid(60);
121       }
122     });

189   @SuppressFBWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
190   public static void testReleaseAck(final DataSource ds, final String semName, final int maxReservations)
191           throws SQLException, InterruptedException, TimeoutException {
192     JdbcSemaphore semaphore = new JdbcSemaphore(ds, semName, maxReservations);
193 
194     // test update;
195     int totalPermits = semaphore.totalPermits();
196     int acquire = totalPermits - 1;
197     semaphore.acquire(acquire, 1, TimeUnit.SECONDS);
198     Assert.assertEquals(1, semaphore.permitsOwned());
199     int nrPermits = semaphore.totalPermits();
200     semaphore.updatePermits(nrPermits + 1);
201     Assert.assertEquals(nrPermits + 1, semaphore.totalPermits());
202     semaphore.updatePermits(nrPermits);
203     Assert.assertEquals(nrPermits, semaphore.totalPermits());
204     semaphore.reducePermits(2);
205     Assert.assertFalse(semaphore.tryAcquire(2, TimeUnit.SECONDS));
206     semaphore.release(acquire);
207     semaphore.increasePermits(2);
208     Assert.assertEquals(totalPermits, semaphore.totalPermits());
209 
210     Assert.assertTrue(semaphore.tryAcquire(1, 10, TimeUnit.SECONDS));
211     semaphore.release(1);
212     Assert.assertTrue(semaphore.tryAcquire(2, 10, TimeUnit.SECONDS));
213     Assert.assertFalse(semaphore.tryAcquire(1, 2, TimeUnit.SECONDS));
214     semaphore.release(1);
215     semaphore.release(1);
216     Assert.assertEquals(maxReservations, semaphore.availablePermits());
217     Assert.assertEquals(maxReservations, semaphore.totalPermits());
218     semaphore.reducePermits(1);
219     Assert.assertEquals(maxReservations - 1, semaphore.totalPermits());
220     Assert.assertEquals(maxReservations - 1, semaphore.availablePermits());
221     Assert.assertTrue(semaphore.tryAcquire(1, 10, TimeUnit.SECONDS));
222     Assert.assertFalse(semaphore.tryAcquire(1, 2, TimeUnit.SECONDS));
223     semaphore.release(1);
224     try {
225       semaphore.release(1);
226       Assert.fail("should not be allow to release!");
227     } catch (IllegalStateException ex) {
228       Assert.assertTrue(ex.getMessage().contains("Trying to release more than you own"));
229     }
230     semaphore.increasePermits(1);
231     Assert.assertTrue(semaphore.tryAcquire(2, 10, TimeUnit.SECONDS));
232     semaphore.reducePermits(1);
233     semaphore.release(2);
234     Assert.assertFalse(semaphore.tryAcquire(2, 10, TimeUnit.SECONDS));
235     semaphore.close();
236   }

239   @SuppressFBWarnings("AFBR_ABNORMAL_FINALLY_BLOCK_RETURN")
240   public void testMultiProcess()
241           throws SQLException, IOException, InterruptedException, ExecutionException, TimeoutException {
242     Server server = Server.createTcpServer(new String[]{"-tcpPort", "9123", "-tcpAllowOthers"}).start();
243 
244     File tempDB = File.createTempFile("test", "h2db");
245     String connStr = "jdbc:h2:tcp://localhost:9123/nio:" + tempDB.getAbsolutePath() + ";AUTO_SERVER=TRUE";
246     try {
247       JdbcDataSource ds = new JdbcDataSource();
248       ds.setURL(connStr);
249       ds.setUser("sa");
250       ds.setPassword("sa");
251 
252       createSchemaObjects(ds);
253       testReleaseAck(ds, "testSem", 2);
254       JdbcSemaphore semaphore = new JdbcSemaphore(ds, "test_sem2", 3);
255       org.spf4j.base.Runtime.jrun(BadSemaphoreHandler.class, 10000, connStr, "test_sem2");
256       org.spf4j.base.Runtime.jrun(BadSemaphoreHandler.class, 10000, connStr, "test_sem2");
257       Assert.assertTrue(semaphore.tryAcquire(1, TimeUnit.SECONDS));
258       Assert.assertTrue(semaphore.tryAcquire(10, TimeUnit.SECONDS));
259       JdbcHeartBeat.stopHeartBeats();
260       server.stop();
261     } finally {
262       if (!tempDB.delete()) {
263         throw new IOException("Cannot delete " + tempDB);
264       }
265     }
266   }

272     Server server = Server.createTcpServer(new String[]{"-tcpPort", "9123", "-tcpAllowOthers"}).start();

274       File tempDB = File.createTempFile("test", "h2db");
275       tempDB.deleteOnExit();

277       JdbcDataSource ds = new JdbcDataSource();
278       ds.setURL(connStr);
279       ds.setUser("sa");
280       ds.setPassword("sa");
281       createSchemaObjects(ds);
282       JdbcSemaphore semaphore = new JdbcSemaphore(ds, "test_sem2", 1, true);
283       String o1 = org.spf4j.base.Runtime.jrun(DecentSemaphoreHandler.class, 10000000, connStr, "test_sem2").toString();
284       String o2 = org.spf4j.base.Runtime.jrun(DecentSemaphoreHandler.class, 10000000, connStr, "test_sem2").toString();
285       Assert.assertTrue(semaphore.tryAcquire(1, TimeUnit.SECONDS));
286       Assert.assertFalse(semaphore.tryAcquire(10, TimeUnit.SECONDS));
287       LOG.debug("P1: {}", o1);
288       LOG.debug("P2: {}", o2);
289       String[] nr1 = o1.split("\n");
290       String[] nr2 = o2.split("\n");
291       int totatl = nr1.length + nr2.length;
292       Set<String> numbers = new HashSet<>(totatl);
293       numbers.addAll(Arrays.asList(nr1));
294       numbers.addAll(Arrays.asList(nr2));
295       Assert.assertEquals(totatl, numbers.size());
296     } finally {
297       JdbcHeartBeat.stopHeartBeats();
298       server.stop();
299     }
