79    @Test
80    public void testVisit() throws IOException {
81      Schema recSchema = new Schema.Parser().parse(SCHEMA);
82      Schemas.visit(recSchema, new PrintingVisitor());
83  
84      String schemaStr = Resources.toString(Resources.getResource("SchemaBuilder.avsc"), StandardCharsets.US_ASCII);
85      Schema schema = new Schema.Parser().parse(schemaStr);
86  
87      Map<String, Schema> schemas = Schemas.visit(schema, new SchemasWithClasses());
88      Assert.assertThat(schemas, Matchers.hasValue(schema));
89  
90      Schema trimmed = Schemas.visit(recSchema, new CloningVisitor(recSchema));
91      Assert.assertNull(trimmed.getDoc());
92      Assert.assertNotNull(recSchema.getDoc());
93  
94      SchemaCompatibility.SchemaCompatibilityType compat
95              = SchemaCompatibility.checkReaderWriterCompatibility(trimmed, recSchema).getType();
96      Assert.assertEquals(SchemaCompatibility.SchemaCompatibilityType.COMPATIBLE, compat);
97      compat = SchemaCompatibility.checkReaderWriterCompatibility(recSchema, trimmed).getType();
98      Assert.assertEquals(SchemaCompatibility.SchemaCompatibilityType.COMPATIBLE, compat);
99  
100     Schema unmodifyable = Schemas.visit(recSchema, new ImmutableCloningVisitor(recSchema, false));
101     Assert.assertNotNull(unmodifyable.getDoc());
102     compat
103             = SchemaCompatibility.checkReaderWriterCompatibility(unmodifyable, recSchema).getType();
104     Assert.assertEquals(SchemaCompatibility.SchemaCompatibilityType.COMPATIBLE, compat);
105     compat = SchemaCompatibility.checkReaderWriterCompatibility(recSchema, unmodifyable).getType();
106     Assert.assertEquals(SchemaCompatibility.SchemaCompatibilityType.COMPATIBLE, compat);
107 
108     Schema schema1 = unmodifyable.getField("subNodes").schema().getElementType().getField("node").schema();
109     try {
110       schema1.addAlias("yahooo");
111       Assert.fail();
112     } catch (UnsupportedOperationException ex) {
113     }
114 
115     try {
116       schema1.setFields(Collections.EMPTY_LIST);
117       Assert.fail();
118     } catch (UnsupportedOperationException ex) {
119     }
120 
121   }

123   private static class PrintingVisitor implements SchemaVisitor {
124 
125     @Override
126     public SchemaVisitorAction visitTerminal(final Schema terminal) {
127       LOG.debug("Terminal: {}", terminal.getFullName());
128       return SchemaVisitorAction.CONTINUE;
129     }
130 
131     @Override
132     public SchemaVisitorAction visitNonTerminal(final Schema terminal) {
133       LOG.debug("NONTerminal start: {}", terminal.getFullName());
134       return SchemaVisitorAction.CONTINUE;
135     }
136 
137     @Override
138     public SchemaVisitorAction afterVisitNonTerminal(final Schema terminal) {
139       LOG.debug("NONTerminal end: {}", terminal.getFullName());
140       return SchemaVisitorAction.CONTINUE;
141     }
142   }

223             (diff) -> {
224               LOG.debug("Diff: {}", diff);
225               Schema atSchema = Schemas.getSubSchema(tSchema, diff.getPath());
226               Assert.assertNotNull("Path does not resolve: " + diff.getPath(), atSchema);
227               LOG.debug("At Schema {}", atSchema.getFullName());
228               Assert.assertNotNull(Schemas.project(tSchema, diff.getPath()).getFullName());
229               diffs.add(diff);
230             });
