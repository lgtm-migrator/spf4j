64    @Test
65    public void testCsvDecoder() throws IOException, CsvParseException {
66      String str = "RGF0ZSxWYWx1ZQoyMDE5LTAxLTAxLDQuNjA0CjIwMTgtMT"
67              + "AtMDEsNC42MDcKMjAxOC0wNy0wMSw0LjYwOQoyMDE4LTA0LTAxLDQuNjEyCg==";
68      byte[] input = Base64.getDecoder().decode(str);
69      Schema array = SchemaBuilder.array().items().record("test").fields()
70              .requiredString("Date")
71              .requiredString("Value")
72              .endRecord();
73      LOG.debug(new String(input, StandardCharsets.UTF_8));
74      CsvReader reader = Csv.CSV.readerILEL(
75              new InputStreamReader(new ByteArrayInputStream(input), StandardCharsets.UTF_8));
76      reader.readRow((c) -> { });
77      CsvDecoder decoder = new CsvDecoder(reader, array);
78      long nrRead = decoder.readArrayStart();
79      String lastValue = null;
80      while (nrRead > 0) {
81        for (int i = 0; i < nrRead; i++) {
82          LOG.debug(decoder.readString());
83          lastValue = decoder.readString();
84        }
85        nrRead = decoder.arrayNext();
86      }
87      Assert.assertEquals("4.612", lastValue);
88    }

91    @Test
92    public void testCsvEncoder() throws IOException, CsvParseException {
93      Schema rowschema = SchemaBuilder.record("CsvRow").fields()
94              .requiredBoolean("boolField")
95              .requiredBytes("bytesField")
96              .requiredInt("intField")
97              .requiredLong("longField")
98              .requiredString("stringField")
99              .requiredDouble("doubleField")
100             .name("recordField").type(PackageInfo.getClassSchema()).noDefault()
101             .nullableString("nullableString", null)
102             .name("enumField").type(Schema.createEnum("myEnum", "", "", Arrays.asList("e1", "e2"))).noDefault()
103             .endRecord();
104 
105     List<GenericRecord> list = testArray(rowschema);
106 
107     Schema csvSchema = Schema.createArray(rowschema);
108 
109     CharSeparatedValues csv = new CharSeparatedValues(',');
110     StringWriter writeTo = new StringWriter();
111 
112     CsvEncoder encoder = new CsvEncoder(csv.writer(writeTo), csvSchema);
113     encoder.writeHeader();
114     DatumWriter writer = new GenericDatumWriter(csvSchema);
115     writer.write(list, encoder);
116     encoder.flush();
117     LOG.debug("written:", writeTo);
118     StringReader reader = new StringReader(writeTo.toString());
119     CsvDecoder decoder = new CsvDecoder(csv.reader(reader), csvSchema);
120     decoder.skipHeader();
121     GenericDatumReader dr = new GenericDatumReader(csvSchema);
122     Object parsed = dr.read(null, decoder);
123     LOG.debug("parser:", parsed);
124   }

126   private static List<GenericRecord> testArray(final Schema rowschema) {
127     Schema enumSchema = rowschema.getField("enumField").schema();
128     GenericRecord rec1 = new GenericData.Record(rowschema);
129     rec1.put("boolField", Boolean.TRUE);
130     rec1.put("bytesField", ByteBuffer.wrap(new byte[] {1, 2, 3}));
131     rec1.put("intField", 2);
132     rec1.put("longField", 5L);
133     rec1.put("stringField", "test1");
134     rec1.put("doubleField", Math.PI);
135     rec1.put("recordField", new PackageInfo("a", "b"));
136     rec1.put("enumField", new GenericData.EnumSymbol(enumSchema, "e1"));
137     GenericRecord rec2 = new GenericData.Record(rowschema);
138     rec2.put("boolField", Boolean.FALSE);
139     rec2.put("bytesField", ByteBuffer.wrap(new byte[] {1, 2, 3}));
140     rec2.put("intField", 2);
141     rec2.put("longField", 5L);
142     rec2.put("stringField", "test1");
143     rec2.put("doubleField", Math.PI);
144     rec2.put("recordField", new PackageInfo("c", "d"));
145     rec2.put("nullableString", "nstr");
146     rec2.put("enumField", new GenericData.EnumSymbol(enumSchema, "e2"));
147     return Arrays.asList(rec1, rec2);
148   }

174   @Test
175   public void testEncodeDecodeSpecific() throws IOException, CsvParseException {
176     ByteArrayOutputStream bos = new ByteArrayOutputStream();
177     Schema aSchema = Schema.createArray(DemoRecordInfo.SCHEMA$);
178     CsvEncoder csvEncoder = new CsvEncoder(Csv.CSV.writer(
179             new OutputStreamWriter(bos, StandardCharsets.UTF_8)), aSchema);
180     csvEncoder.writeHeader();
181     DatumWriter writer = new GenericDatumWriter(aSchema);
182     List<DemoRecordInfo> testRecords = testRecords();
183     writer.write(testRecords, csvEncoder);
184     csvEncoder.flush();
185     LOG.debug("serialized", bos.toString("UTF8"));
186 
187     ByteArrayInputStream bis = new ByteArrayInputStream(bos.toByteArray());
188     DecodedSchema ds = CsvDecoder.tryDecodeSchema(bis, aSchema);
189     Decoder decoder = ds.getDecoder();
190     SpecificDatumReader reader = new SpecificDatumReader(ds.getSchema());
191     Object read = reader.read(reader, decoder);
192     LOG.debug("deserialized", read);
193     Assert.assertEquals(testRecords, read);
194   }
