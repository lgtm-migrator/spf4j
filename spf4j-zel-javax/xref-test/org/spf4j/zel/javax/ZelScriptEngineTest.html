47    @Test
48    public void testEngine() throws ScriptException {
49      ScriptEngineManager em = new ScriptEngineManager();
50      ScriptEngine se = em.getEngineByName("zel");
51      Bindings bindings = se.createBindings();
52      bindings.put("a", 1);
53      bindings.put("b", 2);
54      Assert.assertEquals(3, se.eval("a + b", bindings));
55    }
56  
57    @Test
58    public void testEngine2() throws ScriptException, NoSuchMethodException {
59      ScriptEngineManager em = new ScriptEngineManager();
60      ScriptEngine se = em.getEngineByName("zel");
61      CompiledScript cs = ((Compilable) se).compile("a < b");
62      Assert.assertTrue((Boolean) ((Invocable) cs).invokeFunction(null, 1, 2));
63    }
64  
65    @Test
66    public void testEngine3() throws ScriptException, NoSuchMethodException {
67      ScriptEngineManager em = new ScriptEngineManager();
68      ScriptEngine se = em.getEngineByName("zel");
69      Bindings bindings = se.createBindings();
70      bindings.put("abort", new JavaMethodCall(RetryDecision.class, "abort"));
71      bindings.put("retry", new JavaMethodCall(RetryDecision.class, "retry"));
72      se.setBindings(bindings, ScriptContext.ENGINE_SCOPE);
73      CompiledScript cs = ((Compilable) se).compile(
74              "startTime;endEndTime;  if (object > 3) {abort()} else {retry(10L, callable)}");
75      Callable<Void> nothing = Callables.from(Runnables.doNothing());
76      Assert.assertEquals(RetryDecision.abort(),
77              (RetryDecision) ((Invocable) cs).invokeFunction(null, 1, 2, 10, nothing));
78      Assert.assertEquals(10L,
79              ((RetryDecision) ((Invocable) cs).invokeFunction(null, 1, 2, 2, nothing)).getDelayNanos());
80    }
81  
82    @Test
83    public void testRetryPolicyWithZEL() throws InvalidRetryPolicyException {
84      org.spf4j.failsafe.RetryPolicy<Integer, Callable<Integer>> policy =
85              RetryPolicies.create(new RetryPolicy(10,
86             Collections.singletonList(new RetryRule("custom",
87                     new ScriptedRetryPredicateSupplier("java", "zel",
88                     "startTime;endEndTime; (object > 3) ? decision.abort() : decision.retry(10L, callable)", ""))),
89              Collections.EMPTY_MAP));
90       long nanoTime = System.nanoTime();
91      RetryPredicate<Integer, Callable<Integer>> pred =  policy.getRetryPredicate(nanoTime, nanoTime + 10000000000L);
92      Callable<Integer> nothing = Callables.constant(5);
93      Assert.assertEquals(RetryDecision.abort(), pred.getDecision(10, nothing));
94      RetryDecision<Integer, Callable<Integer>> decision = pred.getDecision(1, nothing);
95      Assert.assertEquals(10L, decision.getDelayNanos());
96    }
