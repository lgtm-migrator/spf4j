57  /**
58   * Improved implementation of the spf4j profiler to get extra profiling
59   * data granularity to be able to correlate to iteration number.
60   *
61   * @author zoly
62   */
63  @SuppressFBWarnings("PATH_TRAVERSAL_IN")
64  public final class Spf4jJmhProfiler implements InternalProfiler {
65  
66    /**
67     * Sampling period
68     */
69    private static final int SAMPLE_PERIOD_MSEC = Integer.getInteger("jmh.stack.period", 10);
70  
71    private static final String DUMP_FOLDER = System.getProperty("jmh.stack.profiles", org.spf4j.base.Runtime.USER_DIR);
72  
73    private static final Sampler SAMPLER = new Sampler(SAMPLE_PERIOD_MSEC, Integer.MAX_VALUE,
74            (t) -> new FastStackCollector(false, true, new Thread[] {t}));
75  
76    private static final AtomicInteger MEASUREMENT_ITERATION_COUNTER = new AtomicInteger(1);
77  
78    private static final AtomicInteger WARMUP_ITERATION_COUNTER = new AtomicInteger(1);
79  
80    private static volatile String benchmarkName;
81  
82    public static String benchmarkName() {
83      return benchmarkName;
84    }
85  
86    @Override
87    @SuppressFBWarnings("ST_WRITE_TO_STATIC_FROM_INSTANCE_METHOD")
88    public void beforeIteration(final BenchmarkParams benchmarkParams, final IterationParams iterationParams) {
89      benchmarkName = benchmarkParams.id();
90      SAMPLER.start();
91    }
92  
93    @Override
94    @Nonnull
95    public Collection<? extends Result> afterIteration(final BenchmarkParams benchmarkParams,
96            final IterationParams iterationParams, final IterationResult ir) {
97      try {
98        SAMPLER.stop();
99      } catch (InterruptedException ex) {
100       Thread.currentThread().interrupt();
101       return Collections.EMPTY_LIST;
102     }
103     Map<String, SampleNode> c = SAMPLER.getStackCollectionsAndReset();
104     SampleNode collected;
105     if (c.isEmpty()) {
106       collected = null;
107     } else {
108       collected = c.values().iterator().next();
109     }
110     String iterationId;
111     IterationType itType = iterationParams.getType();
112     switch (itType) {
113       case MEASUREMENT:
114         iterationId = "m" + MEASUREMENT_ITERATION_COUNTER.getAndIncrement();
115         break;
116       case WARMUP:
117         iterationId = "w" + WARMUP_ITERATION_COUNTER.getAndIncrement();
118         break;
119       default:
120         throw new IllegalStateException("Unknown type of iteration " + itType);
121     }
122     return Collections.singletonList(new StackResult(collected, benchmarkParams.id(), iterationId));
123   }
124 
125   @Override
126   public String getDescription() {
127     return "spf4j stack sampler";
128   }
129 
130   public static final class StackResult extends Result<StackResult> {
131 
132     private static final long serialVersionUID = 1L;
133 
134     private final String benchmark;
135 
136     private final File perfDataFile;
137 
138     private final String id;
139 
140     public StackResult(@Nullable final SampleNode samples, final String benchmark, final String iterationId) {
141       super(ResultRole.SECONDARY, "@stack", of(Double.NaN), "---", AggregationPolicy.AVG);
142       this.id = iterationId;
143       this.benchmark = benchmark;
144       if (samples != null) {
145         File file = new File(DUMP_FOLDER + '/' + benchmark + '_' + iterationId + ".ssdump2");
146         try {
147           Converter.save(file, samples);
148           this.perfDataFile = file;
149         } catch (IOException ex) {
150           throw new Spf4jProfilerException("Failed to persist profile data to " + file, ex);
151         }
152       } else {
153         this.perfDataFile = null;
154       }
155     }
156 
157     public String getId() {
158       return id;
159     }
160 
161     @Nullable
162     public SampleNode getSamples() throws IOException {
163       if (this.perfDataFile == null) {
164         return null;
165       } else {
166         return Converter.load(perfDataFile);
167       }
168     }
169 
170     public String getBenchmark() {
171       return benchmark;
172     }
173 
174     @Override
175     protected Aggregator<StackResult> getThreadAggregator() {
176       return new StackAggregator();
177     }
178 
179     @Override
180     protected Aggregator<StackResult> getIterationAggregator() {
181       return new StackAggregator();
182     }
183 
184     @Override
185     public String toString() {
186       return "<delayed till summary>";
187     }
188 
189   }
190 
191   public static final class StackAggregator implements Aggregator<StackResult> {
192 
193     private static final  AtomicSequence SEQ = new AtomicSequence(0);
194 
195     @Override
196     public StackResult aggregate(final Collection<StackResult> results) {
197       if (results.isEmpty()) {
198         throw new IllegalArgumentException("Nothig to aggregate: " + results);
199       }
200       Iterator<StackResult> it = results.iterator();
201       if (results.size() == 1) {
202         return it.next();
203       }
204       StackResult result = it.next();
205       StringBuilder aggId = new StringBuilder();
206       aggId.append("aggregation_");
207       SampleNode agg;
208       try {
209         agg = result.getSamples();
210         aggId.append(result.getId());
211         final String benchmark = result.getBenchmark();
212         while (it.hasNext()) {
213           result = it.next();
214           String obenchmark = result.getBenchmark();
215           if (!benchmark.equals(obenchmark)) {
216             throw new UnsupportedOperationException("Should not aggregate " + benchmark + " with " + obenchmark);
217           }
218           SampleNode sss = result.getSamples();
219           if (agg == null) {
220             agg = sss;
221           } else if (sss != null) {
222             agg = SampleNode.aggregate(agg, sss);
223           }
224           aggId.append('_').append(result.getId());
225         }
226         truncateId(aggId, 100);
227         return new StackResult(agg, benchmark, aggId.toString());
228       } catch (IOException ex) {
229         throw new Spf4jProfilerException("Failed to aggregate " + results, ex);
230       }
231     }
232 
233     private static void truncateId(final StringBuilder aggId, final int size) {
234       int length = aggId.length();
235       if (length > size) {
236         CharSequence id = new UIDGenerator(SEQ).next();
237         aggId.setLength(length - id.length() - 1);
238         aggId.append('_');
239         aggId.append(id);
240       }
241     }

243   }
