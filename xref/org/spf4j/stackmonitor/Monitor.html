32  package org.spf4j.stackmonitor;
33  
34  import java.io.IOException;
35  import java.lang.reflect.InvocationTargetException;
36  import org.kohsuke.args4j.CmdLineException;
37  import org.kohsuke.args4j.CmdLineParser;
38  import org.kohsuke.args4j.Option;
39  import org.spf4j.base.AbstractRunnable;
40  import org.spf4j.base.SysExits;
41  import static org.spf4j.stackmonitor.Sampler.DEFAULT_SS_DUMP_FILE_NAME_PREFIX;
42  import static org.spf4j.stackmonitor.Sampler.DEFAULT_SS_DUMP_FOLDER;
43  
44  /**
45   * @author zoly
46   */
47  public final class Monitor {
48  
49      private static class Options {
50  
51          @Option(name = "-df", usage = "dump folder")
52          private String dumpFolder = DEFAULT_SS_DUMP_FOLDER;
53  
54          @Option(name = "-dp", usage = "dump file prefix")
55          private String dumpFilePrefix = DEFAULT_SS_DUMP_FILE_NAME_PREFIX;
56  
57          @Option(name = "-main", usage = "the main class name", required = true)
58          private String mainClass;
59  
60          @Option(name = "-si", usage = "the stack sampling interval in milliseconds")
61          private int sampleInterval = 100;
62  
63          @Option(name = "-di", usage = "the stack dump to file interval in milliseconds")
64          private int dumpInterval = 3600000;
65  
66          @Option(name = "-ss", usage = "start the stack sampling. (can also be done manually via jmx)")
67          private boolean startSampler = false;
68  
69  
70      }
71  
72      private Monitor() { }
73  
74      @SuppressWarnings("checkstyle:regexp")
75      public static void main(final String[] args)
76              throws ClassNotFoundException, NoSuchMethodException, IllegalAccessException,
77              InvocationTargetException  {
78  
79          int sepPos = args.length;
80          for (int i = 0; i < args.length; i++) {
81              if ("--".equals(args[i])) {
82                  sepPos = i;
83                  break;
84              }
85          }
86          String[] newArgs;
87          String[] ownArgs;
88          if (sepPos == args.length) {
89              newArgs = new String[0];
90              ownArgs = args;
91          } else {
92              newArgs = new String[args.length - sepPos - 1];
93              ownArgs = new String[sepPos];
94              System.arraycopy(args, sepPos + 1, newArgs, 0, newArgs.length);
95              System.arraycopy(args, 0, ownArgs, 0, sepPos);
96          }
97          Options options = new Options();
98          CmdLineParser parser = new CmdLineParser(options);
99          try {
100             parser.parseArgument(ownArgs);
101         } catch (CmdLineException e) {
102             System.err.println("Error: " + e.getMessage() + "\nUsage:");
103             parser.printUsage(System.err);
104             System.exit(SysExits.EX_USAGE.exitCode());
105         }
106 
107         final Sampler sampler = new Sampler(options.sampleInterval, options.dumpInterval,
108                 (t) -> new FastStackCollector(false, true, new Thread[]{t}),
109                 options.dumpFolder, options.dumpFilePrefix);
110         Runtime.getRuntime().addShutdownHook(new Thread(new AbstractRunnable() {
111 
112             @Override
113             public void doRun() throws InterruptedException, IOException {
114                 sampler.stop();
115                 sampler.dumpToFile();
116                 sampler.dispose();
117             }
118 
119         }, "Sampling report"));
120         sampler.registerJmx();
121 
122         if (options.startSampler) {
123             sampler.start();
124         }
125         Class.forName(options.mainClass).getMethod("main", String[].class).invoke(null, (Object) newArgs);
126         System.exit(SysExits.OK.exitCode());
127     }
128 
129 
130 }
