154   public Sampler(final int sampleTimeMillis, final int dumpTimeMillis, final SamplerSupplier collector,
155           final ProfilePersister persister) {
156     stopped = true;
157     if (sampleTimeMillis < 1) {
158       throw new IllegalArgumentException("Invalid sample time " + sampleTimeMillis);
159     }
160     this.sampleTimeNanos = TimeUnit.MILLISECONDS.toNanos(sampleTimeMillis);
161     if (sampleTimeNanos < 0) {
162       throw new IllegalArgumentException("Invalid sample time " + sampleTimeMillis);
163     }
164     this.dumpTimeNanos = TimeUnit.MILLISECONDS.toNanos(dumpTimeMillis);
165     this.stackCollectorSupp = collector;
166     this.persister = persister;
167   }

170   public static synchronized Sampler getSampler(final int sampleTimeMillis,
171           final int dumpTimeMillis,
172           final File dumpFolder, final String dumpFilePrefix) throws InterruptedException, IOException {
173     return getSampler(sampleTimeMillis, dumpTimeMillis,
174             (t) -> new FastStackCollector(false, true, new Thread[]{t}), dumpFolder, dumpFilePrefix);
175   }

177   /**
178    * Gets a sampler.
179    * This method will always creates a new sampler instance and will dispose of any sampler
180    * instances previously created with this method.
181    * There can only be one sampler instance running at a time.
182    * @param sampleTimeMillis
183    * @param dumpTimeMillis
184    * @param collector
185    * @param dumpFolder
186    * @param dumpFilePrefix
187    * @return
188    * @throws InterruptedException
189    */
190   @SuppressFBWarnings("MS_EXPOSE_REP")
191   public static synchronized Sampler getSampler(final int sampleTimeMillis,
192           final int dumpTimeMillis, final SamplerSupplier collector,
193           final File dumpFolder, final String dumpFilePrefix) throws InterruptedException, IOException {
194     if (instance != null) {
195       instance.dispose();
196     }
197     try {
198       instance = new Sampler(sampleTimeMillis, dumpTimeMillis, collector,
199               dumpFolder.getCanonicalFile(), dumpFilePrefix);
200     } catch (IOException ex) {
201       throw new UncheckedIOException(ex);
202     }
203     instance.registerJmx();
204     return instance;
205   }

211   @JmxExport(description = "start stack sampling")
212   public void start() {
213     synchronized (sync) {
214       if (stopped) {
215         stopped = false;
216         final long stNanos = sampleTimeNanos;
217         samplerFuture = DefaultExecutor.INSTANCE.submit(new AbstractRunnable("SPF4J-Sampling-Thread") {
218 
219           @SuppressWarnings("SleepWhileInLoop")
220           @SuppressFBWarnings({"MDM_THREAD_YIELD", "PREDICTABLE_RANDOM"})
221           @Override
222           public void doRun() {
223             lastDumpTimeNanos = TimeSource.nanoTime();
224             synchronized (sync) {
225               stackCollector = stackCollectorSupp.get(Thread.currentThread());
226             }
227             final long lDumpTimeNanos = dumpTimeNanos;
228             final ThreadLocalRandom random = ThreadLocalRandom.current();
229             long dumpCounterNanos = 0;
230             long sleepTimeNanos = 0;
231             long halfStNanos = stNanos / 2;
232             if (halfStNanos == 0) {
233               halfStNanos = 1;
234             }
235             long maxSleeepNanos = stNanos + halfStNanos;
236             while (true) {
237               try {
238                 synchronized (sync) {
239                   stackCollector.sample();
240                   if (stopped) {
241                     break;
242                   }
243                 }
244                 dumpCounterNanos += sleepTimeNanos;
245                 if (dumpCounterNanos >= lDumpTimeNanos) {
246                   long nanosSinceLastDump = TimeSource.nanoTime() - lastDumpTimeNanos;
247                   if (nanosSinceLastDump >= lDumpTimeNanos) {
248                     dumpCounterNanos = 0;
249                     File dumpFile = dumpToFile();
250                     if (dumpFile != null) {
251                       Logger.getLogger(Sampler.class.getName())
252                               .log(Level.INFO, "Stack samples written to {0}", dumpFile);
253                     }
254                   } else {
255                     dumpCounterNanos = nanosSinceLastDump;
256                   }
257                 }
258                 sleepTimeNanos = random.nextLong(halfStNanos, maxSleeepNanos);
259                 TimeUnit.NANOSECONDS.sleep(sleepTimeNanos);
260               } catch (InterruptedException ex) {
261                 Thread.currentThread().interrupt();
262                 return;
263               } catch (IOException | RuntimeException ex) {
264                 Logger.getLogger(Sampler.class.getName()).log(Level.SEVERE,
265                         "Exception encountered while samplig, will continue sampling", ex);
266               }
267             }
268           }
269         });
270       } else {
271         throw new IllegalStateException("Sampling can only be started once for " + this);
272       }
273     }
274   }

289   private static class ProfileData {

291     private final Instant to;
292     private final Map<String, SampleNode> samples;

294     ProfileData(final Instant from, final Instant to, final Map<String, SampleNode> samples) {

313   @Nullable
314   private ProfileData getAndResetProfileSamples() {
315     long fromNanos;
316     Map<String, SampleNode> collections;
317     long nowNanos;
318     synchronized (sync) {
319       if (stackCollector == null) {
320         return null;
321       }
322       collections = stackCollector.getCollectionsAndReset();
323       if (collections.isEmpty()) {
324         return null;
325       }
326       fromNanos = lastDumpTimeNanos;
327       nowNanos = TimeSource.nanoTime();
328       lastDumpTimeNanos = nowNanos;
329     }
330     Timing currentTiming = Timing.getCurrentTiming();
331     return new ProfileData(currentTiming.fromNanoTimeToInstant(fromNanos),
332             currentTiming.fromNanoTimeToInstant(nowNanos), collections);
333 
334   }

336   @JmxExport(description = "save stack samples to file")
337   @Nullable
338   public File dumpToFile() throws IOException {
339     ProfileData data = getAndResetProfileSamples();
340     if (data == null) {
341       return null;
342     }
343     return persister.persist(data.getSamples(), null, data.getFrom(), data.getTo()).toFile();
344   }

346   /**
347    * Dumps the sampled stacks to file. the collected samples are reset
348    *
349    * @param id - id will be added to file name returns the name of the file.
350    * @return - the file name where the data was persisted or null if there was no data to persist. The file name
351    * will be of format: [filePrefix](_[id])?_["yyyyMMdd'T'HHmmssSSSz]_["yyyyMMdd'T'HHmmssSSSz]_[label].ssdump2(.gz)?
352    * or [filePrefix](_[id])?_["yyyyMMdd'T'HHmmssSSSz]_["yyyyMMdd'T'HHmmssSSSz].ssdump3(.gz)?
353    * @throws IOException - io issues while persisting data.
354    */
355   @JmxExport(value = "dumpToSpecificFile", description = "save stack samples to file")
356   @Nullable
357   public File dumpToFile(
358           @JmxExport(value = "fileID", description = "the ID that will be part of the file name")
359           @Nullable final String id) throws IOException {
360     ProfileData data = getAndResetProfileSamples();
361     if (data == null) {
362       return null;
363     }
364     return persister.persist(data.getSamples(), id, data.getFrom(), data.getTo()).toFile();
365   }

367   /**
368    * Dump  collected samples to disk.
369    * @param destinationFolder the destination folder.
370    * @param pbaseFileName file name base, will be URLEncoded.
371    * The appropriate extension will be added depending on situation.
372    * either ssdump2 or ssdump3 will be used depending on the number of sample aggregate groups.
373    * additionally .gz will be added if compression is enabled.
374    * in case of ssdump2 the label will be appended to the file name like _[label].ssdump2
375    *
376    * @return the saved file, or null if no collected samples to save.
377    * @throws IOException
378    */
379   @Nullable
380   @SuppressFBWarnings("PATH_TRAVERSAL_IN") // not possible the provided ID is validated for path separators.
381   public File dumpToFile(@Nonnull final File destinationFolder, final String pbaseFileName) throws IOException {
382     ProfileData data = getAndResetProfileSamples();
383     if (data == null) {
384       return null;
385     }
386     return persister.withBaseFileName(destinationFolder.toPath(), pbaseFileName)
387            .persist(data.getSamples(), null, data.getFrom(), data.getTo()).toFile();
388   }

390   @JmxExport(description = "stop stack sampling")
391   public void stop() throws InterruptedException {
392     Future<?> toCancel = null;
393     synchronized (sync) {
394       if (!stopped) {
395         stopped = true;
396         toCancel = samplerFuture;
397       }
398     }
399     if (toCancel != null) {
400       try {
401         toCancel.get(dumpTimeNanos * 3, TimeUnit.NANOSECONDS);
402       } catch (TimeoutException ex) {
403         toCancel.cancel(true);
404         throw new Spf4jProfilerException(ex);
405       } catch (ExecutionException ex) {
406         throw new Spf4jProfilerException(ex);
407       }
408     }
409   }

456   @PreDestroy
457   public void dispose() throws InterruptedException, IOException {
458     try {
459       stop();
460     } finally {

463     }
