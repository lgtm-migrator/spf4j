32  package org.spf4j.perf.aspects;
33  
34  import com.google.common.cache.CacheLoader;
35  import com.google.common.cache.LoadingCache;
36  import java.util.concurrent.TimeUnit;
37  import org.spf4j.perf.MeasurementRecorderSource;
38  import org.aspectj.lang.ProceedingJoinPoint;
39  import org.aspectj.lang.annotation.Around;
40  import org.aspectj.lang.annotation.Aspect;
41  import org.slf4j.Logger;
42  import org.slf4j.LoggerFactory;
43  import org.spf4j.concurrent.UnboundedLoadingCache;
44  import org.spf4j.annotations.PerformanceMonitor;
45  import org.spf4j.annotations.RecorderSourceInstance;
46  import org.spf4j.base.TimeSource;
47  
48  /**
49   * Aspect that measures execution time and does performance logging
50   * for all methods annotated with: PerformanceMonitor annotation.
51   *
52   * @author zoly
53   */
54  @Aspect
55  public final class PerformanceMonitorAspect {
56  
57      private static final Logger LOG = LoggerFactory.getLogger(PerformanceMonitorAspect.class);
58      private static final LoadingCache<Class<? extends RecorderSourceInstance>, MeasurementRecorderSource> REC_SOURCES =
59              new UnboundedLoadingCache<>(32,
60                      new CacheLoader<Class<? extends RecorderSourceInstance>, MeasurementRecorderSource>() {
61          @Override
62          public MeasurementRecorderSource load(final Class<? extends RecorderSourceInstance> key) throws Exception {
63              return (MeasurementRecorderSource) key.getField("INSTANCE").get(null);
64          }
65      });
66      @Around(value = "@annotation(annot)"
67              + " && execution(@org.spf4j.annotations.PerformanceMonitor * *(..))",
68              argNames = "pjp,annot")
69      public Object performanceMonitoredMethod(final ProceedingJoinPoint pjp, final PerformanceMonitor annot)
70              throws Throwable {
71          final long start = TimeSource.nanoTime();
72          Object result = pjp.proceed();
73          final long elapsedNanos = TimeSource.nanoTime() - start;
74          final long elapsed = TimeUnit.NANOSECONDS.toMillis(elapsedNanos);
75          MeasurementRecorderSource mrs = REC_SOURCES.getUnchecked(annot.recorderSource());
76          mrs.getRecorder(pjp.toLongString()).record(elapsed);
77          final long warnThresholdMillis = annot.warnThresholdMillis();
78          if (elapsed > warnThresholdMillis) {
79              final long errorThresholdMillis = annot.errorThresholdMillis();
80              if (elapsed > errorThresholdMillis) {
81                  LOG.error("Execution time  {} ms for {} exceeds error threshold of {} ms, arguments {}",
82                              elapsed, pjp.toShortString(), errorThresholdMillis, pjp.getArgs());
83              } else {
84                  LOG.warn("Execution time  {} ms for {} exceeds warning threshold of {} ms, arguments {}",
85                              elapsed, pjp.toShortString(), warnThresholdMillis, pjp.getArgs());
86              }
87          } else {
88              if (annot.defaultInfoLog()) {
89                  LOG.info("Execution time {} ms for {}, arguments {}",
90                          elapsed, pjp.toShortString(), pjp.getArgs());
91              } else {
92                  LOG.debug("Execution time {} ms for {}, arguments {}",
93                          elapsed, pjp.toShortString(), pjp.getArgs());
94              }
95          }
96          return result;
97      }
98  }
