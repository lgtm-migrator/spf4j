53      private final long location;

55      private long firstDataFragment;
56      private long lastDataFragment;

64      TSTable(final String tableName, final byte[] tableMetaData,
65              final String[] columnNames, final byte[][] columnMetaData,
66              final int sampleTime, final long location) {
67          this.location = location;
68          this.nextTableInfo = 0;
69          this.firstDataFragment = 0;
70          this.lastDataFragment = 0;
71          this.tableName = tableName;
72          this.sampleTime = sampleTime;
73          this.tableMetaData = tableMetaData;
74          this.columnNames = columnNames;
75          this.columnMetaData = columnMetaData;
76          this.nameToIndex = new HashMap<>(columnNames.length + columnNames.length / 3);
77          for (int i = 0; i < columnNames.length; i++) {
78              this.nameToIndex.put(columnNames[i], i);
79          }
80      }
81

88      TSTable(final TSTable copyOf) {
89          this.location = copyOf.location;
90          this.nextTableInfo = copyOf.nextTableInfo;
91          this.firstDataFragment = copyOf.firstDataFragment;
92          this.lastDataFragment = copyOf.lastDataFragment;
93          this.tableName = copyOf.tableName;
94          this.sampleTime = copyOf.sampleTime;
95          this.tableMetaData = copyOf.tableMetaData;
96          this.columnNames = copyOf.columnNames;
97          this.columnMetaData = copyOf.columnMetaData;
98          this.nameToIndex = copyOf.nameToIndex;
99      }

102     /**
103      * Read table structure from file.
104      *
105      * @param raf
106      * @throws IOException
107      */
108     TSTable(final RandomAccessFile raf) throws IOException {
109         this(raf, raf.getFilePointer());
110     }
111 
112     TSTable(final RandomAccessFile raf, final long location) throws IOException {
113         this.location = location;
114         raf.seek(location);
115         this.nextTableInfo = raf.readLong();
116         this.firstDataFragment = raf.readLong();
117         this.lastDataFragment = raf.readLong();
118         this.tableName = raf.readUTF();
119         this.sampleTime = raf.readInt();
120         int grMetaSize = raf.readInt();
121         this.tableMetaData = new byte[grMetaSize];
122         raf.readFully(tableMetaData);
123         int nrColumns = raf.readShort();
124         columnNames = new String[nrColumns];
125         this.nameToIndex = new HashMap<>(nrColumns + nrColumns / 3);
126         for (int i = 0; i < columnNames.length; i++) {
127             String colName = raf.readUTF();
128             columnNames[i] = colName;
129             this.nameToIndex.put(colName, i);
130         }
131         columnMetaData = new byte[raf.readInt()][];
132         for (int i = 0; i < columnMetaData.length; i++) {
133             int metaLength = raf.readInt();
134             byte[] colMetaData = new byte[metaLength];
135             raf.readFully(colMetaData);
136             columnMetaData[i] = colMetaData;
137         }
138     }

142     void writeTo(final DataOutput dos) throws IOException {
143         dos.writeLong(nextTableInfo);
144         dos.writeLong(firstDataFragment);
145         dos.writeLong(lastDataFragment);
146         dos.writeUTF(tableName);
147         dos.writeInt(sampleTime);
148         dos.writeInt(tableMetaData.length);
149         dos.write(tableMetaData);
150         dos.writeShort(columnNames.length);
151         for (String columnName : columnNames) {
152             dos.writeUTF(columnName);
153         }
154         dos.writeInt(columnMetaData.length);
155         for (byte[] colMeta : columnMetaData) {
156             dos.writeInt(colMeta.length);
157             dos.write(colMeta);
158         }
159 
160     }
161 
162     void writeTo(final RandomAccessFile raf) throws IOException {
163         try (ByteArrayBuilder bos = new ByteArrayBuilder()) {
164             DataOutput dos = new DataOutputStream(bos);
165             writeTo(dos);
166 
167             raf.seek(location);
168             raf.write(bos.getBuffer(), 0, bos.size());
169         }
170     }

177         this.nextTableInfo = pnextColumnInfo;
178 
179         raf.seek(location);

181     }

183     void setFirstDataFragment(final long pfirstDataFragment, final RandomAccessFile raf) throws IOException {
184         this.firstDataFragment = pfirstDataFragment;
185         raf.seek(location + 8);
186         raf.writeLong(firstDataFragment);
187     }

189     void setLastDataFragment(final long plastDataFragment, final RandomAccessFile raf) throws IOException {
190         this.lastDataFragment = plastDataFragment;
191         raf.seek(location + 16);
192         raf.writeLong(lastDataFragment);
193     }

219     public String[] getColumnMetaDataAsStrings() {
220         String[] result = new String[columnMetaData.length];

223         }
224         return result;
225     }

227     public int getColumnIndex(final String columnName) {
228         Integer result = this.nameToIndex.get(columnName);
229         if (result == null) {
230             return -1;
231         } else {
232             return result;
233         }
234     }
