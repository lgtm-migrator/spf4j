32  package org.spf4j.perf;
33  
34  import com.google.common.annotations.Beta;
35  import com.google.common.collect.Iterables;
36  import java.io.IOException;
37  import java.time.Instant;
38  import java.util.Collection;
39  import java.util.concurrent.TimeUnit;
40  import java.util.function.Predicate;
41  import javax.annotation.Nullable;
42  import org.apache.avro.Schema;
43  import org.spf4j.base.avro.AvroCloseableIterable;
44  import org.spf4j.tsdb2.TableDefs;
45  import org.spf4j.tsdb2.avro.Observation;
46  
47  /**
48   *
49   * @author Zoltan Farkas
50   */
51  public interface MeasurementStoreQuery {
52  
53    Collection<Schema> getMeasurements(Predicate<String> filter) throws IOException;
54  
55    AvroCloseableIterable<Observation> getObservations() throws IOException;
56  
57    default AvroCloseableIterable<Observation> getObservations(final Schema measurement,
58            @Nullable final Instant from, @Nullable final Instant to) throws IOException {
59      @SuppressWarnings("unchecked")
60      Collection<Long> mids = (Collection<Long>) measurement.getObjectProp(TimeSeriesRecord.IDS_PROP);
61      @SuppressWarnings("unchecked")
62      final long fromMs;
63      if (from == null) {
64        fromMs = Long.MIN_VALUE;
65      } else {
66        fromMs = from.toEpochMilli();
67      }
68      final long toMs;
69      if (to == null) {
70        toMs = Long.MAX_VALUE;
71      } else {
72        toMs = to.toEpochMilli();
73      }
74      AvroCloseableIterable<Observation> observations = getObservations();
75      Iterable<Observation> filtered = Iterables.filter(observations, (Observation row) -> {
76        long ts = row.getRelTimeStamp();
77        return ts >= fromMs && ts <= toMs && mids.contains(row.getTableDefId());
78      });
79      return AvroCloseableIterable.from(filtered, observations, measurement);
80    }
81  
82    default AvroCloseableIterable<Observation> getAggregatedObservations(final Schema measurement,
83            final Instant from, final Instant to, final int aggFreq,  final TimeUnit tu) throws IOException {
84      long aggTime = tu.toMillis(aggFreq);
85      int frequencyMillis = TimeSeriesRecord.getFrequencyMillis(measurement);
86      AvroCloseableIterable<Observation> observations = getObservations(measurement, from, to);
87       return AvroCloseableIterable.from(() -> new TimeSeriesAggregatingIterator<>(observations,
88               Observation::getRelTimeStamp,
89              (a, b) -> TimeSeriesRecord.accumulateObservations(measurement, a, b), aggTime, frequencyMillis),
90              observations, measurement);
91    }
92  
93    /**
94     * Query measurement data.
95     * @param measurement
96     * @param from
97     * @param to
98     * @return data iterable
99     * @throws IOException
100    */
101   default AvroCloseableIterable<TimeSeriesRecord> getMeasurementData(final Schema measurement,
102           final Instant from, final Instant to) throws IOException {
103     AvroCloseableIterable<Observation> observations = getObservations(measurement, from, to);
104     Iterable<TimeSeriesRecord> tsr = Iterables.transform(observations,
105              obs -> TableDefs.toRecord(measurement, obs));
106     return AvroCloseableIterable.from(tsr, observations, measurement);
107   }
108 
109   @Beta
110   default AvroCloseableIterable<TimeSeriesRecord> getAggregatedMeasurementData(
111           final Schema measurement,
112           final Instant from, final Instant to,
113           final long aggFreq,  final TimeUnit tu) throws IOException {
114     long aggTime = tu.toMillis(aggFreq);
115     long frequencyMillis = TimeSeriesRecord.getFrequencyMillis(measurement);
116     AvroCloseableIterable<TimeSeriesRecord> iterable = getMeasurementData(measurement, from, to);
117     return AvroCloseableIterable.from(() -> new TimeSeriesAggregatingIterator<>(iterable,
118             (TimeSeriesRecord rec) -> rec.getTimeStamp().toEpochMilli(),
119             TimeSeriesRecord::accumulate, aggTime, frequencyMillis),
120             iterable, measurement);
121   }

123 }
