32  package org.spf4j.io;
33  
34  import edu.umd.cs.findbugs.annotations.CleanupObligation;
35  import edu.umd.cs.findbugs.annotations.DischargesObligation;
36  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
37  import java.io.IOException;
38  import java.io.OutputStream;
39  import java.io.Writer;
40  import java.nio.charset.Charset;
41  import java.nio.charset.CharsetDecoder;
42  import java.nio.charset.CodingErrorAction;
43  
44  /**
45   * {@link OutputStream} implementation that transforms a byte stream to a
46   * character stream using a specified character set encoding and writes the resulting
47   * stream to a {@link Writer}. The stream is transformed using a
48   * {@link CharsetDecoder} object, guaranteeing that all character set
49   * encodings supported by the JRE are handled correctly.
50   * <p>
51   * The output of the {@link CharsetDecoder} is buffered using a fixed size buffer.
52   * This implies that the data is written to the underlying {@link Writer} in chunks
53   * that are no larger than the size of this buffer. By default, the buffer is
54   * flushed only when it overflows or when {@link #flush()} or {@link #close()}
55   * is called. In general there is therefore no need to wrap the underlying {@link Writer}
56   * in a {@link java.io.BufferedWriter}. {@link WriterOutputStream} can also
57   * be instructed to flush the buffer after each write operation. In this case, all
58   * available data is written immediately to the underlying {@link Writer}, implying that
59   * the current position of the {@link Writer} is correlated to the current position
60   * of the {@link WriterOutputStream}.
61   * <p>
62   * {@link WriterOutputStream} implements the inverse transformation of {@link java.io.OutputStreamWriter};
63   * in the following example, writing to {@code out2} would have the same result as writing to
64   * {@code out} directly (provided that the byte sequence is legal with respect to the
65   * character set encoding):
66   * <pre>
67   * OutputStream out = ...
68   * Charset cs = ...
69   * OutputStreamWriter writer = new OutputStreamWriter(out, cs);
70   * WriterOutputStream out2 = new WriterOutputStream(writer, cs);</pre>
71   * {@link WriterOutputStream} implements the same transformation as {@link java.io.InputStreamReader},
72   * except that the control flow is reversed: both classes transform a byte stream
73   * into a character stream, but {@link java.io.InputStreamReader} pulls data from the underlying stream,
74   * while {@link WriterOutputStream} pushes it to the underlying stream.
75   * <p>
76   * Note that while there are use cases where there is no alternative to using
77   * this class, very often the need to use this class is an indication of a flaw
78   * in the design of the code. This class is typically used in situations where an existing
79   * API only accepts an {@link OutputStream} object, but where the stream is known to represent
80   * character data that must be decoded for further use.
81   * <p>
82   * Instances of {@link WriterOutputStream} are not thread safe.
83   *
84   * @see org.apache.commons.io.input.ReaderInputStream
85   *
86   * @since 7.2.25
87   */
88  @CleanupObligation
89  public final class WriterOutputStream extends AppendableOutputStream {
90      private static final int DEFAULT_BUFFER_SIZE = 8192;
91  
92      private final Writer writer;
93  
94  
95      /**
96       * Constructs a new {@link WriterOutputStream} with a default output buffer size of
97       * 1024 characters. The output buffer will only be flushed when it overflows or when
98       * {@link #flush()} or {@link #close()} is called.
99       *
100      * @param writer the target {@link Writer}
101      * @param decoder the charset decoder
102      * @since 2.1
103      */
104     public WriterOutputStream(final Writer writer, final CharsetDecoder decoder) {
105         this(writer, decoder, DEFAULT_BUFFER_SIZE);
106     }
107 
108     /**
109      * Constructs a new {@link WriterOutputStream}.
110      *
111      * @param writer the target {@link Writer}
112      * @param decoder the charset decoder
113      * @param bufferSize the size of the output buffer in number of characters
114      * @since 2.1
115      */
116     @SuppressFBWarnings("EI_EXPOSE_REP2")
117     public WriterOutputStream(final Writer writer, final CharsetDecoder decoder, final int bufferSize) {
118         super(writer, decoder, bufferSize);
119         this.writer = writer;
120     }
121 
122     /**
123      * Constructs a new {@link WriterOutputStream}.
124      *
125      * @param writer the target {@link Writer}
126      * @param charset the charset encoding
127      * @param bufferSize the size of the output buffer in number of characters
128      */
129     public WriterOutputStream(final Writer writer, final Charset charset, final int bufferSize) {
130         this(writer,
131              charset.newDecoder()
132                     .onMalformedInput(CodingErrorAction.REPLACE)
133                     .onUnmappableCharacter(CodingErrorAction.REPLACE)
134                     .replaceWith("?"),
135              bufferSize);
136     }
137 
138     /**
139      * Constructs a new {@link WriterOutputStream} with a default output buffer size of
140      * 1024 characters. The output buffer will only be flushed when it overflows or when
141      * {@link #flush()} or {@link #close()} is called.
142      *
143      * @param writer the target {@link Writer}
144      * @param charset the charset encoding
145      */
146     public WriterOutputStream(final Writer writer, final Charset charset) {
147         this(writer, charset, DEFAULT_BUFFER_SIZE);
148     }
149 
150     /**
151      * Constructs a new {@link WriterOutputStream}.
152      *
153      * @param writer the target {@link Writer}
154      * @param charsetName the name of the charset encoding
155      * @param bufferSize the size of the output buffer in number of characters
156      */
157     public WriterOutputStream(final Writer writer, final String charsetName, final int bufferSize) {
158         this(writer, Charset.forName(charsetName), bufferSize);
159     }
160 
161     /**
162      * Constructs a new {@link WriterOutputStream} with a default output buffer size of
163      * 1024 characters. The output buffer will only be flushed when it overflows or when
164      * {@link #flush()} or {@link #close()} is called.
165      *
166      * @param writer the target {@link Writer}
167      * @param charsetName the name of the charset encoding
168      */
169     public WriterOutputStream(final Writer writer, final String charsetName) {
170         this(writer, charsetName, DEFAULT_BUFFER_SIZE);
171     }
172 
173 
174 
175     /**
176      * Flush the stream. Any remaining content accumulated in the output buffer
177      * will be written to the underlying {@link Writer}. After that
178      * {@link Writer#flush()} will be called.
179      * @throws IOException if an I/O error occurs
180      */
181     @Override
182     public void flush() throws IOException {
183         super.flush();
184         writer.flush();
185     }
186 
187     /**
188      * Close the stream. Any remaining content accumulated in the output buffer
189      * will be written to the underlying {@link Writer}. After that
190      * {@link Writer#close()} will be called.
191      * @throws IOException if an I/O error occurs
192      */
193     @Override
194     @DischargesObligation
195     public void close() throws IOException {
196         super.close();
197         writer.close();
198     }
199 
200 
201     /**
202      * Flush the output.
203      *
204      * @throws IOException if an I/O error occurs
205      */
206     @Override
207     protected void flushOutput() throws IOException {
208         if (decoderOut.position() > 0) {
209             writer.write(decoderOut.array(), 0, decoderOut.position());
210             decoderOut.rewind();
211         }
212     }
213 
214     @Override
215     public String toString() {
216         return "WriterOutputStream{" + "writer=" + writer + '}';
217     }
218 
219 }
