94    private static void diff(final String path, final Schema s1,
95            final Schema s2, final Consumer<SchemaDiff> diffs, final Set<Schema> visited) {
96      if (visited.contains(s1)) {
97        return;
98      }
99      Schema.Type type1 = s1.getType();
100     if (s2.getType() != type1) {
101       diffs.accept(SchemaDiff.of(path, s1, s2, SchemaDiff.Type.DIFFERENT_TYPES));
102       return;
103     } else {
104       switch (type1) {
105         case BOOLEAN:
106         case BYTES:
107         case DOUBLE:
108         case FLOAT:
109         case INT:
110         case LONG:
111         case NULL:
112         case STRING:
113           break;
114         case FIXED:
115           if (s1.getFixedSize() != s2.getFixedSize()) {
116             diffs.accept(SchemaDiff.of(path, s1, s2, SchemaDiff.Type.DIFFERENT_FIXED_SIZE));
117           }
118           if (!s1.getFullName().equals(s2.getFullName())) {
119             diffs.accept(SchemaDiff.of(path, s1, s2, SchemaDiff.Type.DIFFERENT_NAMES));
120           }
121           break;
122         case ARRAY:
123           visited.add(s1);
124           diff(pathAdd(path, "[]"), s1.getElementType(), s2.getElementType(), diffs, visited);
125           break;
126         case MAP:
127           visited.add(s1);
128           diff(pathAdd(path, "{}"), s1.getValueType(), s2.getValueType(), diffs, visited);
129           break;
130         case ENUM:
131           if (!s1.getFullName().equals(s2.getFullName())) {
132             diffs.accept(SchemaDiff.of(path, s1, s2, SchemaDiff.Type.DIFFERENT_NAMES));
133           }
134           List<String> s1s = s1.getEnumStringSymbols();
135           List<String> s2s = s2.getEnumStringSymbols();
136           if (!(s1s.containsAll(s2s)
137                   && s2s.containsAll(s1s))) {
138             diffs.accept(SchemaDiff.of(path, s1, s2, SchemaDiff.Type.DIFFERENT_ENUM_VALUES));
139           }
140           break;
141         case UNION:
142           visited.add(s1);
143           for (Schema s : s1.getTypes()) {
144             Schema os = getFromUnion(s, s2);
145             if (os == null) {
146               diffs.accept(SchemaDiff.of(path, s, null, SchemaDiff.Type.SCHEMA_MISSING_RIGHT));
147             } else {
148               diff(path, s, os, diffs, visited);
149             }
150           }
151           for (Schema s : s2.getTypes()) {
152             Schema os = getFromUnion(s, s2);
153             if (os == null) {
154               diffs.accept(SchemaDiff.of(path, null, s, SchemaDiff.Type.SCHEMA_MISSING_LEFT));
155             }
156           }
157           break;
158         case RECORD:
159           visited.add(s1);
160          if (!s1.getFullName().equals(s2.getFullName())) {
161             diffs.accept(SchemaDiff.of(path, s1, s2, SchemaDiff.Type.DIFFERENT_NAMES));
162           }
163           for (Schema.Field field1 : s1.getFields()) {
164             Field field2 = s2.getField(field1.name());
165             if (field2 == null) {
166               diffs.accept(SchemaDiff.of(path,
167                       field1, null, SchemaDiff.Type.FIELD_MISSING_RIGHT));
168             } else {
169               diff(pathAdd(path, field1.name()), field1.schema(), field2.schema(), diffs, visited);
170               if (!Objects.equals(field1.defaultVal(), field2.defaultVal())) {
171                 diffs.accept(SchemaDiff.of(path,
172                         field1, field2, SchemaDiff.Type.DIFFERENT_FIELD_DEFAULTS));
173               }
174               if (!field1.getObjectProps().equals(field2.getObjectProps())) {
175                 diffs.accept(SchemaDiff.of(path,
176                         field1, field2, SchemaDiff.Type.DIFFERENT_FIELD_PROPERTIES));
177               }
178               if (!Objects.equals(field1.doc(), field2.doc())) {
179                 diffs.accept(SchemaDiff.of(path,
180                         field1, field2, SchemaDiff.Type.DIFFERRENT_FIELD_DOC));
181               }
182             }
183           }
184           for (Schema.Field field2 : s2.getFields()) {
185             Field field1 = s1.getField(field2.name());
186             if (field1 == null) {
187               diffs.accept(SchemaDiff.of(path, null, field2, SchemaDiff.Type.FIELD_MISSING_LEFT));
188             }
189           }
190           break;
191           default:
192             throw new IllegalStateException("Invalid Schema " + s1);
193       }
194     }
195     if (!Objects.equals(s1.getLogicalType(), s2.getLogicalType())) {
196       diffs.accept(SchemaDiff.of(path, s1, s2, SchemaDiff.Type.DIFFERENT_LOGICAL_TYPES));
197     }
198     if (!s1.getObjectProps().equals(s2.getObjectProps())) {
199       diffs.accept(SchemaDiff.of(path, s1, s2, SchemaDiff.Type.DIFFERENT_SCHEMA_PROPERTIES));
200     }
201     if (!Objects.equals(s1.getDoc(), s2.getDoc())) {
202       diffs.accept(SchemaDiff.of(path, s1, s2, SchemaDiff.Type.DIFFERRENT_SCHEMA_DOC));
203     }
204   }

206   private static String pathAdd(final String p, final String ref) {
207     if (p.isEmpty()) {
208       return SCHEMA_PATH_CSV.toCsvElement(ref);
209     } else {
210       return p + '.' + SCHEMA_PATH_CSV.toCsvElement(ref);
211     }

216     String fullName = what.getFullName();
217     for (Schema s : unionSchema.getTypes()) {
218       if (fullName.equals(s.getFullName())) {

222     return null;

244   /**
245    * depth first visit.
246    *
247    * @param start
248    * @param visitor
249    */
250   public static <T> T visit(final Schema start, final SchemaVisitor<T> visitor) {
251     // Set of Visited Schemas
252     IdentityHashSet<Schema> visited = new IdentityHashSet<>();
253     // Stack that contains the Schams to process and afterVisitNonTerminal functions.
254     // Deque<Either<Schema, Supplier<SchemaVisitorAction>>>
255     // Using either has a cost which we want to avoid...
256     Deque<Object> dq = new ArrayDeque<>();
257     dq.addLast(start);
258     Object current;
259     while ((current = dq.pollLast()) != null) {
260       if (current instanceof Supplier) {
261         // we are executing a non terminal post visit.
262         SchemaVisitorAction action = ((Supplier<SchemaVisitorAction>) current).get();
263         switch (action) {
264           case CONTINUE:
265             break;
266           case SKIP_SUBTREE:
267             throw new UnsupportedOperationException();
268           case SKIP_SIBLINGS:
269             while (dq.getLast() instanceof Schema) {
270               dq.removeLast();
271             }
272             break;
273           case TERMINATE:
274             return visitor.get();
275           default:
276             throw new UnsupportedOperationException("Invalid action " + action);
277         }
278       } else {
279         Schema schema = (Schema) current;
280         boolean terminate;
281         if (!visited.contains(schema)) {
282           Schema.Type type = schema.getType();
283           switch (type) {
284             case ARRAY:
285               terminate = visitNonTerminal(visitor, schema, dq, Collections.singletonList(schema.getElementType()));
286               visited.add(schema);
287               break;
288             case RECORD:
289               terminate = visitNonTerminal(visitor, schema, dq,
290                       Lists.transform(Lists.reverse(schema.getFields()), Field::schema));
291               visited.add(schema);
292               break;
293             case UNION:
294               terminate = visitNonTerminal(visitor, schema, dq, schema.getTypes());
295               visited.add(schema);
296               break;
297             case MAP:
298               terminate = visitNonTerminal(visitor, schema, dq, Collections.singletonList(schema.getValueType()));
299               visited.add(schema);
300               break;
301             case NULL:
302             case BOOLEAN:
303             case BYTES:
304             case DOUBLE:
305             case ENUM:
306             case FIXED:
307             case FLOAT:
308             case INT:
309             case LONG:
310             case STRING:
311               terminate = visitTerminal(visitor, schema, dq);
312               break;
313             default:
314               throw new UnsupportedOperationException("Invalid type " + type);
315           }
316         } else {
317           terminate = visitTerminal(visitor, schema, dq);
318         }
319         if (terminate) {
320           return visitor.get();
321         }
322       }
323     }
324     return visitor.get();
325   }

327   private static boolean visitNonTerminal(final SchemaVisitor visitor,
328           final Schema schema, final Deque<Object> dq,
329           final Iterable<Schema> itSupp) {
330     SchemaVisitorAction action = visitor.visitNonTerminal(schema);
331     switch (action) {
332       case CONTINUE:
333         dq.addLast((Supplier<SchemaVisitorAction>) () -> visitor.afterVisitNonTerminal(schema));
334         Iterator<Schema> it = itSupp.iterator();
335         while (it.hasNext()) {
336           Schema child = it.next();
337           dq.addLast(child);
338         }
339         break;
340       case SKIP_SUBTREE:
341         dq.addLast((Supplier<SchemaVisitorAction>) () -> visitor.afterVisitNonTerminal(schema));
342         break;
343       case SKIP_SIBLINGS:
344         while (!dq.isEmpty() && dq.getLast() instanceof Schema) {
345           dq.removeLast();
346         }
347         break;
348       case TERMINATE:
349         return true;
350       default:
351         throw new UnsupportedOperationException("Invalid action " + action + " for " + schema);
352     }
353     return false;
354   }
355 
356   private static boolean visitTerminal(final SchemaVisitor visitor, final Schema schema,
357           final Deque<Object> dq) {
358     SchemaVisitorAction action = visitor.visitTerminal(schema);
359     switch (action) {
360       case CONTINUE:
361         break;
362       case SKIP_SUBTREE:
363         throw new UnsupportedOperationException("Invalid action " + action + " for " + schema);
364       case SKIP_SIBLINGS:
365         Object current;
366         //CHECKSTYLE:OFF InnerAssignment
367         while ((current = dq.getLast()) instanceof Schema) {
368           // just skip
369         }
370         //CHECKSTYLE:ON
371         dq.addLast(current);
372         break;
373       case TERMINATE:
374         return true;
375       default:
376         throw new UnsupportedOperationException("Invalid action " + action + " for " + schema);
377     }
378     return false;
379   }

381   public static boolean isNullableUnion(final Schema schema) {
382     if (schema.getType() != Schema.Type.UNION) {
383       return false;
384     }
385     for (Schema ss : schema.getTypes()) {
386       if (ss.getType() == Schema.Type.NULL) {
387         return true;
388       }
389     }
390     return false;
391   }

394   public static Schema nullableUnionSchema(final Schema schema) {
395     if (!(schema.getType() == Schema.Type.UNION)) {
396       return null;
397     }
398     List<Schema> types = schema.getTypes();
399     if (types.size() != 2) {
400       return null;
401     }
402     if (types.get(0).getType() == Schema.Type.NULL) {
403       return types.get(1);
404     } else if (types.get(1).getType() == Schema.Type.NULL) {
405       return types.get(0);
406     } else {
407       return null;
408     }
409   }

425   @Nullable
426   public static Schema getSubSchema(final Schema schema, final CharSequence path) {
427     if (path.length() == 0) {
428       return schema;
429     }
430     List<String> parsedPath;
431     try {
432       parsedPath = SCHEMA_PATH_CSV.readRow(CharSequences.reader(path));
433     } catch (IOException ex) {
434       throw new RuntimeException(ex);
435     } catch (CsvParseException ex) {
436       throw new IllegalArgumentException("Invalid path " + path, ex);
437     }
438     Schema result = schema;
439     for (String part : parsedPath) {
440       result = getSegment(result, part);
441       if (result == null) {
442         return null;
443       }
444     }
445     return result;
446   }

448   @Nullable
449   private static Schema getSegment(final Schema result, final String part) {
450     switch (result.getType()) {
451         case ARRAY:
452           if ("[]".equals(part)) {
453             return result.getElementType();
454           }
455           break;
456         case MAP:
457           if ("{}".equals(part)) {
458             return result.getValueType();
459           }
460           break;
461         case UNION:
462           for (Schema us : result.getTypes()) {
463             Schema ur = getSegment(us, part);
464             if (ur != null) {
465               return ur;
466             }
467           }
468           break;
469         case RECORD:
470           Field field = result.getField(part);
471           if (field == null) {
472             for (Schema.Field f : result.getFields()) {
473               if (f.aliases().contains(part)) {
474                 field = f;
475               }
476             }
477           }
478           if (field != null) {
479             return field.schema();
480           }
481           break;
482         default:
483           return null;
484       }
485       return null;
486   }

488   /**
489    * Obtain a sub-schema of a record schema.
490    * the sub-schema will contain only the fields referenced by the definition order.
491    * @param schema the source schema.
492    * @param projection the order ids of the fields in the sub-schema.
493    * @return the sub schema. Will have the same name as the original schema
494    * with "_p." prepended to the name space
495    */
496   public static Schema projectRecord(final Schema schema, final int[] projection) {
497     List<Field> fields = schema.getFields();
498     List<Field> nFields = new ArrayList<>(projection.length);
499     for (int i = 0; i < projection.length; i++) {
500       Field of = fields.get(projection[i]);
501       Field nfield = new Schema.Field(of, of.schema());
502       nFields.add(nfield);
503     }
504     if (isSameFields(fields, nFields)) {
505       return schema;
506     }
507     Schema rec = Schema.createRecord(schema.getName(), schema.getDoc(),
508             "_p." + schema.getNamespace(), schema.isError());
509     rec.setFields(nFields);
510     String dep = schema.getProp("deprecated");
511     if (dep != null) {
512       rec.addProp("deprecated", dep);
513     }
514     return rec;
515   }

523   /**
524    * create a schema projection(partial schema) from a destination schema.
525    *
526    * @param schema the schema to project.
527    * @param paths the paths to elements (paths to record fields) included in the subschema.
528    * @return the subschema.
529    */
530   @Nullable
531   public static Schema project(final Schema schema, final List<? extends CharSequence> paths) {
532     final List<List<String>> p = new ArrayList<>(paths.size());
533     for (CharSequence cs : paths) {
534       try {
535         p.add(SCHEMA_PATH_CSV.readRow(CharSequences.reader(cs)));
536       } catch (IOException ex) {
537         throw new UncheckedIOException(ex);
538       } catch (CsvParseException ex) {
539         throw new IllegalArgumentException("Invalid projection path " + cs, ex);
540       }
541     }
542     return projectInternal(schema, p);
543   }

545   @Nullable
546   private static Schema projectInternal(final Schema schema, final List<List<String>> paths) {
547     int length = paths.size();
548     if (length == 0) {
549       return schema;
550     }
551     if (length == 1) {
552        List<String> first = paths.get(0);
553        if (first.isEmpty() || (first.size() == 1 && first.get(0).isEmpty())) {
554          return schema;
555        }
556     }
557     List<List<String>> seqs;
558     switch (schema.getType()) {
559       case ARRAY:
560         seqs = new ArrayList<>(length);
561         for (List<String> path : paths) {
562           String part = path.get(0);
563           if ("[]".equals(part)) {
564             int pSize = path.size();
565             if (pSize == 1) {
566               return schema;
567             }
568             seqs.add(path.subList(1, pSize));
569           } else {
570             return null;
571           }
572         }
573         if (seqs.isEmpty()) {
574           return null;
575         }
576         return Schema.createArray(projectInternal(schema.getElementType(), seqs));
577       case MAP:
578         seqs = new ArrayList<>(length);
579         for (List<String>  path : paths) {
580           String part = path.get(0);
581           if ("{}".equals(part)) {
582             int pSize = path.size();
583             if (1 == pSize) {
584               return schema;
585             }
586             seqs.add(path.subList(1, pSize));
587           }
588         }
589         if (seqs.isEmpty()) {
590           return null;
591         }
592         return Schema.createMap(projectInternal(schema.getElementType(), seqs));
593       case RECORD:
594         List<Field> fields = schema.getFields();
595         List<Schema.Field> nFields = new ArrayList<>(fields.size());
596         List<List<String>> tPaths = new LinkedList<>(paths);
597         do {
598           Field extract = extract(fields, tPaths);
599           if (extract == null) {
600             return null;
601           }
602           nFields.add(extract);
603         } while (!tPaths.isEmpty());
604         if (isSameFields(fields, nFields)) {
605           return schema;
606         }
607         Schema rec = AvroCompatUtils.createRecordSchema(schema.getName(), schema.getDoc(),
608                 "_p." + schema.getNamespace(), schema.isError(), false);
609         rec.setFields(nFields);
610         String dep = schema.getProp("deprecated");
611         if (dep != null) {
612           rec.addProp("deprecated", dep);
613         }
614         return rec;
615       case UNION:
616         List<Schema> types = schema.getTypes();
617         List<Schema> nTypes = new ArrayList<>(types.size());
618         for (Schema us : types) {
619           if (us.getType() == Schema.Type.NULL) {
620             nTypes.add(us);
621           } else {
622             Schema project = projectInternal(us, paths);
623             if (project != null) {
624               nTypes.add(project);
625             }
626           }
627         }
628         return Schema.createUnion(nTypes);
629       default:
630         if (paths.contains(Collections.emptyList())) {
631           return schema;
632         } else {
633           return null;
634         }
635     }
636   }

638   public static boolean isSameFields(final List<Schema.Field> f1s, final List<Schema.Field> f2s) {
639     int l1 = f1s.size();
640     int l2 = f2s.size();
641     if (l1 != l2) {
642       return false;
643     }
644     for (int i = 0; i < l1; i++) {
645       Schema.Field f1 = f1s.get(i);
646       Schema.Field f2 = f2s.get(i);
647       if (!f1.name().equals(f2.name())
648               || !f1.schema().equals(f2.schema())) {
649         return false;
650       }
651     }
652     return true;
653   }

669   /**
670    * Project a avro object to a destination.
671    * @param toSchema the object schema
672    * @param fromSchema the destination schema
673    * @param object the object to project.
674    * @return the projected object.
675    */
676   @Nullable
677   @SuppressFBWarnings("URV_UNRELATED_RETURN_VALUES")
678   public static Object project(final Schema toSchema, final Schema fromSchema, @Nullable final Object object) {
679     if (toSchema == fromSchema) {
680       return object;
681     }
682     Schema.Type type = toSchema.getType();
683     if (fromSchema.getType() != type) {
684       throw new IllegalArgumentException("Unable to project " + object + " from "  + fromSchema +  " to " + toSchema);
685     }
686     switch (type) {
687       case INT:
688       case LONG:
689       case FLOAT:
690       case STRING:
691       case ENUM:
692       case FIXED:
693       case BOOLEAN:
694       case BYTES:
695       case DOUBLE:
696         return object;
697       case NULL:
698         return null;
699       case ARRAY:
700         List from = (List) object;
701         List to = new ArrayList(from.size());
702         Schema toElementType = toSchema.getElementType();
703         Schema fromElementType = fromSchema.getElementType();
704         for (Object o : from) {
705           to.add(project(toElementType, fromElementType, o));
706         }
707         return to;
708       case MAP:
709         Map<String, Object> fromMap = (Map<String, Object>) object;
710         Map<String, Object> toMap = Maps.newLinkedHashMapWithExpectedSize(fromMap.size());
711         Schema toValueType = toSchema.getValueType();
712         Schema fromValueType = fromSchema.getValueType();
713         for (Map.Entry<String, Object> entry : fromMap.entrySet()) {
714           toMap.put(entry.getKey(), project(toValueType, fromValueType, entry.getValue()));
715         }
716         return toMap;
717       case UNION:
718         ExtendedReflectData rftor = ExtendedReflectData.get();
719         int unionIdx = rftor.resolveUnion(fromSchema, object);
720         Schema objSchema = fromSchema.getTypes().get(unionIdx);
721         String name = rftor.getSchemaName(objSchema);
722         Set<String> aliases = objSchema.getAliases();
723         if (aliases.isEmpty()) {
724           for (Schema toMatch : toSchema.getTypes()) {
725             if (name.equals(rftor.getSchemaName(toMatch)) || toMatch.getAliases().contains(name)) {
726               return project(toMatch, objSchema, object);
727             }
728           }
729         } else {
730           for (Schema toMatch : toSchema.getTypes()) {
731             String toMatchName = rftor.getSchemaName(toMatch);
732             if (name.equals(toMatchName) || aliases.contains(toMatchName)
733                     || !Sets.intersection(aliases, toMatch.getAliases()).isEmpty()) {
734               return project(toMatch, objSchema, object);
735             }
736           }
737         }
738         throw new IllegalArgumentException("Unable to project " + object + " to " + toSchema);
739 
740       case RECORD:
741         GenericData.Record record = new SpecificData.Record(toSchema);
742         GenericRecord fromRec = (GenericRecord) object;
743         Schema frSchema = fromRec.getSchema();
744         for (Field field : toSchema.getFields()) {
745           Field frField = getPos(field, frSchema);
746           int toPos = field.pos();
747           if (frField == null) {
748             record.put(toPos, field.defaultVal());
749           } else {
750             record.put(toPos, project(field.schema(), frField.schema(), fromRec.get(frField.pos())));
751           }
752         }
753         return record;
754       default:
755         throw new IllegalStateException("Unsupported type " + type);
756     }
757   }

760   @Nullable
761   public static Schema.Field getPos(final Schema.Field field, final Schema recordSchema) {
762     Field rf = recordSchema.getField(field.name());
763     if (rf == null) {
764       for (String alias : field.aliases()) {
765         rf = recordSchema.getField(alias);
766         if (rf != null) {
767           return rf;
768         }
769       }
770       return null;
771     } else {
772       return rf;
773     }
774   }

777   @Nullable
778   private static Schema.Field extract(final Iterable<Field> fields, final Iterable<List<String>> paths) {
779     Iterator<List<String>> iterator = paths.iterator();
780     if (iterator.hasNext()) {
781       List<List<String>> proj = new ArrayList<>(2);
782       List<String> path = iterator.next();
783       String part = path.get(0);
784       Field field = getField(part, fields);
785       if (field == null) {
786         return null;
787       }
788       iterator.remove();
789       if (1 == path.size()) {
790         proj.add(Collections.EMPTY_LIST);
791       } else {
792         proj.add(path.subList(1, path.size()));
793       }
794       while (iterator.hasNext()) {
795         path = iterator.next();
796         part = path.get(0);
797         if (isNamed(part, field)) {
798           if (1 == path.size()) {
799             proj.add(Collections.EMPTY_LIST);
800           } else {
801             proj.add(path.subList(1,  path.size()));
802           }
803           iterator.remove();
804         }
805       }
806       return  new Schema.Field(field, projectInternal(field.schema(), proj));
807     } else {
808       return null;
809     }
810   }

815     for (Schema.Field field : fields) {
816       if (isNamed(nameOrAlias, field)) {
817         return field;
818       }
819     }
820     return null;

845   private static class DeprecationVisitor implements SchemaVisitor<Void> {
846 
847     private final BiConsumer<String, String> toPut;
848 
849     DeprecationVisitor(final BiConsumer<String, String> toPut) {
850       this.toPut = toPut;
851     }
852 
853     @Override
854     public SchemaVisitorAction visitTerminal(final Schema terminal) {
855       if (terminal.getType() == Schema.Type.FIXED) {
856         checkDeprecation(terminal);
857       }
858       // do nothing for base types (will ignore if somebody deprercates an int....)
859       return SchemaVisitorAction.CONTINUE;
860     }
861 
862     public void checkDeprecation(final Schema terminal) {
863       String dMsg = terminal.getProp("deprecated");
864       if (dMsg != null) {
865         toPut.accept(terminal.getFullName(), dMsg);
866       }
867     }
868 
869     @Override
870     public SchemaVisitorAction visitNonTerminal(final Schema nonTerminal) {
871       checkDeprecation(nonTerminal);
872       if  (nonTerminal.getType() == Schema.Type.RECORD) {
873         for (Schema.Field field : nonTerminal.getFields()) {
874           String msg = field.getProp("deprecated");
875           if (msg != null) {
876             toPut.accept(nonTerminal.getFullName() + '.' + field.name(), msg);
877           }
878         }
879       }
880       return SchemaVisitorAction.CONTINUE;
881     }
882 
883     @Override
884     public SchemaVisitorAction afterVisitNonTerminal(final Schema nonTerminal) {
885       return SchemaVisitorAction.CONTINUE;
886     }
887   }
