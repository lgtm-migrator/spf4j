85            = (a, b) -> {

117   public static void copyAliases(final Schema from, final Schema to) {
118     //CHECKSTYLE:OFF
119     switch (from.getType()) { // only named types.
120       case RECORD:
121       case ENUM:
122       case FIXED:
123         Set<String> aliases = from.getAliases();
124         for (String alias : aliases) {
125           to.addAlias(alias);
126         }
127         break;
128       default:
129       //ignore unnamed one's
130     }
131     //CHECKSTYLE:OFF
132   }
133 
134   public static void copyAliases(final Schema.Field from, final Schema.Field to) {
135     Set<String> aliases = from.aliases();
136     for (String alias : aliases) {
137       to.addAlias(alias);
138     }
139   }

141   public static void copyLogicalTypes(final Schema from, final Schema to) {
142     LogicalType logicalType = from.getLogicalType();
143     if (logicalType != null) {
144       logicalType.addToSchema(to);
145     }
146   }

148   public static void copyProperties(final JsonProperties from, final JsonProperties to) {
149     Map<String, Object> objectProps = from.getObjectProps();
150     for (Map.Entry<String, Object> entry : objectProps.entrySet()) {
151       to.addProp(entry.getKey(), entry.getValue());
152     }
153   }

155   public static boolean isNamedType(final Schema schema) {
156     Schema.Type type = schema.getType();
157     switch (type) {
158       case ENUM:
159       case RECORD:
160       case FIXED:
161         return true;
162       default:
163         return false;
164     }

167   public static String getJavaClassName(final Schema schema) {
168     String namespace = schema.getNamespace();
169     if (namespace == null || namespace.isEmpty()) {
170       return SpecificCompiler.mangle(schema.getName());
171     } else {
172       return namespace + '.' + SpecificCompiler.mangle(schema.getName());
173     }
174   }

176   @Beta
177   public static void writeIdlProtocol(final String protocolName, final String protocolNameSpace,
178           final Appendable appendable, final Schema... schemas) throws IOException {
179     if (protocolNameSpace != null) {
180       appendable.append("@namespace(\"").append(protocolNameSpace).append("\")\n");
181     }
182     appendable.append("protocol ").append(protocolName).append(" {\n\n");
183     if (isIdlCycleSupport()) {
184       writeIdl(appendable, new HashSet<>(4), protocolNameSpace, schemas);
185     } else {
186       writeIdlLegacy(appendable, new HashSet<>(4), protocolNameSpace, schemas);
187     }
188     appendable.append("}\n");
189   }

191   @Beta
192   public static void writeIdl(final Appendable appendable,
193           final Set<String> alreadyDeclared, final String protocolNameSpace, final Schema ... pschemas)
194           throws IOException {
195     try(JsonGenerator jsonGen = createJsonGenerator(appendable)) {
196       final Set<Schema> toDeclare = new HashSet<>(4);
197       toDeclare.addAll(Arrays.asList(pschemas));
198       while (!toDeclare.isEmpty()) {
199         Iterator<Schema> iterator = toDeclare.iterator();
200         Schema schema = iterator.next();
201         iterator.remove();
202         writeSchema(schema, appendable, jsonGen, protocolNameSpace, alreadyDeclared, toDeclare);
203         appendable.append('\n');
204       }
205     }
206   }

208   public static JsonGenerator createJsonGenerator(final Appendable appendable) throws IOException {
209     JsonGenerator jsonGen;
210     if (appendable instanceof Writer) {
211       jsonGen = Json.FACTORY.createGenerator((Writer) appendable);
212     } else {
213       jsonGen = Json.FACTORY.createGenerator(new AppendableWriter(appendable));
214     }
215     return jsonGen;
216   }

218   @Beta
219   @SuppressFBWarnings("RV_RETURN_VALUE_IGNORED")
220   public static void writeIdlLegacy(final Appendable appendable,
221           final Set<String> alreadyDeclared, final String protocolNameSpace, final Schema ... pschemas)
222           throws IOException {
223     MutableGraph<Schema> schemaDeps = GraphBuilder.directed().allowsSelfLoops(false)
224             .expectedNodeCount(4).build();
225     Map<Schema, String> idlRepresentation = new HashMap<>(4);
226     final Set<Schema> toDeclare = new HashSet<>(4);
227     toDeclare.addAll(Arrays.asList(pschemas));
228     while (!toDeclare.isEmpty()) {
229       Iterator<Schema> iterator = toDeclare.iterator();
230       Schema schema = iterator.next();
231       iterator.remove();
232       StringWriter schemaIdrStr = new StringWriter();
233       JsonGenerator jsonGen = Schema.FACTORY.createGenerator(schemaIdrStr);
234       Set<Schema> orig = new HashSet<>(toDeclare);
235       writeSchema(schema, schemaIdrStr, jsonGen, protocolNameSpace, alreadyDeclared, toDeclare);
236       idlRepresentation.put(schema, schemaIdrStr.toString());
237       schemaDeps.addNode(schema);
238       Set<Schema> dependencies = Sets.difference(toDeclare, orig);
239       for (Schema dep : dependencies) {
240         schemaDeps.putEdge(schema, dep);
241       }
242      }
243      // Now process the nodes from leaves onward.
244 
245     MutableGraph<Schema> traverseGraph = Graphs.clone(schemaDeps);
246     Set<Schema> nodes = traverseGraph.nodes();
247     List<Schema> nodesToRemove = new ArrayList<>();
248     do {
249       for (Schema token : nodes) {
250         if (traverseGraph.outDegree(token) == 0) {
251           nodesToRemove.add(token);
252           appendable.append(idlRepresentation.get(token));
253           appendable.append('\n');
254         }
255       }
256       if (nodesToRemove.isEmpty() && !nodes.isEmpty()) {
257         throw new IllegalArgumentException("Schema definition cycle for" + nodes);
258       }
259       for (Schema token : nodesToRemove) {
260         traverseGraph.removeNode(token);
261       }
262       nodesToRemove.clear();
263       nodes = traverseGraph.nodes();
264     } while (!nodes.isEmpty());
265 
266 
267   }

270   private static void writeSchema(Schema schema, final Appendable appendable, JsonGenerator jsonGen,
271           final String protocolNameSpace, final Set<String> alreadyDeclared, final Set<Schema> toDeclare)
272           throws IOException {
273     Schema.Type type = schema.getType();
274     writeSchemaAttributes(schema, appendable, jsonGen, true);
275     String namespace = schema.getNamespace();
276     if (!Objects.equals(namespace, protocolNameSpace)) {
277       appendable.append("@namespace(\"").append(namespace).append("\")\n");
278     }
279     Set<String> saliases = schema.getAliases();
280     if (!saliases.isEmpty()) {
281       appendable.append("@aliases(");
282       toJson(saliases, jsonGen);
283       jsonGen.flush();
284       appendable.append(")\n");
285     }
286     switch (type) {
287       case RECORD:
288         appendable.append("record ").append(schema.getName()).append(" {\n\n");
289         alreadyDeclared.add(schema.getFullName());
290         for (Field field : schema.getFields()) {
291           String fDoc = field.doc();
292           if (fDoc != null) {
293             appendable.append("  /** ").append(fDoc).append(" */\n");
294           }
295           appendable.append("  ");
296           writeFieldSchema(field.schema(), appendable, jsonGen, alreadyDeclared, toDeclare, schema.getNamespace());
297           appendable.append(' ');
298           Set<String> faliases = field.aliases();
299           if (!faliases.isEmpty()) {
300             appendable.append("@aliases(");
301             toJson(faliases, jsonGen);
302             jsonGen.flush();
303             appendable.append(") ");
304           }
305           Field.Order order = field.order();
306           if (order != null) {
307             appendable.append(" @order(\"").append(order.name()).append("\") ");
308           }
309           writeJsonProperties(field, appendable, jsonGen, false);
310           appendable.append(' ');
311           appendable.append(field.name());
312           JsonNode defaultValue = field.defaultValue();
313           if (defaultValue != null) {
314             appendable.append(" = ");
315             toJson(field.defaultVal(), jsonGen);
316             jsonGen.flush();
317           }
318           appendable.append(";\n\n");
319         }
320         appendable.append("}\n");
321         break;
322       case ARRAY:
323       case MAP:
324       case UNION:
325       case BOOLEAN:
326       case BYTES:
327       case DOUBLE:
328       case FLOAT:
329       case INT:
330       case LONG:
331       case NULL:
332       case STRING:
333         throw new UnsupportedOperationException("No IDL entity, nust be part of a record: " + schema);
334       case ENUM:
335         appendable.append("enum ").append(schema.getName()).append(" {");
336         alreadyDeclared.add(schema.getFullName());
337         Iterator<String> i = schema.getEnumSymbols().iterator();
338         if (i.hasNext()) {
339           appendable.append(i.next());
340           while (i.hasNext()) {
341             appendable.append(',');
342             appendable.append(i.next());
343           }
344         } else {
345           throw new IllegalStateException("Enum schema must have at least a symbol " + schema);
346         }
347         appendable.append("}\n");
348         break;
349       case FIXED:
350         appendable.append("fixed ").append(schema.getName()).append('(')
351                 .append(Integer.toString(schema.getFixedSize())).append(");\n");
352         alreadyDeclared.add(schema.getFullName());
353         break;
354       default:
355         throw new IllegalStateException("Unsupported schema " + schema);
356     }
357   }

359   private static void writeFieldSchema(final Schema schema,
360           final Appendable appendable, final JsonGenerator jsonGen,
361           final Set<String> alreadyDeclared, final Set<Schema> toDeclare,
362           final String recordNameSpace) throws IOException {
363     Schema.Type type = schema.getType();
364     switch (type) {
365       case RECORD:
366       case ENUM:
367       case FIXED:
368         if (Objects.equals(recordNameSpace, schema.getNamespace())) {
369           appendable.append(schema.getName());
370         } else {
371           appendable.append(schema.getFullName());
372         }
373         if (!alreadyDeclared.contains(schema.getFullName())) {
374           toDeclare.add(schema);
375         }
376         break;
377       case ARRAY:
378         writeSchemaAttributes(schema, appendable, jsonGen, false);
379         appendable.append("array<");
380         writeFieldSchema(schema.getElementType(), appendable, jsonGen, alreadyDeclared, toDeclare, recordNameSpace);
381         appendable.append('>');
382         break;
383       case MAP:
384         writeSchemaAttributes(schema, appendable, jsonGen, false);
385         appendable.append("map<");
386         writeFieldSchema(schema.getValueType(), appendable, jsonGen, alreadyDeclared, toDeclare, recordNameSpace);
387         appendable.append('>');
388         break;
389       case UNION:
390         writeSchemaAttributes(schema, appendable, jsonGen, false);
391         appendable.append("union {");
392         List<Schema> types = schema.getTypes();
393         Iterator<Schema> iterator = types.iterator();
394         if (iterator.hasNext()) {
395           writeFieldSchema(iterator.next(), appendable, jsonGen, alreadyDeclared, toDeclare, recordNameSpace);
396           while (iterator.hasNext()) {
397             appendable.append(',');
398             writeFieldSchema(iterator.next(), appendable, jsonGen, alreadyDeclared, toDeclare, recordNameSpace);
399           }
400         } else {
401           throw new IllegalStateException("Union schmemas must have member types " + schema);
402         }
403         appendable.append('}');
404         break;
405       case BOOLEAN:
406       case BYTES:
407       case DOUBLE:
408       case FLOAT:
409       case INT:
410       case LONG:
411       case NULL:
412       case STRING:
413         writeSchemaAttributes(schema, appendable, jsonGen, false);
414         appendable.append(schema.getName());
415         break;
416       default:
417         throw new IllegalStateException("Unsupported schema " + schema);
418     }
419   }

421   public static void writeSchemaAttributes(final Schema schema,
422           final Appendable appendable, final JsonGenerator jsonGen, final boolean crBetween)
423           throws IOException {
424     String doc = schema.getDoc();
425     if (doc != null) {
426       appendable.append("/** ").append(doc).append(" */");
427       if (crBetween) {
428         appendable.append('\n');
429       } else {
430         appendable.append(' ');
431       }
432     }
433     writeJsonProperties(schema, appendable, jsonGen, crBetween);
434   }

436   public static void writeJsonProperties(final JsonProperties props,
437           final Appendable appendable, final JsonGenerator jsonGen, final boolean crBetween) throws IOException {
438     Map<String, Object> objectProps = props.getObjectProps();
439     for (Map.Entry<String, Object> entry : objectProps.entrySet()) {
440       appendable.append('@').append(entry.getKey()).append('(');
441       toJson(entry.getValue(), jsonGen);
442       jsonGen.flush();
443       appendable.append(')');
444       if (crBetween) {
445         appendable.append('\n');
446       } else {
447         appendable.append(' ');
448       }
449     }
450   }

452   @SuppressWarnings(value="unchecked")
453   @SuppressFBWarnings("ITC_INHERITANCE_TYPE_CHECKING")
454   static void toJson(final Object datum, final JsonGenerator generator) throws IOException {
455     if (datum == JsonProperties.NULL_VALUE) { // null
456       generator.writeNull();
457     } else if (datum instanceof Map) { // record, map
458       generator.writeStartObject();
459       for (Map.Entry<Object,Object> entry : ((Map<Object,Object>) datum).entrySet()) {
460         generator.writeFieldName(entry.getKey().toString());
461         toJson(entry.getValue(), generator);
462       }
463       generator.writeEndObject();
464     } else if (datum instanceof Collection) { // array
465       generator.writeStartArray();
466       for (Object element : (Collection<?>) datum) {
467         toJson(element, generator);
468       }
469       generator.writeEndArray();
470     } else if (datum instanceof byte[]) { // bytes, fixed
471       generator.writeString(new String((byte[]) datum, StandardCharsets.ISO_8859_1));
472     } else if (datum instanceof CharSequence || datum instanceof Enum<?>) { // string, enum
473       generator.writeString(datum.toString());
474     } else if (datum instanceof Double) { // double
475       generator.writeNumber((Double) datum);
476     } else if (datum instanceof Float) { // float
477       generator.writeNumber((Float) datum);
478     } else if (datum instanceof Long) { // long
479       generator.writeNumber((Long) datum);
480     } else if (datum instanceof Integer) { // int
481       generator.writeNumber((Integer) datum);
482     } else if (datum instanceof Boolean) { // boolean
483       generator.writeBoolean((Boolean) datum);
484     } else {
485       throw new AvroRuntimeException("Unknown datum class: " + datum.getClass());
486     }
487   }
