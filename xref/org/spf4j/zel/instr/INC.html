32  package org.spf4j.zel.instr;
33  
34  import java.util.concurrent.ExecutionException;
35  import org.spf4j.base.Arrays;
36  import org.spf4j.zel.operators.Operator;
37  import org.spf4j.zel.operators.Operators;
38  import org.spf4j.zel.vm.AssignableValue;
39  import org.spf4j.zel.vm.ExecutionContext;
40  import org.spf4j.zel.vm.SuspendedException;
41  
42  public final class INC extends Instruction {
43  
44    private static final long serialVersionUID = 1L;
45  
46    public static final Instruction INSTANCE = new INC();
47  
48    private INC() {
49    }
50  
51    @Override
52    @edu.umd.cs.findbugs.annotations.SuppressWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
53    public int execute(final ExecutionContext context)
54            throws SuspendedException, ExecutionException, InterruptedException {
55      Object val = context.popSyncStackVal();
56      if (val instanceof AssignableValue) {
57        AssignableValue aval = (AssignableValue) val;
58        Number nr = (Number) aval.get();
59        Number result = (Number) Operators.apply(Operator.Enum.Add, nr, 1);
60        aval.assign(result);
61      } else {
62        context.push(Operators.apply(Operator.Enum.Add, val, 1));
63      }
64      return 1;
65    }
66  
67    @Override
68    public Object[] getParameters() {
69      return Arrays.EMPTY_OBJ_ARRAY;
70    }
71  
72  }
