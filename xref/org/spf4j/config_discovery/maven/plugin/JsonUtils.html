32  package org.spf4j.config_discovery.maven.plugin;
33  
34  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
35  import javax.annotation.Nonnull;
36  
37  /**
38   *
39   * @author zoly
40   */
41  public final class JsonUtils {
42  
43    private static final ThreadLocal<char[]> BUFF = new ThreadLocal<char[]>() {
44  
45      @Override
46      @SuppressFBWarnings("SUA_SUSPICIOUS_UNINITIALIZED_ARRAY")
47      protected char[] initialValue() {
48        return new char[64];
49      }
50  
51    };
52  
53    private static final char[] DIGITS = {
54      '0', '1', '2', '3', '4', '5',
55      '6', '7', '8', '9', 'a', 'b',
56      'c', 'd', 'e', 'f', 'g', 'h',
57      'i', 'j', 'k', 'l', 'm', 'n',
58      'o', 'p', 'q', 'r', 's', 't',
59      'u', 'v', 'w', 'x', 'y', 'z'
60    };
61  
62  
63    private JsonUtils() {
64    }
65  
66    public static CharSequence toJsonString(final String str) {
67      StringBuilder sb = new StringBuilder(str.length() + 4);
68      sb.append('"');
69      escapeJsonString(str, sb);
70      sb.append('"');
71      return sb;
72    }
73  
74    public static void escapeJsonString(@Nonnull final String toEscape, final StringBuilder jsonString) {
75      int len = toEscape.length();
76      for (int i = 0; i < len; i++) {
77        char c = toEscape.charAt(i);
78        appendJsonStringEscapedChar(c, jsonString);
79      }
80    }
81  
82    public static void appendJsonStringEscapedChar(final char c, final StringBuilder jsonString) {
83      switch (c) {
84        case '\\':
85        case '"':
86          jsonString.append('\\');
87          jsonString.append(c);
88          break;
89        case '\b':
90          jsonString.append("\\b");
91          break;
92        case '\t':
93          jsonString.append("\\t");
94          break;
95        case '\n':
96          jsonString.append("\\n");
97          break;
98        case '\f':
99          jsonString.append("\\f");
100         break;
101       case '\r':
102         jsonString.append("\\r");
103         break;
104       default:
105         if (c < ' ') {
106           jsonString.append("\\u");
107           appendUnsignedStringPadded(jsonString, (int) c, 4, 4);
108         } else {
109           jsonString.append(c);
110         }
111     }
112   }
113 
114   public static void appendUnsignedStringPadded(final StringBuilder sb, final int nr, final int shift,
115           final int padTo) {
116     long i = nr;
117     char[] buf = BUFF.get();
118     int charPos = 32;
119     int radix = 1 << shift;
120     long mask = radix - 1;
121     do {
122       buf[--charPos] = DIGITS[(int) (i & mask)];
123       i >>>= shift;
124     } while (i != 0);
125     final int nrChars = 32 - charPos;
126     if (nrChars > padTo) {
127       throw new IllegalArgumentException("Your pad to value " + padTo
128               + " is to small, must be at least " + nrChars);
129     } else {
130       for (int j = 0, n = padTo - nrChars; j < n; j++) {
131         sb.append('0');
132       }
133     }
134     sb.append(buf, charPos, nrChars);
135   }
