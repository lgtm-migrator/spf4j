89    @SuppressForbiden
90    StackDumpJInternalFrame(final StackSampleSupplier samplesSupplier,
91            final String title, final boolean isgraph) throws IOException {
92      super(title);
93      this.samplesSupplier = samplesSupplier;
94      setName(title);
95      Instant min = samplesSupplier.getMin();
96      Instant max = samplesSupplier.getMax();
97      long startMillis = min.toEpochMilli();
98      Date minDate = new Date(startMillis - startMillis % 1000);
99      long endMillis = max.toEpochMilli();
100     Date maxDate = new Date(endMillis - endMillis % 1000 + 1000 );
101     startModel = new SpinnerDateModel(minDate, minDate, maxDate, Calendar.SECOND);
102     endModel = new SpinnerDateModel(maxDate, minDate, maxDate, Calendar.SECOND);
103     initComponents();
104     ProfileMetaData metaData = samplesSupplier.getMetaData(min, max);
105     DefaultComboBoxModel<String> tagsModel = new DefaultComboBoxModel<>(metaData.getTags().toArray(new String[] {}));
106     this.tagsSelector.setModel(tagsModel);
107     DefaultComboBoxModel<String> contextModel = new DefaultComboBoxModel<>(
108             metaData.getContexts().toArray(new String[] {}));
109     this.contextSelector.setModel(contextModel);
110     this.samples = samplesSupplier.getSamples((String) contextSelector.getSelectedItem(),
111               (String) tagsSelector.getSelectedItem(), min, max);
112     if (samples == null) {
113       this.samples = SampleNode.createSampleNode(
114               new StackTraceElement[]{new StackTraceElement("NO SAMPLES", "", "", -1)});
115     }
116     setViewType(isgraph);
117     ssScrollPanel.setVisible(true);
118     pack();
119   }

122   private static void sync(final Collection<String> newValues, final JComboBox<String> combo) {
123     Set<String> newvals = new HashSet<>(newValues);
124     int l = combo.getItemCount();
125     List<Integer> toRemove = new ArrayList<>(4);
126     for (int i = 0; i < l; i++) {
127       String item = combo.getItemAt(i);
128       if (!newvals.remove(item)) {
129         toRemove.add(i);
130       }
131     }
132     for (Integer remove : toRemove) {
133       combo.removeItemAt(remove);
134     }
135     for (String newval : newvals) {
136       combo.addItem(newval);
137     }
138   }

140   private void update() throws IOException {
141     if (IS_UPDATE.get()) {
142       return;
143     }
144     IS_UPDATE.set(Boolean.TRUE);
145     try {
146       Instant start = Instant.ofEpochMilli(((Date) this.startDate.getValue()).getTime());
147       Instant end = Instant.ofEpochMilli(((Date) this.endDate.getValue()).getTime());
148       ProfileMetaData metaData = samplesSupplier.getMetaData(start, end);
149       sync(metaData.getTags(), tagsSelector);
150       sync(metaData.getContexts(), contextSelector);
151       this.samples = samplesSupplier.getSamples((String) contextSelector.getSelectedItem(),
152                 (String) tagsSelector.getSelectedItem(), start, end);
153       if (this.samples == null) {
154         this.samples = SampleNode.createSampleNode(
155                 new StackTraceElement[]{new StackTraceElement("NO SAMPLES", "", "", -1)});
156       }
157       resetViewType(graphToggle.isSelected());
158     } finally {
159       IS_UPDATE.set(Boolean.FALSE);
160     }
161   }

163   private void resetViewType(final boolean isgraph) {
164     if (isgraph) {
165       ssScrollPanel.setViewportView(new HotFlameStackPanel(Methods.ROOT, this.samples, new LinkedList<>()));
166     } else {
167       ssScrollPanel.setViewportView(new FlameStackPanel(Methods.ROOT, this.samples, new LinkedList<>()));
168     }
169   }

172   private void setViewType(final boolean isgraph) {
173     StackPanelBase view = (StackPanelBase) ssScrollPanel.getViewport().getView();
174     if (isgraph) {
175       graphToggle.setSelected(true);
176       //ssScrollPanel.setViewportView(new ZStackPanel(this.samples));
177       if (view != null) {
178         ssScrollPanel.setViewportView(new HotFlameStackPanel(view.getMethod(), view.getSamples(), view.getHistory()));
179       } else {
180         ssScrollPanel.setViewportView(new HotFlameStackPanel(Methods.ROOT, this.samples, new LinkedList<>()));
181       }
182     } else {
183       graphToggle.setSelected(false);
184       if (view != null) {
185         ssScrollPanel.setViewportView(new FlameStackPanel(view.getMethod(), view.getSamples(), view.getHistory()));
186       } else {
187         ssScrollPanel.setViewportView(new FlameStackPanel(Methods.ROOT, this.samples, new LinkedList<>()));
188       }
189     }
190   }

310   private void exportButtonActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_exportButtonActionPerformed
311     JFileChooser fc = new JFileChooser();
312     FileFilter[] xFF = fc.getChoosableFileFilters();
313     for (FileFilter ff : xFF) {
314       fc.removeChoosableFileFilter(ff);
315     }
316     fc.addChoosableFileFilter(Spf4jFileFilter.D3_JSON);
317     fc.addChoosableFileFilter(Spf4jFileFilter.SPF4J_JSON);
318     fc.addChoosableFileFilter(Spf4jFileFilter.SSDUMP2);
319     fc.addChoosableFileFilter(Spf4jFileFilter.SSDUMP2_GZ);
320     fc.setFileFilter(Spf4jFileFilter.D3_JSON);
321     int retrival = fc.showSaveDialog(null);
322     if (retrival == JFileChooser.APPROVE_OPTION) {
323       File selectedFile = fc.getSelectedFile();
324       FileFilter fileFilter = fc.getFileFilter();
325       if (!(fileFilter instanceof Spf4jFileFilter)) {
326         throw new IllegalArgumentException("Invalid file type selected " + fileFilter);
327       }
328       String suffix = ((Spf4jFileFilter) fileFilter).getSuffix();
329       if (!selectedFile.getName().endsWith(suffix)) {
330         selectedFile = new File(selectedFile.getParentFile(), selectedFile.getName() + suffix);
331       }
332       if (Spf4jFileFilter.D3_JSON.accept(selectedFile)) { // D3 format.
333         try (BufferedWriter wr = Files.newBufferedWriter(selectedFile.toPath(), StandardCharsets.UTF_8)) {
334           samples.writeD3JsonTo(wr);
335         } catch (IOException ex) {
336           throw new UncheckedIOException(ex);
337         }
338       } else if (Spf4jFileFilter.SPF4J_JSON.accept(selectedFile)) {
339         try (BufferedWriter wr = Files.newBufferedWriter(selectedFile.toPath(), StandardCharsets.UTF_8)) {
340           samples.writeJsonTo(wr);
341         } catch (IOException ex) {
342           throw new UncheckedIOException(ex);
343         }
344       } else if (Spf4jFileFilter.SSDUMP2.accept(selectedFile) || Spf4jFileFilter.SSDUMP2_GZ.accept(selectedFile)) {
345         try {
346           Converter.save(selectedFile, samples);
347         } catch (IOException ex) {
348           throw new UncheckedIOException(ex);
349         }
350       } else {
351         throw new IllegalArgumentException("No ecognized extension for file " + selectedFile);
352       }
353     }
354   }//GEN-LAST:event_exportButtonActionPerformed
