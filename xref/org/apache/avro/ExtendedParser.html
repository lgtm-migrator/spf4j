29  /**

31   * @author Zoltan Farkas

33  public class ExtendedParser {
34  
35    public static final int MAX_POWER_OF_TWO = 1 << (Integer.SIZE - 2);
36  
37    private static final ThreadLocal<Boolean> ALLOW_UNDEF_LT = new ThreadLocal<>();
38  
39  
40    private final ExtendedNames names;
41    private boolean validateNames = true;
42    private boolean validateDefaults = false;
43  
44    public static boolean isAllowUndefinedLogicalTypes() {
45      Boolean tlb = ALLOW_UNDEF_LT.get();
46      if (tlb == null) {
47        return Boolean.getBoolean("allowUndefinedLogicalTypes");
48      } else {
49        return tlb;
50      }
51    }
52  
53    public static void setAllowUndefinedLogicalTypesThreadLocal(final Boolean isAllowUndefinedLogicalTypes) {
54      ALLOW_UNDEF_LT.set(isAllowUndefinedLogicalTypes);
55    }
56  
57    public static Boolean getAllowUndefinedLogicalTypesThreadLocal() {
58      return ALLOW_UNDEF_LT.get();
59    }
60  
61    @SuppressFBWarnings("EI_EXPOSE_REP2")
62    public ExtendedParser(final ExtendedNames names) {
63      this.names = names;
64    }
65  
66    public ExtendedParser() {
67      SchemaResolver resolver = SchemaResolvers.getDefault();
68      names = new ExtendedAvroNamesRefResolver(resolver);
69    }
70  
71    /**
72     * Adds the provided types to the set of defined, named types known to this parser.
73     */
74    public final ExtendedParser addTypes(final Map<String, Schema> types) {
75      for (Schema s : types.values()) {
76        names.add(s);
77      }
78      return this;
79    }
80  
81    /**
82     * Returns the set of defined, named types known to this parser.
83     */
84    public final Map<String, Schema> getTypes() {
85      Collection<Schema> values = names.values();
86      Map<String, Schema> result = newLinkedHashMapWithExpectedSize(values.size());
87      for (Schema s : values) {
88        result.put(s.getFullName(), s);
89      }
90      return result;
91    }
92  
93    /**
94     * Enable or disable name validation.
95     */
96    public final ExtendedParser setValidate(final boolean validate) {
97      this.validateNames = validate;
98      return this;
99    }
100 
101   /**
102    * True iff names are validated. True by default.
103    */
104   public final boolean getValidate() {
105     return this.validateNames;
106   }
107 
108   /**
109    * Enable or disable default value validation.
110    */
111   public final ExtendedParser setValidateDefaults(final boolean pvalidateDefaults) {
112     this.validateDefaults = pvalidateDefaults;
113     return this;
114   }
115 
116   /**
117    * True iff default values are validated. False by default.
118    */
119   public final boolean getValidateDefaults() {
120     return this.validateDefaults;
121   }
122 
123   /**
124    * Parse a schema from the provided file. If named, the schema is added to the names known to this parser.
125    */
126   public final Schema parse(final File file) throws IOException {
127     return parse(FACTORY.createParser(file));
128   }
129 
130   public final Schema parse(final InputStream in) throws IOException {
131     return parse(in, isAllowUndefinedLogicalTypes());
132   }
133 
134   /**
135    * Parse a schema from the provided stream. If named, the schema is added to the names known to this parser.
136    */
137   public final Schema parse(final InputStream in, final boolean allowUndefinedLogicalTypes) throws IOException {
138     return parse(FACTORY.createParser(in), allowUndefinedLogicalTypes);
139   }
140 
141   /**
142    * Read a schema from one or more json strings
143    */
144   public final Schema parse(final String s, final String... more) {
145     StringBuilder b = new StringBuilder(s.length() * (more.length + 1));
146     b.append(s);
147     for (String part : more) {
148       b.append(part);
149     }
150     return parse(b.toString());
151   }
152 
153   public final Schema parse(final String s) {
154     return parse(s, isAllowUndefinedLogicalTypes());
155   }
156 
157   /**
158    * Parse a schema from the provided string. If named, the schema is added to the names known to this parser.
159    */
160   public final Schema parse(final String s, final boolean allowUndefinedLogicalTypes) {
161     try {
162       return parse(FACTORY.createParser(new StringReader(s)), allowUndefinedLogicalTypes);
163     } catch (IOException e) {
164       throw new SchemaParseException(e);
165     }
166   }
167 
168   public final Schema parse(final JsonParser parser) throws IOException {
169     return parse(parser, isAllowUndefinedLogicalTypes());
170   }
171 
172   @SuppressFBWarnings("AI_ANNOTATION_ISSUES_NEEDS_NULLABLE")
173   public final Schema parse(final JsonParser parser, final boolean allowUndefinedLogicalTypes) throws IOException {
174     return SchemaAdapter.parse(parser, names, allowUndefinedLogicalTypes, validateNames, validateDefaults);
175   }
176 
177   public static <K, V> LinkedHashMap<K, V> newLinkedHashMapWithExpectedSize(final int expectedSize) {
178     return new LinkedHashMap<K, V>(capacity(expectedSize));
179   }
180 
181   private static int capacity(final int expectedSize) {
182     if (expectedSize < 3) {
183       if (expectedSize < 0) {
184         throw new IllegalArgumentException("Invalid capacity: " + expectedSize);
185       }
186       return expectedSize + 1;
187     }
188     if (expectedSize < MAX_POWER_OF_TWO) {
189       // This is the calculation used in JDK8 to resize when a putAll
190       // happens; it seems to be the most conservative calculation we
191       // can make.  0.75 is the default load factor.ss
192       return (int) ((float) expectedSize / 0.75F + 1.0F);
193     }
194     return Integer.MAX_VALUE; // any large value
195   }
196 
197   /**
198    * Overwrite as appropriate.
199    * @return
200    */
201   @Override
202   public String toString() {
203     return "ExtendedParser{" + "names=" + names + ", validate=" + validateNames
204             + ", validateDefaults=" + validateDefaults + '}';
205   }
