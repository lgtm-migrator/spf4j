16  package org.spf4j.maven.plugin.avro.avscp.validation.impl;
17  
18  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
19  import java.util.ArrayList;
20  import java.util.List;
21  import javax.annotation.Nonnull;
22  import javax.annotation.Nullable;
23  import org.apache.avro.Schema;
24  import org.apache.avro.Schema.Field;
25  import org.spf4j.avro.schema.SchemaVisitor;
26  import org.spf4j.avro.schema.SchemaVisitorAction;
27  import org.spf4j.maven.plugin.avro.avscp.validation.Validator;
28  import org.spf4j.avro.schema.Schemas;
29  import org.spf4j.base.CharSequences;
30  import org.spf4j.maven.plugin.avro.avscp.ValidatorMojo;
31  import org.spf4j.maven.plugin.avro.avscp.validation.Validators;
32  
33  /**
34   * Validates schema documentation fields are not empty for:
35   * Records, Fixed, Enum, Record Fields.
36   * <p> Additionally for record fields of union type where one type is null, the doc is validated for the
37   * presence of the "null" string, with the expectation that  the documentation contains a explanation on what
38   * in means. </p>
39   *
40   * @author Zoltan Farkas
41   */
42  public final class SchemaDocValidator implements Validator<Schema> {
43  
44    @Override
45    public String getName() {
46      return "docValidator";
47    }
48  
49    @Override
50    @Nonnull
51    @SuppressFBWarnings("AI_ANNOTATION_ISSUES_NEEDS_NULLABLE") // not in this case
52    public Result validate(final Schema schema, final ValidatorMojo mojo) {
53      return Schemas.visit(schema, new DocValidatorVisitor(schema));
54    }
55  
56    @Nullable
57    private static Schema.Type getCollectionType(final Schema unionSchema) {
58      for (Schema schema : unionSchema.getTypes()) {
59        Schema.Type type = schema.getType();
60        switch (type) {
61          case ARRAY:
62          case MAP:
63          case STRING:
64            return type;
65          default:
66          // skip
67        }
68      }
69      return null;
70    }
71  
72    @Override
73    public Class<Schema> getValidationInput() {
74      return Schema.class;
75    }
76  
77    private static class DocValidatorVisitor implements SchemaVisitor<Result> {
78  
79      private final List<String> issues;
80  
81      private final Schema root;
82  
83      DocValidatorVisitor(final Schema root) {
84        issues = new ArrayList<>(4);
85        this.root = root;
86      }
87  
88      @Override
89      public SchemaVisitorAction visitTerminal(final Schema schema) {
90        switch (schema.getType()) {
91          case ENUM:
92          case FIXED:
93            String doc = schema.getDoc();
94            if (doc == null || doc.trim().isEmpty()) {
95              issues.add("Please document " + schema.getFullName());
96            }
97            break;
98          default:
99        }
100       return SchemaVisitorAction.CONTINUE;
101     }
102 
103     @Override
104     public SchemaVisitorAction visitNonTerminal(final Schema schema) {
105       if (schema.getType() == Schema.Type.RECORD) {
106         if (Validators.skipValidator(schema, "docValidator")) {
107           return SchemaVisitorAction.CONTINUE;
108         }
109         String doc = schema.getDoc();
110         if (doc == null || doc.trim().isEmpty()) {
111           issues.add("Please document " + schema.getFullName());
112         }
113         for (Field field : schema.getFields()) {
114           doc = field.doc();
115           if (doc == null || doc.trim().isEmpty()) {
116             String errText = "Please document " + field.name() + '@' + schema.getFullName();
117             String source = schema.getProp("sourceIdl");
118             if (source != null) {
119               errText += " from " + source;
120             }
121             issues.add(errText);
122           } else {
123             Schema fs = field.schema();
124             if (Schemas.isNullableUnion(fs) && !CharSequences.containsIgnoreCase(doc, "null")) {
125               String issue = "please document the meaning of null for field " + field.name() + '@'
126                       + schema.getFullName();
127               String source = schema.getProp("sourceIdl");
128               if (source != null) {
129                 issue += " from " + source;
130               }
131               Schema.Type collectionType = getCollectionType(fs);
132               if (collectionType != null) {
133                 issue += " and explain how its meaning is different from empty " + collectionType;
134               }
135               issues.add(issue);
136             }
137           }
138 
139         }
140       }
141       return SchemaVisitorAction.CONTINUE;
142     }
143 
144     @Override
145     public SchemaVisitorAction afterVisitNonTerminal(final Schema nonTerminal) {
146       return SchemaVisitorAction.CONTINUE;
147     }
148 
149     @Override
150     @Nonnull
151     public Result get() {
152       return issues.isEmpty() ? Result.valid()
153               : Result.failed("Schema " + root.getFullName() + " doc issues:\n" + String.join("\n", issues));
154     }
155   }
