144   @SuppressWarnings("unchecked")
145   private void loadPrevReleaseId2Map() throws IOException {
146     MavenProject mavenProject = getMavenProject();
147     Log log = getLog();
148     String versionRange = "[," + mavenProject.getVersion() +  ')';
149     String groupId = mavenProject.getGroupId();
150     String artifactId = mavenProject.getArtifactId();
151     List<RemoteRepository> remoteProjectRepositories = mavenProject.getRemoteProjectRepositories();
152     RepositorySystem repoSystem = getRepoSystem();
153     RepositorySystemSession repositorySession = getMavenSession().getRepositorySession();
154     List<Version> rangeVersions;
155     try {
156       rangeVersions = MavenRepositoryUtils.getVersions(groupId, artifactId, versionRange,
157               remoteProjectRepositories, repoSystem, repositorySession);
158     } catch (VersionRangeResolutionException ex) {
159       throw new RuntimeException("Invalid compatibiliy.versionRange = " + versionRange + " setting", ex);
160     }
161     rangeVersions = rangeVersions.stream().filter((v) -> !v.toString().endsWith("SNAPSHOT"))
162             .collect(Collectors.toList());
163     int tSize = rangeVersions.size();
164     rangeVersions = rangeVersions.subList(Math.max(tSize - 1, 0), tSize);
165     log.info("Loading id 2 name map from " + rangeVersions);
166     if (rangeVersions.isEmpty()) {
167       return;
168     }
169     Version version  = rangeVersions.get(0);
170     Path targetPath = getTarget().toPath();
171     File prevSchemaArchive;
172     try {
173       prevSchemaArchive = MavenRepositoryUtils.resolveArtifact(
174               groupId, artifactId, schemaArtifactClassifier, schemaArtifactExtension, version.toString(),
175               remoteProjectRepositories, repoSystem, repositorySession);
176     } catch (ArtifactResolutionException ex) {
177       throw new RuntimeException("Cannot resolve previous version "  + version, ex);
178     }
179     Path dest = targetPath.resolve("prevSchema").resolve(version.toString());
180     Files.createDirectories(dest);
181     log.debug("Unzipping " + prevSchemaArchive + " to " + dest);
182     List<Path> indexFiles = Compress.unzip2(prevSchemaArchive.toPath(), dest, (Path p) -> {
183       Path fileName = p.getFileName();
184       if (fileName == null) {
185         return false;
186       }
187       return SCHEMA_INDEX_FILENAME.equals(fileName.toString());
188     });
189     Properties prevIndex = new Properties();
190     if (indexFiles.size() != 1) {
191       log.info("no index file or to many in previous version: " + indexFiles);
192     } else {
193       // load previous index file
194       Path indexFile = indexFiles.get(0);
195       try (BufferedReader br = Files.newBufferedReader(indexFile, StandardCharsets.UTF_8)) {
196         prevIndex.load(br);
197       }
198     }
199     for (Map.Entry<String, String> entry : (Set<Map.Entry<String, String>>) (Set) prevIndex.entrySet()) {
200       String key = entry.getKey();
201       if (SCHEMA_INDEX_PGK_KEY.equals(key)) {
202         continue;
203       }
204       int idx = Integer.parseInt(key, 32);
205       if (idx >= idSequence) {
206         idSequence = idx + 1;
207       }
208       prevReleaseName2Index.put(entry.getValue(), idx);
209     }
210     log.debug("loaded existing mappings: " + prevReleaseName2Index);
211     log.info("loaded existing mappings, new id sequence: " + idSequence);
212   }

214   private String attachMavenId(final Schema schema) {
215     String exMvnId = schema.getProp("mvnId");
216     if (exMvnId == null) {
217       String newId = genMnvId(schema);
218       schema.addProp("mvnId", newId);
219       return newId;
220     } else {
221       return exMvnId;
222     }

232   public String genMnvId(final Schema schema) {
233     StringBuilder idBuilder = new StringBuilder(64);
234     idBuilder.append(mavenProject.getGroupId()).append(':').append(mavenProject.getArtifactId())
235             .append(':').append(mavenProject.getVersion());
236     Integer idx = prevReleaseName2Index.get(schema.getFullName());
237     if (idx == null) {
238       idx = idSequence++;
239     }
240     StringBuilder idb = new StringBuilder(4);
241     AppendableUtils.appendUnsignedString(idb, idx, 5);
242     idBuilder.append(':').append(idb);
243     index.put(idb.toString(), schema);
244     return idBuilder.toString();
245   }

247   protected void doCompileIDL(final File sourceDir,  final String filename) throws IOException {
248     Thread currentThread = Thread.currentThread();
249     ClassLoader contextClassLoader = currentThread.getContextClassLoader();
250     try {
251       List<URL> runtimeUrls = createPathUrls(sourceDir);
252       getLog().info("IDL Compile classpath: " + runtimeUrls);
253       URLClassLoader projPathLoader = AccessController.doPrivileged(
254               (PrivilegedAction<URLClassLoader>) ()
255               -> new URLClassLoader(runtimeUrls.toArray(new URL[runtimeUrls.size()]), contextClassLoader));
256       currentThread.setContextClassLoader(projPathLoader);
257       File file = new File(sourceDir, filename);
258       String sourceAbsolutePath = sourceDir.getAbsolutePath();
259       // set the current dir do that sourceIdl will be computed relative to it.
260       // This makes this plugin not thread safe.
261       Idl parser;
262       String origCurrentDir = org.spf4j.base.Runtime.getCurrentDir();
263       org.spf4j.base.Runtime.setCurrentDir(sourceAbsolutePath);
264       try {
265         parser = new Idl(file, projPathLoader);
266       } finally {
267         org.spf4j.base.Runtime.setCurrentDir(origCurrentDir);
268       }
269       Protocol protocol = parser.CompilationUnit();
270       publishSchemasAndAttachMvnIdToProtocol(protocol, false, useSchemaReferencesForAvsc);
271       SpecificCompiler compiler = new SpecificCompiler(protocol);
272       compiler.setOutputCharacterEncoding(mavenProject.getProperties().getProperty("project.build.sourceEncoding"));
273       compiler.setStringType(GenericData.StringType.valueOf(stringType));
274       compiler.setTemplateDir(templateDirectory);
275       compiler.setFieldVisibility(SpecificCompiler.FieldVisibility.valueOf(fieldVisibility));
276       compiler.setCreateSetters(createSetters);
277       compiler.compileToDestination(null, generatedJavaTarget);
278     } catch (ParseException e) {
279       throw new IOException(e);
280     } catch (DependencyResolutionRequiredException drre) {
281       throw new IOException(drre);
282     } finally {
283       currentThread.setContextClassLoader(contextClassLoader);
284     }
285   }

287   public List<URL> createPathUrls(final File sourceFolder)
288           throws MalformedURLException, DependencyResolutionRequiredException {
289     List<String> cpElements = mavenProject.getRuntimeClasspathElements();
290     List<URL> runtimeUrls = new ArrayList<>();
291     runtimeUrls.add(sourceFolder.toURI().toURL());
292     // If runtimeClasspathElements is not empty values add its values to Idl path.
293     if (cpElements != null && !cpElements.isEmpty()) {
294       for (String element : cpElements) {
295         runtimeUrls.add(new File(element).toURI().toURL());
296       }
297     }
298     return runtimeUrls;
299   }

301   private File addMvnIdsToIdl(final File idl, final URLClassLoader cl)
302           throws IOException, ParseException {
303     if (!addMavenId) {
304       return idl;
305     }
306     String charsetStr = mavenProject.getProperties().getProperty("project.build.sourceEncoding");
307     Charset charset = charsetStr == null ? Charset.defaultCharset() : Charset.forName(charsetStr);
308     List<String> readAllLines = Files.readAllLines(idl.toPath(), charset);
309     Idl parser = new Idl(idl, cl);
310     Protocol protocol = parser.CompilationUnit();
311     // hack uses the same logic and Idl...
312     String idlSource = new File(".").toURI().relativize(idl.toURI()).toString();
313     Log log = getLog();
314     log.debug("Injecting mvnIds to " + idlSource);
315     for (Schema s : protocol.getTypes()) {
316       if (s.getProp("mvnId") != null) {
317         continue;
318       }
319       String sourceIdl = s.getProp("sourceIdl");
320       if (sourceIdl == null) {
321         log.warn("sourceIdl not available, will not attach mvnId for IDLs");
322         continue;
323       }
324       SourceLocation sl = new SourceLocation(sourceIdl);
325       if (!idlSource.equals(sl.getFilePath())) {
326         continue;
327       }
328       int zbLineNr = sl.getLineNr() - 1;
329       String line = readAllLines.get(zbLineNr);
330       String sMvnId = genMnvId(s);
331       log.debug("inserting mvnId: " + sMvnId + " at "
332               + sl + " for line \"" + line + "\" schema: " + s.getFullName());
333       int zbColNr = sl.getColNr() - 1;
334       readAllLines.set(zbLineNr, line.substring(0, zbColNr)
335               + " @mvnId(\"" + sMvnId + "\") "
336               + line.substring(zbColNr, line.length()));
337     }
338     Path tempIdl = Files.createTempFile(this.target.toPath(), idl.getName(), ".tmp");
339     Files.write(tempIdl, readAllLines, charset);
340     return tempIdl.toFile();
341   }

344   protected void doCompileSchemas(final String[] filenames)
345           throws IOException {
346     getLog().debug("Compiling: " + Arrays.toString(filenames) + ", from " + sourceDirectory);
347     ClassLoader avroLibClassLoader = org.apache.avro.Schema.class.getClassLoader();
348     Thread currentThread = Thread.currentThread();
349     ClassLoader contextClassLoader = currentThread.getContextClassLoader();
350     currentThread.setContextClassLoader(avroLibClassLoader);
351     try {
352     ExtendedParser parser = new UnresolvedExtendedParser();
353     Map<String, Schema> schemas = Maps.newHashMapWithExpectedSize(filenames.length);
354     Map<String, File> srcFiles = Maps.newHashMapWithExpectedSize(filenames.length);
355     for (String fileName : filenames) {
356       File src = new File(sourceDirectory, fileName);
357       Schema schema = parser.parse(src);
358       if (addMavenId) {
359         attachMavenId(schema);
360       }
361       String fullName = schema.getFullName();
362       schemas.put(fullName, schema);
363       srcFiles.put(fullName, src);
364     }
365     List<Schema> schemaList = org.apache.avro.avsc.SchemaResolver.resolve(schemas,
366             Boolean.getBoolean("allowUndefinedLogicalTypes"));
367     for (Schema schema : schemaList) {
368       writeSchemaToTarget(schema, srcFiles.get(schema.getFullName()));
369     }
370     } finally {
371       currentThread.setContextClassLoader(contextClassLoader);
372     }
373   }

375   private void writeSchemaToTarget(final Schema schema, final File src) throws IOException {
376     String targetName = schema.getFullName().replace('.', File.separatorChar) + ".avsc";
377     Path destination = generatedAvscTarget.toPath().resolve(targetName);
378     Path parent = destination.getParent();
379     if (parent != null) {
380       Files.createDirectories(parent);
381     }
382     Files.write(destination,
383             schema.toString().getBytes(StandardCharsets.UTF_8),
384             StandardOpenOption.CREATE, StandardOpenOption.TRUNCATE_EXISTING);
385     SpecificCompiler compiler = new SpecificCompiler(schema);
386     compiler.setOutputCharacterEncoding(mavenProject.getProperties().getProperty("project.build.sourceEncoding"));
387     compiler.setTemplateDir(templateDirectory);
388     compiler.setStringType(GenericData.StringType.valueOf(stringType));
389     compiler.setFieldVisibility(SpecificCompiler.FieldVisibility.valueOf(fieldVisibility));
390     compiler.setCreateSetters(createSetters);
391     compiler.compileToDestination(src, generatedJavaTarget);
392   }

411   private void publishSchemasAndAttachMvnIdToProtocol(final Protocol protocol,
412           final boolean addMvnId, final boolean useSchemaReferences) throws IOException {
413     Collection<Schema> types = protocol.getTypes();
414     Set<String> typeNames = Sets.newHashSetWithExpectedSize(types.size());
415     for (Schema schema : types) {
416       String fullName = schema.getFullName();
417       if (!typeNames.add(fullName)) {
418         continue;
419       }
420       String targetName = fullName.replace('.', File.separatorChar) + ".avsc";
421       Path destinationFile = generatedAvscTarget.toPath().resolve(targetName);
422       Path parent = destinationFile.getParent();
423       if (parent != null) {
424         Files.createDirectories(parent);
425       }
426       if (addMvnId) {
427         attachMavenId(schema);
428       }
429       if (useSchemaReferences) {
430         try (OutputStream fos =
431                 new BufferedOutputStream(Files.newOutputStream(destinationFile,
432                         StandardOpenOption.CREATE, StandardOpenOption.TRUNCATE_EXISTING))) {
433           SchemaRefWriter.write(schema, fos);
434         }
435       } else {
436         Files.write(destinationFile, schema.toString().getBytes(StandardCharsets.UTF_8),
437                 StandardOpenOption.CREATE, StandardOpenOption.TRUNCATE_EXISTING);
438       }
439     }
440   }

442   public void deleteGeneratedAvailableInDependencies() throws IOException {
443     Path classesInfo = dependenciesDirectory.toPath().resolve("classes.txt");
444     Set<String> classes = new HashSet(Files.readAllLines(classesInfo, StandardCharsets.UTF_8));
445     Path javaPath = generatedJavaTarget.toPath();
446     try (Stream<Path> fsStream = Files.walk(javaPath)) {
447       List<Path> dupes = fsStream
448               .filter((p) -> {
449                 Path relativize = javaPath.relativize(p);
450                 return classes.contains(relativize.toString().replace(".java", ".class"));
451               }).collect(Collectors.toList());
452       for (Path p : dupes) {
453         Files.delete(p);
454         getLog().info("Deleted dupes: " + dupes);
455       }
456     }

459   public void deleteProtocolClasses() throws IOException {
460     String detectionString = "org.apache.avro.Protocol PROTOCOL";
461     Path javaPath = generatedJavaTarget.toPath();
462     String mSourceEncoding = mavenProject.getProperties().getProperty("project.build.sourceEncoding");
463     String sourceEncoding;
464     if (mSourceEncoding == null) {
465       sourceEncoding = Charset.defaultCharset().name();
466     } else {
467       sourceEncoding = mSourceEncoding;
468     }
469     try (Stream<Path> fsStream = Files.walk(javaPath)) {
470       List<Path> protocolFiles = fsStream
471               .filter((p) -> {
472                 Path fileName = p.getFileName();
473                 if (fileName == null || !fileName.toString().endsWith(".java")) {
474                   return false;
475                 }
476                 try (BufferedReader br = Files.newBufferedReader(p, Charset.forName(sourceEncoding))) {
477                   String line;
478                   while ((line = br.readLine()) != null) {
479                     if (line.contains(detectionString)) {
480                       return true;
481                     }
482                   }
483                 } catch (IOException ex) {
484                   getLog().info("cannot read file " + p + ", ignoring for cleanup", ex);
485                 }
486                 return false;
487               }).collect(Collectors.toList());
488       for (Path p : protocolFiles) {
489         Files.delete(p);
490       }
491     }
492   }

494   public void deleteSchemasAvailableInDependencies(final Path schTargetPath) throws IOException {
495     Path classesInfo = dependenciesDirectory.toPath();
496     Set<Path> schemas = Files.walk(classesInfo).filter(
497             (p) -> {
498               Path fileName = p.getFileName();
499               return fileName == null ? false : fileName.toString().endsWith("avsc");
500             })
501             .map((p) -> classesInfo.relativize(p)).collect(Collectors.toSet());
502     List<Path> dupes = Files.walk(schTargetPath).filter((p) -> schemas.contains(schTargetPath.relativize(p)))
503             .collect(Collectors.toList());
504     for (Path p : dupes) {
505       Files.delete(p);
506     }
507     getLog().info("Deleted dupes: " + dupes);
508   }

510   @Override
511   public void execute() throws MojoExecutionException, MojoFailureException {
512     try {
513       loadPrevReleaseId2Map();
514     } catch (IOException ex) {
515       throw new MojoExecutionException("Unable to proces previous release of "
516               + getMavenProject().getVersion(), ex);
517     }
518     super.execute();
519     Log logger = this.getLog();
520     logger.info("Generating java code + schemas, using avro "
521             + PackageInfo.getPackageInfo(org.apache.avro.Schema.class.getName()));
522     synchronized (String.class) {
523       Properties properties = new Properties(System.getProperties());
524       try {
525         for (Map.Entry<String, String> entry : ((Set<Map.Entry<String, String>>) ((Set) systemProperties.entrySet()))) {
526           System.setProperty(entry.getKey(), entry.getValue());
527         }
528         Path generatedAvscTargetPath = generatedAvscTarget.toPath();
529         Files.createDirectories(generatedAvscTargetPath);
530         Files.createDirectories(generatedJavaTarget.toPath());
531         String[] sourceFiles = getSourceFiles("**/*.avsc");
532         Arrays.sort(sourceFiles); // make this predictable, (mostly easier to test)
533         try {
534           doCompileSchemas(sourceFiles);
535         } catch (IOException ex) {
536           throw new MojoExecutionException("cannot compile schemas " + Arrays.toString(sourceFiles), ex);
537         }
538         Path tmpSourceTarget = this.target.toPath().resolve("avro-sources");
539         compileAvpr(tmpSourceTarget);
540         addMvnIdToIdlsAndMoveToDestination(tmpSourceTarget);
541         compileIdl(tmpSourceTarget);
542 
543         Path codegenManifest = generatedAvscTargetPath.resolve(SCHEMA_MANIFEST);
544         try {
545           Files.write(codegenManifest,
546                   Collections.singletonList("Build-Time=" + DateTimeFormatter.ISO_INSTANT.format(Instant.now()) + '\n'),
547                   StandardCharsets.UTF_8);
548         } catch (IOException ex) {
549           throw new MojoExecutionException("Cannot create codegen manifest file " + codegenManifest, ex);
550         }
551         try {
552           deleteGeneratedAvailableInDependencies();
553           deleteSchemasAvailableInDependencies(getGeneratedAvscTarget().toPath());
554           if (deleteProtocolInterface) {
555             deleteProtocolClasses();
556           }
557         } catch (IOException ex) {
558           throw new MojoExecutionException("Cannot delete dependency dupes " + this, ex);
559         }
560         Path indexFile = generatedAvscTargetPath.resolve(SCHEMA_INDEX_FILENAME);
561         try (BufferedWriter bw = Files.newBufferedWriter(indexFile,
562                 StandardCharsets.UTF_8, StandardOpenOption.CREATE, StandardOpenOption.TRUNCATE_EXISTING)) {
563           bw.append(SCHEMA_INDEX_PGK_KEY);
564           bw.append('=');
565           bw.append(getPackageMvnIdPrefix());
566           bw.append('\n');
567           for (Map.Entry<String, Schema> entry : index.entrySet()) {
568             bw.append(entry.getKey());
569             bw.append('=');
570             bw.append(entry.getValue().getFullName());
571             bw.append('\n');
572           }
573         } catch (IOException ex) {
574           throw new MojoExecutionException("Cannot generate schema index " + this, ex);
575         }
576         mavenProject.addCompileSourceRoot(generatedJavaTarget.getAbsolutePath());
577         Resource resource = new Resource();
578         resource.setDirectory(this.generatedAvscTarget.getAbsolutePath());
579         resource.addInclude("**/*.avsc");
580         resource.addInclude("*.properties");
581         mavenProject.addResource(resource);
582         Resource resource2 = new Resource();
583         resource2.setDirectory(tmpSourceTarget.toString());
584         resource2.addInclude("**/*.avpr");
585         resource2.addInclude("**/*.avdl");
586         mavenProject.addResource(resource2);
587       } catch (IOException ex) {
588         throw new MojoExecutionException("cannot compile schemas, cfg = " + this, ex);
589       } finally {
590         System.setProperties(properties);
591       }
592     }
593   }

606   public void addMvnIdToIdlsAndMoveToDestination(final Path destPath) throws MojoExecutionException {
607     Thread currentThread = Thread.currentThread();
608     ClassLoader contextClassLoader = currentThread.getContextClassLoader();
609     try {
610       List<URL> runtimeUrls = createPathUrls(this.sourceDirectory);
611       getLog().info("Compile classpath: " + runtimeUrls);
612       URLClassLoader projPathLoader = AccessController.doPrivileged(
613               (PrivilegedAction<URLClassLoader>) ()
614                       -> new URLClassLoader(runtimeUrls.toArray(new URL[runtimeUrls.size()]), contextClassLoader));
615       currentThread.setContextClassLoader(projPathLoader);
616       for (String file : getSourceFiles("**/*.avdl")) {
617         Path destination = destPath.resolve(file);
618         Path parent = destination.getParent();
619         if (parent != null) {
620           Files.createDirectories(parent);
621         }
622         File idlFile = new File(sourceDirectory, file);
623         try {
624           idlFile = addMvnIdsToIdl(idlFile, projPathLoader);
625         } catch (ParseException | IOException | RuntimeException ex) {
626           throw new MojoExecutionException("cannot add mvnId to  IDL " + idlFile + ", " + ex.getMessage(), ex);
627         }
628         Files.copy(idlFile.toPath(), destination, StandardCopyOption.REPLACE_EXISTING);
629       }
630     } catch (IOException | DependencyResolutionRequiredException ex) {
631       throw new MojoExecutionException("cannot add mvnId to  IDL " + this, ex);
632     } finally {
633       currentThread.setContextClassLoader(contextClassLoader);
634     }
635   }
