16  package org.spf4j.avro.csv;
17  
18  import edu.umd.cs.findbugs.annotations.SuppressFBWarnings;
19  import java.io.IOException;
20  import java.nio.ByteBuffer;
21  import javax.annotation.Nullable;
22  import org.apache.avro.Schema;
23  import org.apache.avro.io.Encoder;
24  import org.apache.avro.io.parsing.JsonGrammarGenerator;
25  import org.apache.avro.io.parsing.Parser;
26  import org.apache.avro.io.parsing.Symbol;
27  import org.apache.avro.util.Utf8;
28  import org.spf4j.base.Base64;
29  import org.spf4j.io.csv.CsvWriter;
30  
31  /**
32   * A Csv decoder for avro records.
33   *
34   * @author Zoltan Farkas
35   */
36  public final class CsvEncoder extends Encoder implements Parser.ActionHandler {
37  
38    private final CsvWriter csvWriter;
39  
40    private final Parser parser;
41  
42    private final Schema writerSchema;
43  
44    private boolean firstItem = true;
45  
46    @SuppressFBWarnings("EI_EXPOSE_REP2")
47    public CsvEncoder(final CsvWriter csvWriter, final Schema writerSchema) {
48      this.csvWriter = csvWriter;
49      this.parser = new Parser(new JsonGrammarGenerator().generate(writerSchema), this);
50      if (writerSchema.getType() != Schema.Type.ARRAY) {
51        throw new IllegalArgumentException("schema must be an array: " + writerSchema);
52      }
53      if (writerSchema.getElementType().getType() != Schema.Type.RECORD) {
54        throw new IllegalArgumentException("schema  array element must be record: " + writerSchema);
55      }
56      this.writerSchema = writerSchema;
57    }
58  
59    public void writeHeader() throws IOException {
60      writeHeader("", writerSchema.getElementType());
61      csvWriter.writeEol();
62    }
63  
64    private void writeHeader(final String prefix, final Schema rowSchema) throws IOException {
65      for (Schema.Field field : rowSchema.getFields()) {
66        Schema schema = field.schema();
67        Schema.Type type = schema.getType();
68        String name = prefix + field.name();
69        switch (type) {
70          case ARRAY:
71          case MAP:
72            throw new UnsupportedOperationException(type + " unsupported in CSV row" + rowSchema);
73          case UNION:
74            csvWriter.writeElement(name + ".type");
75            for (Schema us : schema.getTypes()) {
76              switch (us.getType()) {
77                case ARRAY:
78                case RECORD:
79                case MAP:
80                  throw new UnsupportedOperationException(type + " unsupported in CSV row union" + rowSchema);
81                default:
82                  continue;
83              }
84            }
85            csvWriter.writeElement(name);
86            break;
87          case RECORD:
88            writeHeader(name + '.', schema);
89            break;
90          default:
91            csvWriter.writeElement(name);
92        }
93      }
94    }
95  
96    @Override
97    public void writeNull() throws IOException {
98      parser.advance(Symbol.NULL);
99      csvWriter.writeElement("");
100   }
101 
102   @Override
103   public void writeBoolean(final boolean bln) throws IOException {
104     parser.advance(Symbol.BOOLEAN);
105     csvWriter.writeElement(Boolean.toString(bln));
106   }
107 
108   @Override
109   public void writeInt(final int i) throws IOException {
110     parser.advance(Symbol.INT);
111     csvWriter.writeElement(Integer.toString(i));
112   }
113 
114   @Override
115   public void writeLong(final long l) throws IOException {
116     parser.advance(Symbol.LONG);
117     csvWriter.writeElement(Long.toString(l));
118   }
119 
120   @Override
121   public void writeFloat(final float f) throws IOException {
122     parser.advance(Symbol.FLOAT);
123     csvWriter.writeElement(Float.toString(f));
124   }
125 
126   @Override
127   public void writeDouble(final double d) throws IOException {
128     parser.advance(Symbol.DOUBLE);
129     csvWriter.writeElement(Double.toString(d));
130   }
131 
132   @Override
133   public void writeString(final Utf8 utf8) throws IOException {
134     parser.advance(Symbol.STRING);
135     csvWriter.writeElement(utf8);
136   }
137 
138   @Override
139   public void writeBytes(final ByteBuffer bb) throws IOException {
140     parser.advance(Symbol.BYTES);
141     int position = bb.position();
142     csvWriter.writeElement(Base64.encodeBase64(bb.array(), bb.arrayOffset() + position, bb.limit() - position));
143   }
144 
145   @Override
146   public void writeBytes(final byte[] bytes, final int i, final int length) throws IOException {
147     parser.advance(Symbol.BYTES);
148     csvWriter.writeElement(Base64.encodeBase64(bytes, i, length));
149   }
150 
151   @Override
152   public void writeFixed(final byte[] bytes, final int i, final int length) throws IOException {
153     parser.advance(Symbol.FIXED);
154     csvWriter.writeElement(Base64.encodeBase64(bytes, i, length));
155   }
156 
157   @Override
158   public void writeEnum(final int e) throws IOException {
159     parser.advance(Symbol.ENUM);
160     Symbol.EnumLabelsAction top = (Symbol.EnumLabelsAction) parser.popSymbol();
161     csvWriter.writeElement(top.getLabel(e));
162   }
163 
164   @Override
165   public void writeArrayStart() throws IOException {
166     firstItem = true;
167     parser.advance(Symbol.ARRAY_START);
168     // do nothing.
169   }
170 
171   @Override
172   public void writeArrayEnd() throws IOException {
173     parser.advance(Symbol.ITEM_END);
174     parser.advance(Symbol.ARRAY_END);
175     // do nothing.
176   }
177 
178   @Override
179   public void writeMapStart() {
180     throw new UnsupportedOperationException();
181   }
182 
183   @Override
184   public void writeMapEnd() {
185     throw new UnsupportedOperationException();
186   }
187 
188   @Override
189   public void writeIndex(final int unionIndex) throws IOException {
190     parser.advance(Symbol.UNION);
191     Symbol.Alternative top = (Symbol.Alternative) parser.popSymbol();
192     Symbol symbol = top.getSymbol(unionIndex);
193     csvWriter.writeElement(top.getLabel(unionIndex));
194     parser.pushSymbol(Symbol.UNION_END);
195     parser.pushSymbol(symbol);
196   }
197 
198   @Override
199   public void flush() throws IOException {
200     csvWriter.flush();
201   }
202 
203   @Override
204   @Nullable
205   public Symbol doAction(final Symbol symbol, final Symbol symbol1) {
206     return null;
207   }
208 
209   @Override
210   public void setItemCount(final long l) {
211     // nothing to do.
212   }
213 
214   @Override
215   public void startItem() throws IOException {
216     if (!firstItem) {
217       parser.advance(Symbol.ITEM_END);
218       csvWriter.writeEol();
219     } else {
220       firstItem = false;
221     }
222     // nothing to do
223   }
224 
225   @Override
226   public String toString() {
227     return "CsvEncoder{" + "csvWriter=" + csvWriter + ", parser=" + parser
228             + ", writerSchema=" + writerSchema + ", firstItem=" + firstItem + '}';
229   }
230 
231 }
