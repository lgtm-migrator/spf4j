61    @Test
62    public void testSqlParser() throws SqlParseException {
63      SqlParser.Config cfg = SqlParser.configBuilder()
64              .setCaseSensitive(true)
65              .setIdentifierMaxLength(255)
66              .setLex(Lex.JAVA).build();
67      SqlParser parser = SqlParser.create("select a.id, a.name as n1, b.name as n2 from a, b where  a.id = b.id", cfg);
68      SqlSelect select = (SqlSelect) parser.parseQuery();
69      LOG.debug("Select", select);
70      Assert.assertNotNull(select);
71    }

101   @Test
102   @SuppressFBWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
103   @PrintLogs(category = "org.codehaus.janino", ideMinLevel = Level.INFO, greedy = true)
104   public void testAvroSql() throws SqlParseException, RelConversionException,
105           ValidationException, InstantiationException, IllegalAccessException {
106     Schema recASchema = SchemaBuilder.record("RecordA")
107             .fields().name("id").type().intType().noDefault()
108             .requiredString("name").endRecord();
109     Schema subRecSchema = SchemaBuilder.record("SubRecord")
110             .fields().name("key").type().stringType().noDefault()
111             .requiredString("value").endRecord();
112     Schema recBSchema = SchemaBuilder.record("RecordB")
113             .fields().name("id").type().intType().noDefault()
114             .requiredString("name")
115             .requiredString("text")
116             .name("adate").type(Schemas.dateString()).noDefault()
117             .name("meta").type(Schema.createArray(subRecSchema)).noDefault()
118             .name("meta2").type(subRecSchema).noDefault()
119             .endRecord();
120 
121     GenericRecordBuilder rb = new GenericRecordBuilder(recASchema, subRecSchema, recBSchema);
122     Class<? extends SpecificRecordBase> raC = rb.getRecordClass(recASchema);
123     Class<? extends SpecificRecordBase> rbC = rb.getRecordClass(recBSchema);
124     Class<? extends SpecificRecordBase> rsC = rb.getRecordClass(subRecSchema);
125 
126 
127     GenericRecord reca1 = raC.newInstance();
128     reca1.put("id", 1);
129     reca1.put("name", "Jim");
130 
131 
132     GenericRecord subRec = rsC.newInstance();
133     subRec.put("key", "key1");
134     subRec.put("value", "val1");
135 
136 
137     GenericRecord recb1 = rbC.newInstance();
138     recb1.put("id", 1);
139     recb1.put("name", "Beam");
140     recb1.put("text", "bla");
141     recb1.put("adate", LocalDate.now());
142     recb1.put("meta", Collections.singletonList(subRec));
143     recb1.put("meta2", subRec);
144 
145     GenericRecord recb2 = rbC.newInstance();
146     recb2.put("id", 2);
147     recb2.put("name", "Xi");
148     recb2.put("text", "blabla");
149     recb2.put("adate", LocalDate.now());
150     recb2.put("meta", Collections.singletonList(subRec));
151     recb2.put("meta2", subRec);
152 
153 
154     SchemaPlus schema = Frameworks.createRootSchema(true);
155     schema.add("a", new AvroIteratorAsProjectableFilterableTable(recASchema,
156             () -> CloseableIterator.from(Collections.singletonList(reca1).iterator())));
157     schema.add("b", new AvroIteratorAsProjectableFilterableTable(recBSchema,
158             () -> CloseableIterator.from(Arrays.asList(recb1, recb2).iterator())));
159 
160     SqlParser.Config cfg = SqlParser.configBuilder()
161             .setCaseSensitive(true)
162             .setIdentifierMaxLength(255)
163             .setLex(Lex.JAVA).build();
164     FrameworkConfig config = Frameworks.newConfigBuilder()
165             .parserConfig(cfg)
166             .defaultSchema(schema).build();
167     Planner planner = Frameworks.getPlanner(config);
168     SqlNode s = planner.parse("select a.id, a.name as n1, b.name as n2,"
169             + " b.adate as adate, b.meta as firstKey, b.meta2.key as blaKey"
170             + " from a"
171             + " inner join b on a.id = b.id where b.text like 'bla%' or b.text like 'cucu%'");
172     SqlNode validated = planner.validate(s);
173     RelRoot rel = planner.rel(validated);
174     RelNode plan = rel.project();
175     LOG.debug("exec plan", RelOptUtil.toString(plan));
176     plan = PlannerUtils.pushDownPredicatesAndProjection(plan);
177     LOG.debug("exec plan optimized", RelOptUtil.toString(plan));
178     RelDataType rowType = plan.getRowType();
179     LOG.debug("Return row type: {}", rowType);
180     Schema from = Types.from(rowType);
181     LOG.debug("Return row schema: {}", from);
182     Interpreter interpreter = new Interpreter(new EmbededDataContext(new JavaTypeFactoryImpl(), null), plan);
183     boolean empty = true;
184     for (Object[] row : interpreter) {
185       LOG.debug("RawRow {} with schema {}",  row, from);
186       GenericRecord record = IndexedRecords.fromRecord(from, row);
187       LOG.debug("Row",  record);
188       empty = false;
189     }
190     Assert.assertFalse(empty);
191 
192   }
