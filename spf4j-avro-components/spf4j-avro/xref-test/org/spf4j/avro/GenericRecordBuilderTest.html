52  public class GenericRecordBuilderTest {
53  
54    private static final Logger LOG = LoggerFactory.getLogger(GenericRecordBuilderTest.class);
55  
56    @Test
57    @SuppressFBWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
58    public void testGenericRecordCreation() throws IOException, Exception {
59      String schemaStr = Resources.toString(Resources.getResource("SchemaBuilder.avsc"), StandardCharsets.US_ASCII);
60      Schema schema = new Schema.Parser().parse(schemaStr);
61      try (GenericRecordBuilder builder = new GenericRecordBuilder(schema)) {
62        Class<? extends SpecificRecordBase> clasz = builder.getRecordClass(schema);
63        GenericRecord record = clasz.newInstance();
64        record.put("requiredBoolean", Boolean.TRUE);
65        Schema fieldSchema = schema.getField("optionalRecord").schema().getTypes().get(1);
66        Class<? extends SpecificRecordBase> nestedRecordClass = builder.getRecordClass(fieldSchema);
67        SpecificRecordBase nr = nestedRecordClass.newInstance();
68        nr.put("nestedRequiredBoolean", Boolean.TRUE);
69        record.put("optionalRecord", nr);
70        LOG.debug("Record = {}", record);
71        Assert.assertEquals(nr, record.get("optionalRecord"));
72      }
73    }
74  
75    @Test
76    @SuppressFBWarnings("PRMC_POSSIBLY_REDUNDANT_METHOD_CALLS")
77    public void testMoreGenericRecordCreation() throws InstantiationException, IllegalAccessException {
78      Schema reuse = SchemaBuilder.builder().record("Reuse").fields().requiredString("field").endRecord();
79      Schema record = SchemaBuilder.builder().record("TestRecord").fields()
80              .requiredInt("number")
81              .name("record").type(reuse).noDefault()
82              .endRecord();
83      Schema record2 = SchemaBuilder.builder().record("TestRecord2").fields()
84              .requiredBoolean("isThere")
85              .name("record").type(reuse).noDefault()
86              .endRecord();
87      try (GenericRecordBuilder builder = new GenericRecordBuilder(record, record2)) {
88        Class<? extends SpecificRecordBase> clasz = builder.getRecordClass(record);
89        Class<? extends SpecificRecordBase> clasz2 = builder.getRecordClass(record);
90        Assert.assertSame(clasz, clasz2);
91        SpecificRecordBase myRecord = clasz.newInstance();
92        myRecord.put("number", 35);
93        Assert.assertEquals(35, (int) myRecord.get("number"));
94        Class<? extends SpecificRecordBase> aClass = builder.getRecordClass(record2);
95        GenericRecord r2 = aClass.newInstance();
96        LOG.debug("Record = {}", r2);
97      }
98    }
99  
100   @Test
101   public void testEnumImplementation() {
102     Schema enumSchema = SchemaBuilder.enumeration("MyEnum").namespace("test").symbols("A", "B", "C");
103     try (GenericRecordBuilder builder = new GenericRecordBuilder(enumSchema)) {
104       Class<? extends GenericEnumSymbol> enumClass = builder.getEnumClass(enumSchema);
105       Assert.assertEquals("test.MyEnum", enumClass.getName());
106     }
107   }
