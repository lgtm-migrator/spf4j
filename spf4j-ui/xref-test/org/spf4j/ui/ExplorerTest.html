48   * @author Zoltan Farkas
49   */
50  public class ExplorerTest {
51  
52    private static final Logger LOG = LoggerFactory.getLogger(ExplorerTest.class);
53  
54    private static NoExitSecurityManagerInstaller installNoExitSecurityManager;
55  
56    @BeforeClass
57    public static void setUpOnce() {
58       FailOnThreadViolationRepaintManager.install();
59      installNoExitSecurityManager = NoExitSecurityManagerInstaller.installNoExitSecurityManager();
60    }
61  
62    @AfterClass
63    public static void tearDown() {
64      installNoExitSecurityManager.uninstall();
65    }
66  
67    @Test
68    @SuppressFBWarnings("MDM_THREAD_YIELD") // need to since assertj does not seem to apply things otherwise...
69    public void testExplorer() throws InterruptedException {
70      Assume.assumeTrue(TestUtils.isExecutedInCI());
71      ObservationAssert expectation = TestLoggers.sys().expectUncaughtException(2, TimeUnit.SECONDS,
72              UncaughtExceptionDetail.hasThrowable((Matcher) Matchers.any(ExitException.class)));
73      JFrame tFrame = GuiActionRunner.execute(() -> new Explorer());
74      FrameFixture window = new FrameFixture(tFrame);
75      Settings settings = window.robot().settings();
76      settings.delayBetweenEvents(130);
77      settings.eventPostingDelay(100);
78      LOG.debug("Show window");
79      window.show(); // shows the frame to test
80      LOG.debug("Window is up");
81      Assert.assertTrue(window.isEnabled());
82      LOG.debug("Looking for open file menu item");
83      JMenuItemFixture openFileMenuItem = window.menuItem(new GenericTypeMatcher<JMenuItem>(JMenuItem.class) {
84        @Override
85        protected boolean isMatching(final JMenuItem component) {
86          return "Open".equals(component.getText());
87        }
88      });
89      openFileMenuItem.click();
90      Thread.sleep(100); // without this somehow things are not found.
91      JFileChooserFixture fileChooser = window.fileChooser("openFileDialog");
92      fileChooser.setCurrentDirectory(new File("src/test/resources"));
93      fileChooser.selectFile(new File(
94              "src/test/resources/com.google.common.io.AppendableWriterBenchmark.spf4jAppendable-Throughput.ssdump2")
95              .getAbsoluteFile());
96      fileChooser.approve();
97      Assert.assertNotNull(window.internalFrame(
98              "com.google.common.io.AppendableWriterBenchmark.spf4jAppendable-Throughput.ssdump2"));
99  
100     openFileMenuItem.click();
101     Thread.sleep(100);
102     fileChooser = window.fileChooser("openFileDialog");
103     fileChooser.setCurrentDirectory(new File("src/test/resources"));
104     Thread.sleep(100);
105     fileChooser.selectFile(new File("src/test/resources/"
106                     + "19156@ZMacBookPro.local.tsdb").getAbsoluteFile());
107     fileChooser.approve();
108     Assert.assertNotNull(window.internalFrame(
109             "19156@ZMacBookPro.local.tsdb"));
110 
111     openFileMenuItem.click();
112     Thread.sleep(100);
113     fileChooser = window.fileChooser("openFileDialog");
114     fileChooser.setCurrentDirectory(new File("src/test/resources"));
115     Thread.sleep(100);
116     fileChooser.selectFile(new File("src/test/resources/"
117                     + "test8381720042200787335.tsdb2").getAbsoluteFile());
118     fileChooser.approve();
119     Assert.assertNotNull(window.internalFrame(
120             "test8381720042200787335.tsdb2"));
121     JMenuItemFixture openJsonText = window.menuItem(new GenericTypeMatcher<JMenuItem>(JMenuItem.class) {
122       @Override
123       protected boolean isMatching(final JMenuItem component) {
124         return "From Text/URL".equals(component.getText());
125       }
126     });
127     openJsonText.click();
128     Thread.sleep(100);
129     DialogFixture dialog = window.dialog("fromTextDialog");
130     dialog.textBox("textBox").setText("{\"ROOT@49924@ZMacBookPro-2.local\":10,"
131             + "\"c\":[{\"m3@C2\":2,\"c\":[{\"m2@C2\":2,\"c\":[{\"m1@C2\":2}]}]},{\"m1@C1\":2},"
132             + "{\"m3@C1\":4,\"c\":[{\"m2@C1\":4,\"c\":[{\"m1@C1\":4}]}]},"
133             + "{\"m4@C1\":2,\"c\":[{\"m2@C1\":2,\"c\":[{\"m1@C1\":2}]}]}]}");
134     dialog.button("display").click();
135     Assert.assertNotNull(window.internalFrame("SampleNode Tree"));
136     window.close();
137     window.cleanUp();
138     expectation.assertObservation();
139   }
