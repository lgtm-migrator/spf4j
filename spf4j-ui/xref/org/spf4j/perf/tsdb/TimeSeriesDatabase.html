76  /**
77   * Yet another time series database. Why? because all the other ts databases had various constraints that restrict the
78   * functionality I can add to spf4j.
79   *
80   * Initial Features:
81   *
82   * 1. measurements can be added dynamically anytime to a database. 2. long measurement names. 3. the stored interval is
83   * not known from the beginning. 4. implementation biased towards write performance.
84   *
85   * Format:
86   *
87   * Header: TSDB[version:int][metadata:bytes] Table of Contents: firstTableInfoPtr, lastTableInfoPtr TableInfo:
88   * DataFragment: ...... ...... TableInfo: DataFragment: EOF
89   *
90   * @deprecated please use org.spf4j.tsdb2
91   * @author zoly
92   */

118   public TimeSeriesDatabase(final File pathToDatabaseFile, final boolean isWrite, final byte... metaData)
119           throws IOException {
120     file = new RandomAccessFile(pathToDatabaseFile, isWrite ? "rw" : "r");
121     // uniques per process string for sync purposes.
122     this.path = INTERNER.intern(pathToDatabaseFile.getCanonicalPath());
123     tables = new ConcurrentHashMap<>();
124     writeDataFragments = new HashMap<>();
125     // read or create header
126     synchronized (path) {
127       this.ch = file.getChannel();
128       FileLock lock;
129       if (isWrite) {
130         lock = ch.lock();
131       } else {
132         lock = ch.lock(0, Long.MAX_VALUE, true);
133       }
134       try {
135         if (file.length() == 0) {
136           this.header = new Header(VERSION, metaData);
137           this.header.writeTo(file);
138           this.toc = new TableOfContents(file.getFilePointer());
139           this.toc.writeTo(file);
140         } else {
141           this.header = new Header(file);
142           this.toc = new TableOfContents(file);
143         }
144       } catch (IOException | RuntimeException e) {
145         try {
146           lock.release();
147           throw e;
148         } catch (IOException ex) {
149           ex.addSuppressed(e);
150           throw ex;
151         }
152       }
153       lock.release();
154       lock = ch.lock(0, Long.MAX_VALUE, true);
155       try {
156         readTableInfos();
157       } catch (IOException | RuntimeException e) {
158         try {
159           lock.release();
160           throw e;
161         } catch (IOException ex) {
162           ex.addSuppressed(e);
163           throw ex;
164         }
165       }
166       lock.release();
167     }
168   }

170   public void reReadTableInfos() throws IOException {
171     synchronized (path) {
172       FileLock lock = ch.lock(0, Long.MAX_VALUE, true);
173       try {
174         toc = new TableOfContents(file, toc.getLocation()); // reread toc
175         readTableInfos();
176       } catch (IOException | RuntimeException e) {
177         try {
178           lock.release();
179           throw e;
180         } catch (IOException ex) {
181           ex.addSuppressed(e);
182           throw ex;
183         }
184       }
185       lock.release();
186     }
187   }

189   private void readTableInfos() throws IOException {
190     final long firstColumnInfo = toc.getFirstTableInfoPtr();
191     if (firstColumnInfo > 0) {
192       file.seek(firstColumnInfo);
193       TSTable colInfo = new TSTable(file);
194       tables.put(colInfo.getTableName(), colInfo);
195       lastTableInfo = colInfo;
196       while (colInfo.getNextTSTable() > 0) {
197         file.seek(colInfo.getNextTSTable());
198         colInfo = new TSTable(file);
199         tables.put(colInfo.getTableName(), colInfo);
200         lastTableInfo = colInfo;
201       }
202     }
203   }
204 
205   private void readLastTableInfo() throws IOException {
206     toc = new TableOfContents(file, toc.getLocation()); // reread toc
207     if (toc.getLastTableInfoPtr() == 0) {
208       return;
209     }
210     lastTableInfo = new TSTable(file, toc.getLastTableInfoPtr()); // update last table info
211   }

238   public void addTSTable(final String tableName,
239           final byte[] tableMetaData, final int sampleTime, final String[] columnNames,
240           final byte[][] columnMetaData) throws IOException {
241     synchronized (path) {
242       if (hasTSTable(tableName)) {
243         throw new IllegalArgumentException("group already exists " + tableName);
244       }
245       flush();
246       FileLock lock = ch.lock();
247       TSTable colInfo;
248       try {
249         readLastTableInfo();
250         //write column information at the end of the file.
251         file.seek(file.length());
252         colInfo = new TSTable(tableName, tableMetaData, columnNames,
253                 columnMetaData, sampleTime, file.getFilePointer());
254         colInfo.writeTo(file);
255         //update refferences to this new TableInfo.
256         if (lastTableInfo != null) {
257           lastTableInfo.setNextColumnInfo(colInfo.getLocation(), file);
258         } else {
259           toc.setFirstTableInfo(colInfo.getLocation(), file);
260         }
261         toc.setLastTableInfo(colInfo.getLocation(), file);
262       } catch (IOException | RuntimeException e) {
263         try {
264           lock.release();
265           throw e;
266         } catch (IOException ex) {
267           ex.addSuppressed(e);
268           throw ex;
269         }
270       }
271       lock.release();
272 
273       lastTableInfo = colInfo;
274       tables.put(tableName, colInfo);
275     }
276   }

278   public void write(final long time, final String tableName, final long[] values) {
279     if (!hasTSTable(tableName)) {
280       throw new IllegalArgumentException("Unknown table name" + tableName);
281     }
282     synchronized (writeDataFragments) {
283       DataFragment writeDataFragment = writeDataFragments.get(tableName);
284       if (writeDataFragment == null) {
285         writeDataFragment = new DataFragment(time);
286         writeDataFragments.put(tableName, writeDataFragment);
287       }
288       writeDataFragment.addData(time, values);
289     }
290   }

292   public void flush() throws IOException {
293     synchronized (path) {
294       List<Map.Entry<String, DataFragment>> lwriteDataFragments;
295       synchronized (writeDataFragments) {
296         if (writeDataFragments.isEmpty()) {
297           return;
298         }
299         lwriteDataFragments = new ArrayList<>(writeDataFragments.entrySet());
300         writeDataFragments.clear();
301       }
302       FileLock lock = ch.lock();
303       try {
304         for (Map.Entry<String, DataFragment> entry : lwriteDataFragments) {
305           DataFragment writeDataFragment = entry.getValue();
306           String groupName = entry.getKey();
307           file.seek(file.length());
308           writeDataFragment.setLocation(file.getFilePointer());
309           writeDataFragment.writeTo(file);
310           TSTable colInfo = tables.get(groupName);
311           colInfo = new TSTable(file, colInfo.getLocation()); // reread colInfo
312           tables.put(groupName, colInfo); // update colInfo
313           final long lastDataFragment = colInfo.getLastDataFragment();
314           final long location = writeDataFragment.getLocation();
315           if (lastDataFragment != 0) {
316             DataFragment.setNextDataFragment(lastDataFragment, location, file);
317           } else {
318             colInfo.setFirstDataFragment(location, file);
319           }
320           colInfo.setLastDataFragment(location, file);
321         }
322         sync();
323       } catch (IOException | RuntimeException e) {
324         try {
325           lock.release();
326           throw e;
327         } catch (IOException ex) {
328           ex.addSuppressed(e);
329           throw ex;
330         }
331       }
332       lock.release();
333     }
334   }

398   /**
399    * Read measurements from table.
400    *
401    * @param tableName
402    * @param startTime start time including
403    * @param endTime end time including
404    * @return
405    * @throws IOException
406    */
407   private TimeSeries read(
408           final long startTime, final long endTime,
409           final long startAtFragment, final long endAtFragment,
410           final boolean skipFirst)
411           throws IOException {
412     synchronized (path) {
413       TLongArrayList timeStamps = new TLongArrayList();
414       List<long[]> data = new ArrayList<>();
415       if (startAtFragment > 0) {
416         FileLock lock = ch.lock(0, Long.MAX_VALUE, true);
417         try {
418           DataFragment frag;
419           long nextFragmentLocation = startAtFragment;
420           boolean last = false;
421           boolean psFirst = skipFirst;
422           do {
423             if (nextFragmentLocation == endAtFragment) {
424               last = true;
425             }
426             file.seek(nextFragmentLocation);
427             frag = new DataFragment(file);
428             if (psFirst) {
429               psFirst = false;
430             } else {
431               long fragStartTime = frag.getStartTimeMillis();
432               if (fragStartTime >= startTime) {
433                 TIntArrayList fragTimestamps = frag.getTimestamps();
434                 int nr = 0;
435                 for (int i = 0; i < fragTimestamps.size(); i++) {
436                   long ts = fragStartTime + fragTimestamps.get(i);
437                   if (ts <= endTime) {
438                     timeStamps.add(ts);
439                     nr++;
440                   } else {
441                     break;
442                   }
443                 }
444                 int i = 0;
445                 for (long[] d : frag.getData()) {
446                   if (i < nr) {
447                     data.add(d);
448                   } else {
449                     break;
450                   }
451                   nr++;
452                 }
453                 if (fragTimestamps.size() > nr) {
454                   break;
455                 }
456               }
457             }
458             nextFragmentLocation = frag.getNextDataFragment();
459           } while (nextFragmentLocation > 0 && !last);
460         } catch (IOException | RuntimeException e) {
461           try {
462             lock.release();
463             throw e;
464           } catch (IOException ex) {
465             ex.addSuppressed(e);
466             throw ex;
467           }
468         }
469         lock.release();
470       }
471       return new TimeSeries(timeStamps.toArray(), data.toArray(new long[data.size()][]));
472     }
473   }

535             + info.getTableName() + ", sampleTime " + info.getSampleTime() + "ms, generated by spf4j");

538   public static JFreeChart createMinMaxAvgJFreeChart(final TimeSeries data, final TSTable info) {
539     long[][] vals = data.getValues();
540     double[] min = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("min"));
541     double[] max = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("max"));
542     final int totalColumnIndex = info.getColumnIndex("total");
543     double[] total = Arrays.getColumnAsDoubles(vals, totalColumnIndex);
544     double[] count = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("count"));
545     for (int i = 0; i < count.length; i++) {
546       if (count[i] == 0) {
547         min[i] = 0;
548         max[i] = 0;
549       }
550     }
551     long[] timestamps = data.getTimeStamps();
552     return Charts.createTimeSeriesJFreeChart("Min,Max,Avg chart for "
553             + info.getTableName() + ", sampleTime " + info.getSampleTime() + "ms, generated by spf4j", timestamps,
554             new String[]{"min", "max", "avg"}, new String(info.getColumnMetaData()[totalColumnIndex],
555                     StandardCharsets.UTF_8), new double[][]{min, max, Arrays.divide(total, count)});
556   }

567   public static List<JFreeChart> createJFreeCharts(final TimeSeries data, final TSTable info) {
568     long[][] vals = data.getValues();
569     Map<String, Pair<List<String>, List<double[]>>> measurementsByUom = new HashMap<>(4);
570     String[] columnMetaData = info.getColumnMetaDataAsStrings();
571     int columnNumber = info.getColumnNumber();
572     for (int i = 0; i < columnNumber; i++) {
573       String uom = columnMetaData[i];
574       Pair<List<String>, List<double[]>> meas = measurementsByUom.get(uom);
575       if (meas == null) {
576         meas = Pair.of((List<String>) new ArrayList<String>(columnNumber),
577                 (List<double[]>) new ArrayList<double[]>(columnNumber));
578         measurementsByUom.put(uom, meas);
579       }
580       meas.getFirst().add(info.getColumnName(i));
581       meas.getSecond().add(Arrays.getColumnAsDoubles(vals, i));
582     }
583     long[] timestamps = data.getTimeStamps();
584     List<JFreeChart> result = new ArrayList<>(measurementsByUom.size());
585     for (Map.Entry<String, Pair<List<String>, List<double[]>>> entry : measurementsByUom.entrySet()) {
586       Pair<List<String>, List<double[]>> p = entry.getValue();
587       final List<String> measurementNames = p.getFirst();
588       final List<double[]> measurements = p.getSecond();
589       result.add(Charts.createTimeSeriesJFreeChart("chart for "
590               + info.getTableName() + ", sampleTime " + info.getSampleTime()
591               + " ms, generated by spf4j", timestamps,
592               measurementNames.toArray(new String[measurementNames.size()]), entry.getKey(),
593               measurements.toArray(new double[measurements.size()][])));
594     }
595     return result;
596   }

602   public void writeCsvTable(final String tableName, final File output) throws IOException {
603     TSTable table = getTSTable(tableName);
604     TimeSeries data = readAll(tableName);
605     DateTimeFormatter formatter = DateTimeFormats.TS_FORMAT;
606     try (Writer writer = new BufferedWriter(new OutputStreamWriter(
607             Files.newOutputStream(output.toPath()), StandardCharsets.UTF_8))) {
608       Csv.writeCsvElement("timestamp", writer);
609       for (String colName : table.getColumnNames()) {
610         writer.append(',');
611         Csv.writeCsvElement(colName, writer);
612       }
613       writer.write('\n');
614       long[] timestamps = data.getTimeStamps();
615       long[][] values = data.getValues();
616       for (int i = 0; i < timestamps.length; i++) {
617         Csv.writeCsvElement(formatter.format(Instant.ofEpochMilli(timestamps[i])), writer);
618         for (long val : values[i]) {
619           writer.append(',');
620           Csv.writeCsvElement(Long.toString(val), writer);
621         }
622         writer.write('\n');
623       }
624     }
625   }

627   public void writeCsvTables(final List<String> tableNames, final File output) throws IOException {
628     DateTimeFormatter formatter = DateTimeFormats.TS_FORMAT;
629     try (Writer writer = new BufferedWriter(new OutputStreamWriter(
630             Files.newOutputStream(output.toPath()), StandardCharsets.UTF_8))) {
631       String firstTable = tableNames.get(0);
632       TSTable table = getTSTable(firstTable);
633       Csv.writeCsvElement("table", writer);
634       writer.append(',');
635       Csv.writeCsvElement("timestamp", writer);
636       for (String colName : table.getColumnNames()) {
637         writer.append(',');
638         Csv.writeCsvElement(colName, writer);
639       }
640       writer.write('\n');
641 
642       for (String tableName : tableNames) {
643         TimeSeries data = readAll(tableName);
644         long[] timestamps = data.getTimeStamps();
645         long[][] values = data.getValues();
646         for (int i = 0; i < timestamps.length; i++) {
647           Csv.writeCsvElement(tableName, writer);
648           writer.append(',');
649           Csv.writeCsvElement(formatter.format(Instant.ofEpochMilli(timestamps[i])), writer);
650           for (long val : values[i]) {
651             writer.append(',');
652             Csv.writeCsvElement(Long.toString(val), writer);
653           }
654           writer.write('\n');
655         }
656       }
657     }
658   }

665   @SuppressFBWarnings("MDM_THREAD_YIELD")
666   public void tail(final long pollMillis, final long from, final TSDataHandler handler)
667           throws IOException {
668     Map<String, TSTable> lastState = new HashMap<>();
669     long lastSize = 0;
670     while (!Thread.currentThread().isInterrupted() && !handler.finish()) {
671       long currSize = this.file.length();
672       if (currSize > lastSize) {
673         // see if we have new Tables;
674         reReadTableInfos();
675         Map<String, TSTable> currState = getTsTables();
676         for (String tableName : Sets.difference(currState.keySet(), lastState.keySet())) {
677           handler.newTable(tableName, currState.get(tableName).getColumnNames());
678         }
679         for (TSTable table : currState.values()) {
680           final String tableName = table.getTableName();
681           TSTable prevTableState = lastState.get(tableName);
682           final long currLastDataFragment = table.getLastDataFragment();
683 
684           if (prevTableState == null) {
685             long lastDataFragment = table.getFirstDataFragment();
686             if (lastDataFragment > 0) {
687               TimeSeries data = read(from, Long.MAX_VALUE, lastDataFragment, currLastDataFragment,
688                       false);
689               handler.newData(tableName, data);
690             }
691           } else {
692             long lastDataFragment = prevTableState.getLastDataFragment();
693             if (lastDataFragment == 0) {
694               lastDataFragment = table.getFirstDataFragment();
695               if (lastDataFragment > 0) {
696                 TimeSeries data = read(from, Long.MAX_VALUE, lastDataFragment, currLastDataFragment,
697                         false);
698                 handler.newData(tableName, data);
699               }
700             } else if (currLastDataFragment > lastDataFragment) {
701               TimeSeries data = read(from, Long.MAX_VALUE, lastDataFragment, currLastDataFragment,
702                       true);
703               handler.newData(tableName, data);
704             }
705           }
706         }
707         lastState = currState;
708       }
709       lastSize = currSize;
710       try {
711         Thread.sleep(pollMillis);
712       } catch (InterruptedException ex) {
713         Thread.currentThread().interrupt();
714         break;
715       }
716     }
717   }

723   /**
724    * Quantized recorders will have min, max avg charts and distribution charts generated. Counting recorders will have
725    * simple charts generated.
726    *
727    * @param startTimeMillis
728    * @param endTimeMillis
729    * @param width
730    * @param height
731    * @return
732    * @throws IOException
733    */
734   @JmxExport(value = "generateChartsInterval",
735           description = "generate charts for all measurements in specified interval")
736   public List<String> generateCharts(final long startTimeMillis, final long endTimeMillis,
737           final int width, final int height) throws IOException {
738     try {
739       this.flush();
740       Collection<TSTable> columnsInfo = this.getTSTables();
741       List<String> result = new ArrayList<>(16);
742       for (TSTable info : columnsInfo) {
743         TimeSeries data = this.read(info.getTableName(), startTimeMillis, endTimeMillis);
744         if (data.getTimeStamps().length > 0) {
745           if (canGenerateMinMaxAvgCount(info)) {
746             result.add(generateMinMaxAvgCountChart(info, data, width, height));
747           }
748           if (canGenerateHeatChart(info)) {
749             result.add(generateHeatChart(info, data, width, height));
750           }
751         }
752       }
753       Multimap<String, TSTable> counters = getCounters(columnsInfo);
754       Map<String, Collection<TSTable>> asMap = counters.asMap();
755       for (Map.Entry<String, Collection<TSTable>> entry : asMap.entrySet()) {
756         Collection<TSTable> ltables = entry.getValue();
757         int l = ltables.size();
758         long[][] timestamps = new long[l][];
759         double[][] cdata = new double[l][];
760         double[][] cdata2 = new double[l][];
761         int i = 0;
762         String[] measurementNames = new String[cdata.length];
763         String[] measurementNames2 = new String[cdata2.length];
764         String uom1 = "count";
765         String uom2 = "";
766         for (TSTable colInfo : ltables) {
767           TimeSeries data = this.read(colInfo.getTableName(), startTimeMillis, endTimeMillis);
768           timestamps[i] = data.getTimeStamps();
769           final long[][] values = data.getValues();
770           cdata[i] = Arrays.getColumnAsDoubles(values, colInfo.getColumnIndex("count"));
771           cdata2[i] = Arrays.getColumnAsDoubles(values, colInfo.getColumnIndex("total"));
772           measurementNames[i] = colInfo.getTableName() + ".count";
773           measurementNames2[i] = colInfo.getTableName() + ".total";
774           uom2 = new String(colInfo.getTableMetaData(), StandardCharsets.UTF_8);
775           i++;
776         }
777         result.add(generateCountChart(entry.getKey(), timestamps, measurementNames,
778                 measurementNames2, uom1, uom2, cdata, cdata2, width, height));
779       }
780       return result;
781     } catch (IOException | RuntimeException ex) {
782       throw ex;
783     }
784   }

786   @SuppressFBWarnings("RV_RETURN_VALUE_IGNORED")
787   private static Multimap<String, TSTable> getCounters(final Collection<TSTable> columnInfos) {
788     Multimap<String, TSTable> result = HashMultimap.create();
789     for (TSTable info : columnInfos) {
790       if (isCounterOnly(info)) {
791         String groupName = info.getTableName();
792         if (groupName.startsWith("(")) {
793           int cidx = groupName.indexOf(',');
794           if (cidx > 0) {
795             groupName = groupName.substring(1, cidx);
796           }
797         }
798         result.put(groupName, info);
799       }
800     }
801     return result;
802   }

831   private String generateMinMaxAvgCountChart(
832           final TSTable info, final TimeSeries data,
833           final int width, final int height) throws IOException {
834     long[][] vals = data.getValues();
835     double[] min = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("min"));
836     double[] max = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("max"));
837     double[] total = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("total"));
838     double[] count = Arrays.getColumnAsDoubles(vals, info.getColumnIndex("count"));
839     for (int i = 0; i < count.length; i++) {
840       if (count[i] == 0) {
841         min[i] = 0;
842         max[i] = 0;
843       }
844     }
845     long[] timestamps = data.getTimeStamps();
846     BufferedImage combined = Charts.createMinMaxAvgCountImg("Measurements for "
847             + info.getTableName() + " generated by spf4j",
848             timestamps, min, max, total, count,
849             new String(info.getTableMetaData(), StandardCharsets.UTF_8), width, height);
850     File dbFile = new File(this.getDBFilePath());
851     File graphicFile = File.createTempFile(dbFile.getName() + '_' + fixName(info.getTableName()), ".mmac.png",
852             dbFile.getParentFile());
853     ImageIO.write(combined, "png", graphicFile);
854     return graphicFile.getPath();
855   }

884   private static String fixName(final String name) {
885     final int length = name.length();
886     StringBuilder result = new StringBuilder(length);
887     for (int i = 0; i < length; i++) {
888       char c = name.charAt(i);
889       if (Character.isJavaIdentifierPart(c)) {
890         result.append(c);
891       }
892     }
893     return result.toString();
894   }
