74      @SuppressForbiden
75      public TSDBViewJInternalFrame(final File databaseFile) throws IOException {
76          super(databaseFile.getAbsolutePath());
77          setName(databaseFile.getName());
78          initComponents();
79          tsDb = new TimeSeriesDatabase(databaseFile, null);
80          Collection<TSTable> columnsInfo = tsDb.getTSTables();
81          Map<String, DefaultMutableTreeNode> gNodes = new HashMap<>();
82          DefaultMutableTreeNode root = new DefaultMutableTreeNode(databaseFile);
83          long startDateMillis = System.currentTimeMillis();
84          for (TSTable info : columnsInfo) {
85              String groupName = info.getTableName();
86              long tableStart = tsDb.readStartDate(groupName);
87              if (tableStart < startDateMillis) {
88                  startDateMillis = tableStart;
89              }
90              Pair<String, String> pair = Pair.from(groupName);
91              if (pair == null) {
92                  DefaultMutableTreeNode child = new DefaultMutableTreeNode(groupName);
93                  for (String colName : info.getColumnNames()) {
94                      child.add(new DefaultMutableTreeNode(colName));
95                  }
96                  root.add(child);
97              } else {
98                  groupName = pair.getFirst();
99                  DefaultMutableTreeNode gNode = gNodes.get(groupName);
100                 if (gNode == null) {
101                     gNode = new DefaultMutableTreeNode(groupName);
102                     gNodes.put(groupName, gNode);
103                     root.add(gNode);
104                 }
105                 DefaultMutableTreeNode child = new DefaultMutableTreeNode(pair.getSecond());
106                 for (String colName : info.getColumnNames()) {
107                     child.add(new DefaultMutableTreeNode(colName));
108                 }
109                 gNode.add(child);
110             }
111         }
112         measurementTree.setModel(new DefaultTreeModel(root));
113         measurementTree.setVisible(true);
114         this.startDate.setValue(new Date(startDateMillis));
115     }

255             JFileChooser chooser = new JFileChooser();

259                 File file = chooser.getSelectedFile();
260                 try {
261                     tsDb.writeCsvTables(selectedTables, file);
262                 } catch (IOException ex) {
263                     throw new RuntimeException(ex);
264                 }
265             }
266         }

282     @edu.umd.cs.findbugs.annotations.SuppressWarnings("CLI_CONSTANT_LIST_INDEX")
283     private static List<String> getSelectedTables(@Nullable final TreePath[] selectionPaths) {
284         if (selectionPaths == null) {
285             return Collections.EMPTY_LIST;
286         }
287         List<String> result = new ArrayList<>();
288         for (TreePath path : selectionPaths) {
289             Object[] pathArr = path.getPath();
290             if (pathArr.length < 2) {
291                 continue;
292             }
293             DefaultMutableTreeNode colNode = (DefaultMutableTreeNode) pathArr[1];
294             int depth = colNode.getDepth();
295             String tableName;
296             if (depth == 1) {
297                 result.add((String) colNode.getUserObject());
298             } else {
299                 Enumeration childEnum = colNode.children();
300                 while (childEnum.hasMoreElements()) {
301                     DefaultMutableTreeNode child = (DefaultMutableTreeNode) childEnum.nextElement();
302                     tableName =
303                             Pair.of((String) colNode.getUserObject(), (String) child.getUserObject()).toString();
304                     result.add(tableName);
305                 }
306             }
307         }
308         return result;
309     }

311     private void addChartToPanel(final String tableName, final JPanel content) throws IOException {
312         TSTable info = tsDb.getTSTable(tableName);
313          long startTime = ((Date) startDate.getValue()).getTime();
314          long endTime = ((Date) endDate.getValue()).getTime();
315         if (TimeSeriesDatabase.canGenerateHeatChart(info)) {
316             JFreeChart chart = tsDb.createHeatJFreeChart(info.getTableName(),
317                     startTime, endTime);
318             ChartPanel pannel = new ChartPanel(chart);
319             pannel.setPreferredSize(new Dimension(600, 800));
320             pannel.setDomainZoomable(false);
321             pannel.setMouseZoomable(false);
322             pannel.setRangeZoomable(false);
323             pannel.setZoomAroundAnchor(false);
324             pannel.setZoomInFactor(1);
325             pannel.setZoomOutFactor(1);
326             content.add(pannel);
327         }
328         if (TimeSeriesDatabase.canGenerateMinMaxAvgCount(info)) {
329             JFreeChart chart = tsDb.createMinMaxAvgJFreeChart(info.getTableName(),
330                     startTime, endTime);
331             ChartPanel pannel = new ChartPanel(chart);
332             pannel.setPreferredSize(new Dimension(600, 600));
333             content.add(pannel);
334 
335         }
336         if (TimeSeriesDatabase.canGenerateCount(info)) {
337             JFreeChart chart = tsDb.createCountJFreeChart(info.getTableName(),
338                     startTime, endTime);
339             ChartPanel pannel = new ChartPanel(chart);
340             pannel.setPreferredSize(new Dimension(600, 600));
341             content.add(pannel);
342         } else {
343             List<JFreeChart> createJFreeCharts = tsDb.createJFreeCharts(info.getTableName(), startTime, endTime);
344             for (JFreeChart chart : createJFreeCharts) {
345                 ChartPanel pannel = new ChartPanel(chart);
346                 pannel.setPreferredSize(new Dimension(600, 600));
347                 content.add(pannel);
348             }
349         }
