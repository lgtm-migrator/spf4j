79      @SuppressForbiden
80      public TSDB2ViewJInternalFrame(final File tsDb) throws IOException {
81          super(tsDb.getPath());
82          setName(tsDb.getName());
83          this.tsDb = tsDb;
84          initComponents();
85          Multimap<String, TableDefEx> columnsInfo = TSDBQuery.getAllTablesWithDataRanges(tsDb);
86          Map<String, DefaultMutableTreeNode> gNodes = new HashMap<>();
87          DefaultMutableTreeNode root = new DefaultMutableTreeNode(tsDb.getName());
88          long startDateMillis = System.currentTimeMillis();
89          for (Map.Entry<String, Collection<TableDefEx>> info : columnsInfo.asMap().entrySet()) {
90              String groupName = info.getKey();
91              final Collection<TableDefEx> defs = info.getValue();
92              for (TableDefEx tde : defs) {
93                  long tableStart = tde.getStartTime();
94                  if (tableStart < startDateMillis) {
95                      startDateMillis = tableStart;
96                  }
97              }
98              Pair<String, String> pair = Pair.from(groupName);
99              if (pair == null) {
100                 DefaultMutableTreeNode child = new DefaultMutableTreeNode(groupName);
101                 for (ColumnDef colDef : defs.iterator().next().getTableDef().getColumns()) {
102                     child.add(new DefaultMutableTreeNode(colDef.getName()));
103                 }
104                 root.add(child);
105             } else {
106                 groupName = pair.getFirst();
107                 DefaultMutableTreeNode gNode = gNodes.get(groupName);
108                 if (gNode == null) {
109                     gNode = new DefaultMutableTreeNode(groupName);
110                     gNodes.put(groupName, gNode);
111                     root.add(gNode);
112                 }
113                 DefaultMutableTreeNode child = new DefaultMutableTreeNode(pair.getSecond());
114                 for (ColumnDef colDef : defs.iterator().next().getTableDef().getColumns()) {
115                     child.add(new DefaultMutableTreeNode(colDef.getName()));
116                 }
117                 gNode.add(child);
118             }
119         }
120         measurementTree.setModel(new DefaultTreeModel(root));
121         measurementTree.setVisible(true);
122         this.startDate.setValue(new Date(startDateMillis));
123     }

258     @SuppressFBWarnings("UP_UNUSED_PARAMETER")
259     private void jButton2ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton2ActionPerformed
260         TreePath[] selectionPaths = measurementTree.getSelectionPaths();
261         Set<String> selectedTables = getSelectedTables(selectionPaths);
262         if (!selectedTables.isEmpty()) {
263             JFileChooser chooser = new JFileChooser();
264             chooser.setDialogType(JFileChooser.SAVE_DIALOG);
265             int returnVal = chooser.showOpenDialog(this);
266             if (returnVal == JFileChooser.APPROVE_OPTION) {
267                 File file = chooser.getSelectedFile();
268                 try {
269                     TSDBQuery.writeCsvTables(tsDb, selectedTables, file);
270                 } catch (IOException ex) {
271                     throw new RuntimeException(ex);
272                 }
273             }
274         }

290     @SuppressFBWarnings("CLI_CONSTANT_LIST_INDEX")
291     private static Set<String> getSelectedTables(@Nullable final TreePath[] selectionPaths) {
292         if (selectionPaths == null) {
293             return Collections.EMPTY_SET;
294         }
295         Set<String> result = new HashSet<>();
296         for (TreePath path : selectionPaths) {
297             Object[] pathArr = path.getPath();
298             if (pathArr.length < 2) {
299                 continue;
300             }
301             DefaultMutableTreeNode colNode = (DefaultMutableTreeNode) pathArr[1];
302             int depth = colNode.getDepth();
303             String tableName;
304             if (depth == 1) {
305                 result.add((String) colNode.getUserObject());
306             } else {
307                 Enumeration childEnum = colNode.children();
308                 while (childEnum.hasMoreElements()) {
309                     DefaultMutableTreeNode child = (DefaultMutableTreeNode) childEnum.nextElement();
310                     tableName =
311                             Pair.of((String) colNode.getUserObject(), (String) child.getUserObject()).toString();
312                     result.add(tableName);
313                 }
314             }
315         }
316         return result;
317     }

319     private void addChartToPanel(final String tableName, final JPanel content) throws IOException {
320         List<TableDef> info =  TSDBQuery.getTableDef(tsDb, tableName);
321         long startTime = ((Date) startDate.getValue()).getTime();
322         long endTime = ((Date) endDate.getValue()).getTime();
323         if (Charts.canGenerateHeatChart(info.get(0))) {
324             JFreeChart chart = Charts.createHeatJFreeChart(tsDb, info,
325                     startTime, endTime);
326             ChartPanel pannel = new ChartPanel(chart);
327             pannel.setPreferredSize(new Dimension(600, 800));
328             pannel.setDomainZoomable(false);
329             pannel.setMouseZoomable(false);
330             pannel.setRangeZoomable(false);
331             pannel.setZoomAroundAnchor(false);
332             pannel.setZoomInFactor(1);
333             pannel.setZoomOutFactor(1);
334             content.add(pannel);
335         }
336         if (Charts.canGenerateMinMaxAvgCount(info.get(0))) {
337             JFreeChart chart = Charts.createMinMaxAvgJFreeChart(tsDb, info,
338                     startTime, endTime);
339             ChartPanel pannel = new ChartPanel(chart);
340             pannel.setPreferredSize(new Dimension(600, 600));
341             content.add(pannel);
342 
343         }
344         if (Charts.canGenerateCount(info.get(0))) {
345             JFreeChart chart = Charts.createCountJFreeChart(tsDb, info,
346                     startTime, endTime);
347             ChartPanel pannel = new ChartPanel(chart);
348             pannel.setPreferredSize(new Dimension(600, 600));
349             content.add(pannel);
350         } else {
351             List<JFreeChart> createJFreeCharts = Charts.createJFreeCharts(tsDb, info, startTime, endTime);
352             for (JFreeChart chart : createJFreeCharts) {
353                 ChartPanel pannel = new ChartPanel(chart);
354                 pannel.setPreferredSize(new Dimension(600, 600));
355                 content.add(pannel);
356             }
357         }
